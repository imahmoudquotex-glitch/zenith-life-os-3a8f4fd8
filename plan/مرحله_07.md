CREATE OR REPLACE FUNCTION reorder_block(p_block_id TEXT, p_new_position DOUBLE PRECISION)
RETURNS VOID LANGUAGE plpgsql SECURITY DEFINER AS $$
BEGIN
  UPDATE blocks SET position = p_new_position, version = version + 1
  WHERE id = p_block_id AND workspace_id = current_workspace_id() AND is_deleted = false;
  -- renormalize on collision
END $$;

-- soft_delete_block: recursive cascade
CREATE OR REPLACE FUNCTION soft_delete_block(p_block_id TEXT)
RETURNS VOID LANGUAGE plpgsql SECURITY DEFINER AS $$
BEGIN
  WITH RECURSIVE children AS (
    SELECT id FROM blocks WHERE id = p_block_id
    UNION ALL
    SELECT b.id FROM blocks b JOIN children c ON b.parent_block_id = c.id
  )
  UPDATE blocks SET is_deleted = true, deleted_at = now(), version = version + 1
  WHERE id IN (SELECT id FROM children) AND workspace_id = current_workspace_id();
END $$;

-- synced_block_refs cycle prevention trigger
CREATE OR REPLACE FUNCTION assert_no_synced_cycle() RETURNS TRIGGER LANGUAGE plpgsql AS $$
DECLARE v INT;
BEGIN
  WITH RECURSIVE chain AS (
    SELECT source_block_id FROM synced_block_refs WHERE block_id = NEW.source_block_id
    UNION ALL
    SELECT s.source_block_id FROM synced_block_refs s JOIN chain c ON s.block_id = c.source_block_id
  )
  SELECT count(*) INTO v FROM chain WHERE source_block_id = NEW.block_id;
  IF v > 0 THEN RAISE EXCEPTION 'synced_block_cycle'; END IF;
  RETURN NEW;
END $$;
