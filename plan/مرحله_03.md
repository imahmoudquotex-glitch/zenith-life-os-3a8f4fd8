```sql
CREATE OR REPLACE FUNCTION audit_row_change() RETURNS TRIGGER AS $$
DECLARE v_diff JSONB;
BEGIN
  IF TG_OP = 'UPDATE' THEN
    SELECT jsonb_object_agg(key, value) INTO v_diff
    FROM jsonb_each(to_jsonb(NEW)) WHERE to_jsonb(OLD)->key IS DISTINCT FROM value;

  RETURN NEW;
END $$ LANGUAGE plpgsql;
```

### Base Repository Contract

```tsx
// packages/repo/src/base-repo.ts
export abstract class BaseRepo<TEntity> {
  protected abstract table: string
  constructor(protected readonly db: DbClient) {}

  async findById(id: string, workspaceId: string): Promise<TEntity | null> {
    return this.db.oneOrNone(
      `SELECT * FROM ${this.table}
       WHERE id = $1 AND workspace_id = $2 AND is_deleted = false`,
      [id, workspaceId],
    )
  }

  async softDelete(id: string, workspaceId: string, actorId: string) {
    return this.db.none(
      `UPDATE ${this.table}
       SET is_deleted = true, deleted_at = now(), last_edited_by_user_id = $3
       WHERE id = $1 AND workspace_id = $2`,
      [id, workspaceId, actorId],
    )
  }
}
```

### Atomic AI Quota RPC

```sql
CREATE OR REPLACE FUNCTION reserve_ai_usage(
  p_workspace_id TEXT,
  p_user_id TEXT,
  p_idempotency_key TEXT,
  p_estimated_tokens INT
) RETURNS TEXT LANGUAGE plpgsql SECURITY DEFINER SET search_path = public, pg_temp AS $$
DECLARE v_reservation_id TEXT := generate_ulid();
DECLARE v_used INT;
BEGIN
  PERFORM 1 FROM ai_usage_events
  WHERE workspace_id = p_workspace_id AND idempotency_key = p_idempotency_key
  FOR UPDATE;
  IF FOUND THEN
    SELECT reservation_id INTO v_reservation_id FROM ai_usage_events
    WHERE workspace_id = p_workspace_id AND idempotency_key = p_idempotency_key;
    RETURN v_reservation_id;
  END IF;

  SELECT COALESCE(SUM(tokens_reserved - tokens_refunded), 0)
  INTO v_used
  FROM ai_usage_events
  WHERE workspace_id = p_workspace_id
    AND created_at >= date_trunc('day', now());

  IF v_used + p_estimated_tokens > 1000000 THEN
    RAISE EXCEPTION 'AI_QUOTA_EXCEEDED';
  END IF;

  INSERT INTO ai_usage_events(id, reservation_id, workspace_id, user_id, idempotency_key, tokens_reserved)
  VALUES (generate_ulid(), v_reservation_id, p_workspace_id, p_user_id, p_idempotency_key, p_estimated_tokens);
  RETURN v_reservation_id;
END $$;
```

```sql
CREATE OR REPLACE FUNCTION public.set_updated_at()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
BEGIN NEW.updated_at = now(); RETURN NEW; END $$;

CREATE OR REPLACE FUNCTION public.local_day(p_user TEXT)
RETURNS DATE LANGUAGE plpgsql SECURITY DEFINER SET search_path = public, pg_temp AS $$
DECLARE v_tz TEXT;
BEGIN
  SELECT COALESCE(timezone, 'UTC') INTO v_tz FROM public.users WHERE id = p_user;
  RETURN (now() AT TIME ZONE v_tz)::DATE;
END $$;

CREATE OR REPLACE FUNCTION public.assert_owner(p_user TEXT)
RETURNS VOID LANGUAGE plpgsql AS $$
BEGIN
  IF p_user IS NULL OR p_user <> current_user_id() THEN
    RAISE EXCEPTION 'unauthorized' USING ERRCODE = '42501';
  END IF;
END $$;
COMMIT;
```

### 7.2 Migration 0210 — RPCs

```sql
-- 0210__ai_quota_functions.sql
BEGIN;
CREATE OR REPLACE FUNCTION public.reserve_ai_usage(
  p_user TEXT, p_workspace TEXT, p_kind TEXT, p_mode TEXT,
  p_request_id TEXT, p_provider TEXT DEFAULT NULL, p_model TEXT DEFAULT NULL
) RETURNS TABLE(allowed BOOLEAN, used INT, remaining INT, day_local DATE)
LANGUAGE plpgsql SECURITY DEFINER SET search_path = public, pg_temp AS $$
DECLARE
  v_day DATE; v_count INT; v_limit INT;
BEGIN
  IF p_user IS NULL OR p_user <> current_user_id() THEN RAISE EXCEPTION 'unauthorized'; END IF;
  IF p_request_id IS NULL OR length(p_request_id) < 12 THEN RAISE EXCEPTION 'request_id_required'; END IF;

  -- limit موحّد للحماية التشغيلية بدون tiers أو paywall
  SELECT 100 INTO v_limit;

  v_day := local_day(p_user);

  INSERT INTO ai_usage(user_id, day_local, count) VALUES (p_user, v_day, 0)
  ON CONFLICT (user_id, day_local) DO NOTHING;

  SELECT count INTO v_count FROM ai_usage
  WHERE user_id = p_user AND day_local = v_day FOR UPDATE;

  IF v_count >= v_limit THEN
    RETURN QUERY SELECT FALSE, v_count, 0, v_day; RETURN;
  END IF;

  INSERT INTO ai_usage_events(id, user_id, workspace_id, day_local, kind, mode, request_id, status, provider, model)
  VALUES (
    encode(gen_random_bytes(13), 'hex'),
    p_user, p_workspace, v_day, p_kind, p_mode, p_request_id, 'reserved', p_provider, p_model
  )
  ON CONFLICT (user_id, request_id) DO NOTHING;

  IF NOT FOUND THEN
    -- نفس request_id حصل قبل كده — idempotent return
    SELECT count INTO v_count FROM ai_usage WHERE user_id = p_user AND day_local = v_day;
    RETURN QUERY SELECT TRUE, v_count, GREATEST(v_limit - v_count, 0), v_day; RETURN;
  END IF;

  UPDATE ai_usage SET count = count + 1, last_kind = p_kind, last_used_at = now()
  WHERE user_id = p_user AND day_local = v_day RETURNING count INTO v_count;

  RETURN QUERY SELECT TRUE, v_count, GREATEST(v_limit - v_count, 0), v_day;
END $$;

CREATE OR REPLACE FUNCTION public.complete_ai_usage(
  p_user TEXT, p_request_id TEXT, p_input_tokens INT DEFAULT NULL,
  p_output_tokens INT DEFAULT NULL, p_cost_cents BIGINT DEFAULT NULL
) RETURNS VOID LANGUAGE plpgsql SECURITY DEFINER SET search_path = public, pg_temp AS $$
BEGIN
  UPDATE ai_usage_events
  SET status = 'completed', completed_at = now(),
      input_tokens = p_input_tokens, output_tokens = p_output_tokens,
      cost_cents = p_cost_cents
  WHERE user_id = p_user AND request_id = p_request_id AND status = 'reserved';
END $$;

CREATE OR REPLACE FUNCTION public.refund_ai_usage(
  p_user TEXT, p_request_id TEXT, p_error_code TEXT DEFAULT NULL
) RETURNS VOID LANGUAGE plpgsql SECURITY DEFINER SET search_path = public, pg_temp AS $$
DECLARE v_day DATE;
BEGIN
  UPDATE ai_usage_events
  SET status = 'refunded', error_code = p_error_code, completed_at = now()
  WHERE user_id = p_user AND request_id = p_request_id AND status = 'reserved'
  RETURNING day_local INTO v_day;
  IF FOUND THEN
    UPDATE ai_usage SET count = GREATEST(count - 1, 0) WHERE user_id = p_user AND day_local = v_day;
  END IF;
END $$;
COMMIT;
```

### 7.3 قواعد الاستخدام

- **ممنوع** أي AI call برّه `runAIWithQuota` (من Wave 00). الـ wrapper بيستدعي الـ RPCs دي.
- **ممنوع** UPDATE مباشر على `ai_usage` أو `ai_usage_events` من الـ app — كله عبر RPCs.
- **ممنوع** quota check بدون reserve (race condition).
- لو الـ AI provider فشل، الـ wrapper لازم يستدعي `refund_ai_usage`.
- لو الـ user رفض الـ output (UI rejection)، **ما نـ refundش** — هو استهلك الـ token.

```tsx
  let raw: unknown;

  catch { throw new ValidationError({ body: ['Invalid JSON'] }); }
```

## 12. Repository Pattern

### 12.1 المبدأ

- كل table فيها repo واحد، اسمه `<table>.repo.ts`.
- ممنوع SQL في route handlers أو services بدون ما يعدّي على repo.
- الـ repo بيستقبل client من `withWorkspaceContext` (مش بيـ get pool بنفسه).
- ممنوع `SELECT *` — كل column explicit.
- كل function في الـ repo بترجع `Result<T>` أو بترمي `AppError`.

### 12.2 مثال: `tasks.repo.ts`

```tsx
import type { PgClient } from '@app/db';
import type { Task } from './tasks.types';
import { createUlid } from '@app/shared/ids';

export const tasksRepo = {
  async getById(db: PgClient, id: string): Promise<Task | null> {
    const { rows } = await db.query(
      `SELECT id, workspace_id, user_id, title, status, priority, due_at, position_key, version, created_at, updated_at
       FROM tasks WHERE id = $1 AND is_deleted = false`,
      [id],
    );
    return rows[0] ?? null;
  },
  async insert(db: PgClient, input: NewTask): Promise<Task> {
    const id = createUlid();
    const { rows } = await db.query(
      `INSERT INTO tasks (id, workspace_id, user_id, title, status, priority, position_key)
       VALUES ($1, $2, $3, $4, 'open', $5, $6)
       RETURNING id, workspace_id, user_id, title, status, priority, position_key, version, created_at, updated_at`,
      [id, input.workspaceId, input.userId, input.title, input.priority, input.positionKey],
    );
    return rows[0];
  },
  async updateById(db: PgClient, id: string, patch: Partial<NewTask>, expectedVersion: number): Promise<Task | null> {
    // optimistic concurrency
    const { rows } = await db.query(
      `UPDATE tasks SET title = COALESCE($2, title), status = COALESCE($3, status),
         priority = COALESCE($4, priority), version = version + 1
       WHERE id = $1 AND version = $5
       RETURNING id, title, status, priority, version, updated_at`,
      [id, patch.title, patch.status, patch.priority, expectedVersion],
    );
    return rows[0] ?? null;
  },
};
```

### 12.3 قاعدة عامة

- service بيستدعي repo + writes audit + writes queue jobs.
- repo دوره **بس** queries.
- route handler بيستدعي service.
- ESLint rule: ممنوع `db.query(` في `apps/web/`. كل queries في `packages/services` أو `packages/repo`.

## 13. Service Layer Rules

- كل domain فيه service file: `tasks.service.ts`, `notes.service.ts`, ...
- Service بيـ export functions نقية (pure functions returning Promises).
- كل mutation بيكتب audit event.
- كل mutation فيها unique constraint بتـ catch ON CONFLICT وترمي `AppError('CONFLICT')`.
- ممنوع service بيستدعي route helper (`withWorkspaceContext` etc) — الـ route layer responsibility.
- كل service function بياخد `db: PgClient` كـ first arg (للـ transaction support).

```tsx
const files = await glob('{apps,packages}/**/*.{ts,tsx}', { ignore: ['**/node_modules/**','**/dist/**'] });
const errors: string[] = [];

for (const file of files) {
  const text = await fs.readFile(file, 'utf8');
  const isClient = text.includes("'use client'");
  if (isClient) {
    CLIENT_FORBIDDEN.forEach((re) => { if (re.test(text)) errors.push(`❌ CLIENT LEAK in ${file}: ${re.source}`); });
  }
  AI_DIRECT_CALLS.forEach((re) => {
    if (re.test(text) && !file.includes('packages/ai/src/providers/')) {
      errors.push(`❌ DIRECT AI CALL in ${file}: ${re.source}`);
    }
  });
}

if (errors.length) {
  console.error(errors.join('\n'));
  process.exit(1);
}
console.log('✅ Preflight passed');
```

## 20. Common Pitfalls (Real Bugs Antigravity Likely Hits)

1. **Concurrent AI requests بـ check-only quota** → racy. Fix: استخدم RPC `reserve_ai_usage` exclusively.
2. **Webhook بدون nonce store** → replay attack. Fix: `webhook_nonces` PRIMARY KEY قبل ما تشتغل.
3. **service-role key في bundle الـ client** → catastrophic leak. Fix: ESLint + preflight scan.
4. **Float في expenses** → rounding errors في الـ budget logic. Fix: BIGINT cents فقط.
5. **`SET LOCAL` في middleware بـ pool client مش transactional** → سطور بترجع 0 أو سطور غلط. Fix: استخدم `BEGIN; SET LOCAL; ...; COMMIT;` في `withWorkspaceContext`.
6. **Updating `ai_usage.count` directly** → race. Fix: عبر RPC فقط.
7. **Missing `Idempotency-Key` على retries** → double-charge. Fix: middleware enforce.
8. **Forgetting RLS على table جديدة** → cross-tenant leak. Fix: `scripts/check-rls.ts` في CI.
9. **Repo بيـ join على table مفيهاش RLS context** → leak. Fix: كل tables الـ join لها workspace_id ومـ RLS-protected.
10. **Cron job بيشتغل بدون workspace context** → بيكتب على workspace غلط. Fix: cron jobs بـ explicit workspace_id arg.

## 21. ADRs الجديدة في Wave 02

- **ADR-0030:** AI quota = atomic RPC reserve/complete/refund.
- **ADR-0031:** Money columns = `<name>_cents BIGINT` only; no NUMERIC anywhere.
- **ADR-0032:** Webhook nonces table = source-of-truth for anti-replay.
- **ADR-0033:** Server/client boundary enforced via `packages/server-env` + ESLint `no-restricted-imports`.
- **ADR-0034:** Repository pattern = single SQL ownership per table.
- **ADR-0035:** Result<T> pattern + AppError; exceptions only for truly unexpected.
- **ADR-0036:** Route wrappers `withApiErrorHandling` / `withUserRoute` / `withWorkspaceRoute` are mandatory.
- **ADR-0037:** Optimistic concurrency via `version` column on tables that need it.
- **ADR-0038:** Required CI checks + release freeze rules.
- **ADR-0039:** `audit_events` extended with `request_id`/`trace_id`/`session_id`.
- **ADR-0040:** AI plan-based quota: `users.plan = 'free'` → 3/day, `'pro'` → 100/day.

## 35. CI/CD + Logs + Tests Baseline — داخل الصفحة الرئيسية

> **Status:** 🔴 BLOCKER — قبل ما تتعمل أول feature.
> 

> **Purpose:** أي تغيير لازم يعدي pipeline. أي error لازم يتسجل. أي endpoint عنده test. القسم ده كان ممنوع يفضل كاختصار أو صفحة فرعية، لذلك أصبح جزءًا مباشرًا من صفحة المرحلة 2.
> 

### 35.1 CI Pipeline minimum

1. Install + cache.
2. Typecheck.
3. Lint.
4. Preflight (secrets, server-only, AI wrapper).
5. Unit tests (Vitest).
6. RLS tests (pgTAP).
7. Build.
8. E2E smoke tests (Playwright) عند أول UI جاهز.
9. Deploy preview لو فيه UI جاهز.
10. Security scan: secret scan + dependency review أو بديله + static analysis.
11. Release workflow عند tags `wXX-frozen`.

### 35.2 Observability baseline

- Structured JSON logger من البداية.
- Redaction list: `password`, `token`, `vault_key`, `secret`, `stripe_*`, `SUPABASE_SERVICE_ROLE_KEY`, `DATABASE_URL`.
- Sentry initialized server + client لو `SENTRY_DSN` متوفر.
- `/api/health` أو `/api/v1/healthz` endpoint.
- Uptime monitor checks `/`, `/login`, `/api/health`.
- Axiom/structured logs optional لو الـ owner وفّر token.

### 35.3 Testing baseline

- Vitest config + first sample test.
- pgTAP scaffold + first RLS test.
- Playwright scaffold + smoke test.
- Coverage gate ≥ 60% للـ MVP, ≥ 80% بعد الإطلاق.
- DB safety workflow يتأكد من:
    - migrations apply cleanly.
    - RLS + FORCE RLS.
    - no `SELECT *`.
    - no float money columns.
    - no leaked secrets.

### 35.4 Hard rules

- ممنوع merge/freeze بدون CI أخضر.
- ممنوع `console.log` في الإنتاج.
- ممنوع secret في الـ logs.

- كل migration جديدة لازم تعدّي `check-migrations`.
- كل tenant table لازم تعدّي `check-rls`.
- كل task في قسم 36 لازم يتعلّم عليه قبل freeze.
