### Install One-Click Action

```tsx
export async function installTemplateAction(input: { templateId: string; targetWorkspaceId: string }) {
  const user = await requireUser()
  const { data: tpl } = await serviceSupabase.from('marketplace_templates')
    .select('*').eq('id', input.templateId).eq('status', 'approved').single()
  if (!tpl) throw new AppError('TEMPLATE_NOT_FOUND','not found', 404)

  // verify bundle integrity
  const bundle = await fetchBundle(tpl.bundle_url)
  const hash = await sha256(bundle)
  if (hash !== tpl.bundle_sha256) throw new AppError('BUNDLE_TAMPERED','sha256 mismatch', 422)

  // unpack & create entities (pages + DBs + views) under target workspace
  const result = await unpackTemplate(bundle, {
    workspaceId: input.targetWorkspaceId, userId: user.id, kind: tpl.kind,
  })

  await serviceSupabase.from('template_installs').insert({
    template_id: tpl.id, workspace_id: input.targetWorkspaceId,
    installed_by: user.id, target_page_id: result.rootPageId ?? null,
  })
  await serviceSupabase.rpc('inc_template_install_count', { p_id: tpl.id })
  await audit({ actorId: user.id, action: 'template.installed', targetId: tpl.id, targetType: 'template' })
  return { rootPageId: result.rootPageId, count: result.entitiesCreated }
}
```

## 🧠 Master Architect Upgrade — Platform Trust Gate

- **API boundary:** كل API request يثبت tenant/capability قبل resolver. scopes لا تُقرأ من client فقط، وrate limits per key/user/workspace/path.
- **OAuth security:** Authorization Code + PKCE mandatory، refresh token rotation، consent diff on scope change، revoke endpoint، وno wildcard redirect URIs.
- **Plugin sandbox:** Web Worker alone ليس كافي؛ add signed bundles, SHA-256 checksums, static scan, permission diff, version pinning، وnetwork allowlist.
- **No vault permissions:** plugins/importers/clipper/API لا يقرأوا Vault/E2EE content مطلقًا. أي طلب vault scope = reject at manifest/API layer.
- **Importer safety:** imports staged + dry-run + size streaming + antivirus/media scanning + rollback map + idempotent retry، بدون OOM أو duplicate pages.

## 🎯 نظرة عامة

المرحلة دي بتحوّل Zenith من تطبيق مغلق إلى منصة مفتوحة للمطورين. بنبني REST API + OAuth 2.0 + API keys + Plugin sandbox (Web Worker isolated) + Web Clipper extension (Chrome/Firefox) + Importers لمنافسين (Notion / Obsidian / Markdown / Evernote / Roam). الهدف: تكسير switching cost وبناء ecosystem moat.

**المستخدم هيكسب إيه:**

- يستخدم REST API رسمي مع OpenAPI spec جاهز ليبني integrations خاصة.
- ينشئ API keys بصلاحيات (scopes) محددة ويلغيها في أي وقت.
- يسجل OAuth app خاصة بيه عشان third-party apps توصل لحسابات المستخدمين بإذنهم.
- يثبت plugins معتمدة من الـ marketplace وكل بلجن يشتغل في sandbox معزول.
- يوافق على صلاحيات كل plugin (read pages / write blocks / call AI / إلخ) واحدة واحدة.
- يحفظ أي صفحة ويب لـ inbox بـ web clipper Chrome/Firefox extension بضغطة زر.
- يستورد Notion export ZIP أو Obsidian vault أو markdown folder بدون فقد محتوى أو backlinks.
- يتابع progress الـ import job لحظياً ويشوف أي errors في صفحات معينة.

**اللي هيتبني في المرحلة دي:**

- جدول `api_keys` بـ SHA-256 hash للـ secret + prefix للعرض + scopes array + last_used.
- جدول `oauth_apps` (للمطورين) + `oauth_grants` (للمستخدمين اللي وافقوا).
- Authorization Code Flow كامل بـ PKCE + state + nonce.
- Rate limit موحّد لحماية الأداء: 1000 req/min افتراضيًا، و100 req/min للمسارات الحساسة، بدون أي tier أو plan-based gating.
- OpenAPI 3.1 spec تتولد أوتوماتيك + Swagger UI على `/docs`.
- جدول `plugins` (catalog) + `installed_plugins` (per workspace) بـ status enum + granted_permissions array.
- Plugin sandbox بـ Web Worker + Comlink + API surface محدودة (read-only أو write بإذن).
- Browser extension Manifest V3 (popup + content script + service worker) بـ Mozilla Readability لاستخراج المحتوى.
- جدول `import_jobs` بـ progress tracking + error log + status enum.
- Notion importer يفك ZIP يوزع .md + .csv + properties.
- Obsidian importer يحل [[wikilinks]] و #tags و ![[images]].
- عزل الـ plugins: مفيش DOM access، مفيش auth tokens، مفيش vault.
- كل الـ webhooks للـ platform layer = HTTPS فقط + HMAC-SHA256 + nonce.

### 2) API key issuer + verifier

```tsx
// src/features/api/keys/issue.ts
import 'server-only'
import { createHash, randomBytes } from 'crypto'
import { serviceSupabase } from '@/lib/supabase/service'
import { requireUser } from '@/lib/auth/require-user'
import { AppError } from '@/lib/result'

const ALLOWED_SCOPES = [
  'pages:read','pages:write',
  'blocks:read','blocks:write',
  'databases:read','databases:write',
  'tasks:read','tasks:write',
  'webhooks:manage',
] as const

export async function issueApiKey(input: { name: string; scopes: string[] }) {
  const user = await requireUser()
  if (input.scopes.some(s => !ALLOWED_SCOPES.includes(s as any)))
    throw new AppError('INVALID_SCOPE','invalid scope', 422)

  const raw = `zth_live_${randomBytes(32).toString('base64url')}`
  const hash = createHash('sha256').update(raw).digest('hex')
  const prefix = raw.slice(0, 12)

  const { error } = await serviceSupabase.from('api_keys').insert({
    workspace_id: user.workspaceId,
    user_id:      user.id,
    name:         input.name,
    key_hash:     hash,
    key_prefix:   prefix,
    scopes:       input.scopes,
  })
  if (error) throw new AppError('KEY_CREATE_FAILED', error.message, 500)
  // الـ raw key بيتعرض للمستخدم مرة واحدة بس
  return { key: raw, prefix }
}

export async function verifyApiKey(authHeader: string) {
  if (!authHeader?.startsWith('Bearer ')) throw new AppError('NO_KEY','missing', 401)
  const raw = authHeader.slice(7)
  if (!raw.startsWith('zth_live_')) throw new AppError('INVALID_KEY','invalid', 401)
  const hash = createHash('sha256').update(raw).digest('hex')

  const { data } = await serviceSupabase.from('api_keys')
    .select('id, user_id, workspace_id, scopes, expires_at, is_revoked')
    .eq('key_hash', hash).maybeSingle()
  if (!data || data.is_revoked) throw new AppError('REVOKED','revoked', 401)
  if (data.expires_at && new Date(data.expires_at) < new Date())
    throw new AppError('EXPIRED','expired', 401)

  await serviceSupabase.from('api_keys')
    .update({ last_used_at: new Date().toISOString() }).eq('id', data.id)

  return { userId: data.user_id, workspaceId: data.workspace_id, scopes: data.scopes }
}
```

### 3) Rate limit موحّد لحماية الأداء

```tsx
// src/lib/api/rate-limit.ts
import { redis } from '@/lib/redis'
import { AppError } from '@/lib/result'

const DEFAULT_LIMIT_PER_MINUTE = 1000
const STRICT_LIMIT_PER_MINUTE = 100

export async function enforceApiRateLimit(userId: string, opts?: { strict?: boolean }) {
  const max = opts?.strict ? STRICT_LIMIT_PER_MINUTE : DEFAULT_LIMIT_PER_MINUTE
  const key = `apirl:${userId}:${Math.floor(Date.now() / 60_000)}`
  const count = await redis.incr(key)
  if (count === 1) await redis.expire(key, 60)
  if (count > max) {
    const retryAfter = 60 - (Math.floor(Date.now() / 1000) % 60)
    throw new AppError('RATE_LIMITED', `${max} req/min`, 429, { retryAfter })
  }
}
```

### 4) Webhook outbound للـ platform layer

```tsx
// src/lib/webhooks/platform-dispatch.ts
import 'server-only'
import { createHmac, randomBytes } from 'crypto'
import { serviceSupabase } from '@/lib/supabase/service'

export async function dispatchPlatformWebhook(
  webhookId: string,
  url: string,
  secret: string,
  event: string,
  data: Record<string, unknown>
) {
  const nonce = randomBytes(16).toString('hex')
  const ts    = Date.now().toString()
  const body  = JSON.stringify({ event, timestamp: Number(ts), data })
  // signed_payload = `${ts}.${nonce}.${body}` عشان مفيش replay
  const sig = createHmac('sha256', secret)
    .update(`${ts}.${nonce}.${body}`).digest('hex')

  const res = await fetch(url, {
    method: 'POST',
    headers: {
      'Content-Type':     'application/json',
      'Zenith-Signature': `v1=${sig}`,
      'Zenith-Nonce':     nonce,
      'Zenith-Timestamp': ts,
    },
    body, signal: AbortSignal.timeout(5000),
  })
  if (!res.ok) {
    await serviceSupabase.from('webhooks').update({
      failure_count: (await serviceSupabase.rpc('inc_failure', { id: webhookId })).data,
    }).eq('id', webhookId)
  }
}
```

```tsx
// src/lib/plugins/sandbox.ts
'use client'
import * as Comlink from 'comlink'

export class PluginSandbox {
  private worker: Worker

  constructor(bundleUrl: string, granted: string[]) {
    // كل plugin = Web Worker معزول
    this.worker = new Worker(bundleUrl, { type: 'module' })
    Comlink.expose(this.buildAPI(granted), this.worker)
  }

  private buildAPI(perms: string[]) {
    const api: Record<string, unknown> = {}

    if (perms.includes('pages:read')) {
      api.getPage = async (id: string) =>
        fetchSecure(`/api/v1/pages/${id}`)
    }
    if (perms.includes('blocks:write')) {
      api.createBlock = async (pageId: string, block: unknown) =>
        fetchSecure(`/api/v1/blocks`, { method: 'POST', body: { pageId, ...block } })
    }
    if (perms.includes('ai:invoke')) {
      api.askAI = async (prompt: string) =>
        fetchSecure(`/api/v1/ai/ask`, { method: 'POST', body: { prompt } })
    }

    // ممنوع تماماً: vault، auth tokens، admin RPCs، serverEnv
    return api
  }

  terminate() { this.worker.terminate() }
}

async function fetchSecure(path: string, init?: RequestInit) {
  const res = await fetch(path, {
    ...init, credentials: 'same-origin',
    headers: { ...(init?.headers ?? {}), 'Content-Type': 'application/json' },
    body: init?.body ? JSON.stringify(init.body) : undefined,
  })
  if (!res.ok) throw new Error(`API ${res.status}`)
  return res.json()
}
```

```tsx
import { Readability } from '@mozilla/readability'

export function extractArticle(): { title: string; html: string; url: string } {
  const docClone = document.cloneNode(true) as Document
  const reader   = new Readability(docClone)
  const article  = reader.parse()
  return {
    title: article?.title ?? document.title,
    html:  article?.content ?? document.body.innerHTML,
    url:   location.href,
  }
}

async function clipCurrentPage() {
  const [tab] = await chrome.tabs.query({ active: true, currentWindow: true })
  const [result] = await chrome.scripting.executeScript({
    target: { tabId: tab.id! },
```

```tsx
// src/lib/import/notion.ts
import JSZip from 'jszip'
import { serviceSupabase } from '@/lib/supabase/service'

export async function* importNotionZip(
  file: Blob, workspaceId: string, userId: string, jobId: string
) {
  const zip = await JSZip.loadAsync(file)
  let total = 0
  for (const f of Object.values(zip.files)) if (f.name.endsWith('.md')) total++

  await serviceSupabase.from('import_jobs')
    .update({ status: 'processing', started_at: new Date().toISOString(), total_pages: total })
    .eq('id', jobId)

  let imported = 0, failed = 0
  for (const [filename, entry] of Object.entries(zip.files)) {
    if (!filename.endsWith('.md') || entry.dir) continue
    try {
      const content = await entry.async('string')
      const { title, body } = parseNotionMarkdown(content)
      await serviceSupabase.from('pages').insert({
        workspace_id: workspaceId,
        title,
        content: body,
        source_meta: { from: 'notion', original_path: filename },
      })
      imported++
    } catch (err) {
      failed++
      await serviceSupabase.rpc('append_import_error', {
        p_job: jobId, p_error: `${filename}: ${String(err)}`
      })
    }
    if (imported % 10 === 0) {
      await serviceSupabase.from('import_jobs')
        .update({ imported, failed }).eq('id', jobId)
    }
    yield { stage: 'importing', filename, imported, failed, total }
  }

  await serviceSupabase.from('import_jobs').update({
    status: 'completed', completed_at: new Date().toISOString(),
    imported, failed,
  }).eq('id', jobId)

  yield { stage: 'complete', imported, failed, total }
}

function parseNotionMarkdown(md: string) {
  const lines = md.split('\n')
  const title = lines[0]?.replace(/^#\s*/, '').trim() || 'Untitled'
  return { title, body: lines.slice(1).join('\n').trim() }
}
```

```tsx
// src/lib/import/obsidian.ts
export function transformObsidianMarkdown(md: string) {
  // [[wikilinks]] → internal Zenith links
  md = md.replace(/\[\[([^\]]+)\]\]/g, (_, name) => `[${name}](/p/${slugify(name)})`)
  // #tags → stored في metadata
  const tags = Array.from(md.matchAll(/(?:^|\s)#([a-zA-Z0-9_-]+)/g)).map(m => m[1])
  // ![[image.png]] → رفع لـ Supabase Storage
  md = md.replace(/!\[\[([^\]]+)\]\]/g, (_, src) => `![${src}](attachments/${src})`)
  return { md, tags }
}

function slugify(s: string) {
  return s.toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/^-|-$/g, '')
}
```
