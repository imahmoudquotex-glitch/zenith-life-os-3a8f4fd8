  'database_table',

  'dialog',
] as const

export function assertShortcutConflict(shortcuts: Array<{ key: string; scope: string }>) {
  const seen = new Set<string>()
  for (const s of shortcuts) {
    const id = `${s.scope}:${s.key}`
    if (seen.has(id)) throw new Error(`SHORTCUT_CONFLICT:${id}`)
    seen.add(id)
  }
}

## 🧠 Master Architect Upgrade — Inclusive Desktop Contract

- **Arabic-first, not mirrored:** RTL layouts تستخدم logical properties، keyboard order طبيعي عربي، وtables/databases لا تنكسر في mixed Arabic/English content.
- **Dark accessibility:** Dark Mode الوحيد لازم يحقق WCAG 2.2 AA/AAA حيث أمكن، مع high-contrast dark variant بدون light theme مفروض.
- **Keyboard-first desktop:** كل surface أساسي — sidebar, command palette, editor, database, canvas, dialogs — قابل للاستخدام بالكامل من لوحة المفاتيح.
- **Assistive tech gates:** CI axe ليس كافي؛ أضف manual screen-reader checklist لـ NVDA/VoiceOver على critical flows.
- **Locale correctness:** لا hardcode timezone/currency/calendar. استخدم user settings + Intl، وراجع pluralization العربي لكل counters المهمة.

        short: { day: 'numeric', month: 'short', year: 'numeric' },

      },
      number: {
        currency: { style: 'currency', currency: 'EGP', minimumFractionDigits: 0 },
      },
    },
  }

### 3) RTL-aware layout

```tsx
// src/app/[locale]/layout.tsx
import { NextIntlClientProvider } from 'next-intl'
import { getMessages } from 'next-intl/server'
import { RTL_LOCALES, type Locale } from '@/i18n/config'

export default async function LocaleLayout({
  children, params,
}: { children: React.ReactNode; params: Promise<{ locale: Locale }> }) {
  const { locale } = await params
  const messages = await getMessages()
  const dir = RTL_LOCALES.includes(locale) ? 'rtl' : 'ltr'
  return (
    <html lang={locale} dir={dir}>
      <body>
        <a href="#main" className="skip-link">تخطي للمحتوى الرئيسي</a>
        <NextIntlClientProvider locale={locale} messages={messages}>
          {children}
        </NextIntlClientProvider>
      </body>
    </html>
  )
}
```

### 4) Locale-aware formatters

```tsx
// src/lib/i18n/format.ts
export function formatDate(
  date: Date | string, locale: string,
  opts: Intl.DateTimeFormatOptions = { dateStyle: 'long' },
  calendar: 'gregory' | 'islamic-umalqura' = 'gregory',
) {
  return new Intl.DateTimeFormat(`${locale}-u-ca-${calendar}`, opts)
    .format(new Date(date))
}

export function formatNumber(n: number, locale: string,
  opts: Intl.NumberFormatOptions = {}) {
  return new Intl.NumberFormat(locale, opts).format(n)
}

export function formatCurrency(cents: number, locale: string,
  currency: string = 'EGP') {
  return new Intl.NumberFormat(locale, {
    style: 'currency', currency, minimumFractionDigits: 0,
  }).format(cents / 100)
}

export function formatRelative(date: Date | string, locale: string) {
  const rtf = new Intl.RelativeTimeFormat(locale, { numeric: 'auto' })
  const diffMs = new Date(date).getTime() - Date.now()
  const abs = Math.abs(diffMs)
  if (abs < 60_000)        return rtf.format(Math.round(diffMs/1000), 'second')
  if (abs < 3_600_000)     return rtf.format(Math.round(diffMs/60_000), 'minute')
  if (abs < 86_400_000)    return rtf.format(Math.round(diffMs/3_600_000), 'hour')
  return rtf.format(Math.round(diffMs/86_400_000), 'day')
}
```

### 5) Locale switcher

```tsx
// src/components/i18n/LocaleSwitcher.tsx
'use client'
import { useLocale } from 'next-intl'
import { useRouter, usePathname } from 'next/navigation'
import { LOCALES } from '@/i18n/config'

const NAMES = { ar: 'العربية', en: 'English' } as const

export function LocaleSwitcher() {
  const locale   = useLocale()
  const router   = useRouter()
  const pathname = usePathname()
  function switchTo(next: string) {
    const newPath = pathname.replace(`/${locale}`, `/${next}`)
    router.replace(newPath)
  }
  return (
    <div role="group" aria-label="اختيار اللغة">
      {LOCALES.map(l => (
        <button key={l} onClick={() => switchTo(l)}
          aria-pressed={l === locale}
          aria-label={`تغيير اللغة إلى ${NAMES[l]}`}
          className={l === locale ? 'btn-primary' : 'btn-secondary'}>
          {NAMES[l]}
        </button>
      ))}
    </div>
  )
}
```

### 10) Missing translation detector

```tsx
// scripts/check-translations.ts
import fs from 'fs'
import path from 'path'

const base = JSON.parse(fs.readFileSync('src/messages/ar.json', 'utf8'))
const translations = ['en'].map(l => ({
  locale: l,
  data: JSON.parse(fs.readFileSync(`src/messages/${l}.json`, 'utf8')),
}))

function flatten(obj: any, prefix = ''): string[] {
  return Object.entries(obj).flatMap(([k, v]) =>
    typeof v === 'object' ? flatten(v, `${prefix}${k}.`) : [`${prefix}${k}`]
  )
}

const baseKeys = flatten(base)
let failed = false
for (const t of translations) {
  const tKeys = flatten(t.data)
  const missing = baseKeys.filter(k => !tKeys.includes(k))
  if (missing.length) {
    console.error(`[${t.locale}] missing ${missing.length}: ${missing.join(', ')}`)
    failed = true
  }
}
if (failed) process.exit(1)
```
