## Phase 40 — CI/CD Pipeline

### Architecture Rules

- Trunk-based: main → staging → production
- No direct push to main (PR required)
- Staging auto-deploy on PR merge to main
- Production: manual promote after staging verified
- Migrations: run BEFORE code deploy (never after)
- Secrets: only via GitHub Secrets / Vercel Env (never in repo)

### TypeScript: Safe DB Migration Helper

```tsx
// scripts/safe-migrate.ts
// Validates migration is additive before running

import { readFileSync } from 'fs'
import { execSync } from 'child_process'

const DESTRUCTIVE_PATTERNS = [
  /^\s*DROP\s+TABLE/im,
  /^\s*DROP\s+COLUMN/im,
  /^\s*ALTER\s+TABLE.*DROP/im,
  /^\s*TRUNCATE/im,
]

function isSafeMigration(sql: string): boolean {
  for (const pattern of DESTRUCTIVE_PATTERNS) {
    if (pattern.test(sql)) {
      console.error(`❌ UNSAFE: Found destructive SQL pattern:${pattern}`)
      return false
    }
  }
  return true
}

const migrationFiles = process.argv.slice(2)

for (const file of migrationFiles) {
  const sql = readFileSync(file, 'utf-8')
  if (!isSafeMigration(sql)) {
    console.error(`Migration${file} contains destructive changes!`)
    console.error('To drop a column/table, use a separate migration after verifying no code references it.')
    process.exit(1)
  }
}

console.log('✅ All migrations are safe (additive only)')
```

```tsx
// src/lib/cron/auth.ts
import 'server-only'
import { serverEnv } from '@/lib/env/server'

export function verifyCronSecret(req: Request): boolean {
  const auth = req.headers.get('authorization')
  if (!auth) return false
  const [scheme, token] = auth.split(' ')
  if (scheme !== 'Bearer') return false
  // Timing-safe comparison
  const a = Buffer.from(token)
  const b = Buffer.from(serverEnv.CRON_SECRET)
  if (a.length !== b.length) return false
  let diff = 0
  for (let i = 0; i < a.length; i++) diff |= a[i] ^ b[i]
  return diff === 0
}
```

### TypeScript: Edge Config Feature Flags

```tsx
// src/lib/feature-flags/edge-config.ts
import { unstable_cache } from 'next/cache'

interface FeatureFlags {
  maintenance_mode:     boolean
  new_onboarding:       boolean
  ai_smart_suggest:     boolean
  canvas_collaboration: boolean
  donations_enabled:    boolean
}

const DEFAULT_FLAGS: FeatureFlags = {
  maintenance_mode:     false,
  new_onboarding:       false,
  ai_smart_suggest:     false,
  canvas_collaboration: false,
  donations_enabled:    false,
}

// Cache for 30s — fast for users, allows quick flag changes
export const getFeatureFlags = unstable_cache(
  async (): Promise<FeatureFlags> => {
    try {
      // Vercel Edge Config (zero-latency global reads)
      const { createClient } = await import('@vercel/edge-config')
      const client = createClient(process.env.EDGE_CONFIG)
      const flags = await client.getAll() as Partial<FeatureFlags>
      return { ...DEFAULT_FLAGS, ...flags }
    } catch {
      return DEFAULT_FLAGS
    }
  },
  ['feature-flags'],
  { revalidate: 30 }
)

export async function isFeatureEnabled(flag: keyof FeatureFlags): Promise<boolean> {
  const flags = await getFeatureFlags()
  return flags[flag] ?? false
}
```
