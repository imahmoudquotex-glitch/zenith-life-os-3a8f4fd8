```tsx
export type RestoreDrillResult = {
  backupId: string
  startedAt: string
  restoredAt: string
  rpoSeconds: number
  rtoSeconds: number
  smokePassed: boolean
  rlsTestsPassed: boolean

export function assertRestoreDrillPassed(r: RestoreDrillResult) {
  if (!r.smokePassed || !r.rlsTestsPassed) throw new Error('RESTORE_DRILL_FAILED')
  if (r.rtoSeconds > 1800) throw new Error('RTO_EXCEEDED')

}
```

## 🧠 Master Architect Upgrade — Recoverability & Test Authority

- **Restore proves backup:** أي backup بدون automated restore drill ناجح = غير معتمد. سجّل RPO/RTO الفعلي وليس المستهدف فقط.
- **E2EE boundary:** backups لا تحتوي vault plaintext أبدًا؛ vault blobs تبقى encrypted، ومفاتيح المستخدم غير موجودة server-side.
- **Tenant isolation tests:** pgTAP/RLS tests لكل migration business. أي table بلا FORCE RLS أو isolation test يفشل CI.
- **Test pyramid:** unit/integration/e2e/a11y/perf/security/mutation tests مرتبطة بـ ownership وcoverage per critical module، لا global coverage فقط.
- **Failure drills:** اختبر restore، corrupted backup، missing WAL، expired KMS key، وpartial region loss ضمن drills ربع سنوية.

## 🎯 نظرة عامة

المرحلة دي بتبني صمام أمان أخير لأي كارثة: backups مشفّرة تلقائية داخل وخارج النطاق، مع أختبار دوري من الـ restore (drill) عشان نتأكد إنها فعلاً بتشتغل. إضافة لـ testing pyramid كامل يغطي كل التطبيق: unit (Vitest) + integration (DB + Redis) + E2E (Playwright) + visual regression (Chromatic/Percy) + performance (k6) + security (OWASP ZAP) + a11y (axe). RPO target: 15 دقيقة، RTO target: 30 دقيقة.

**الفريق والمستخدم هيكسبوا إيه:**

- ضمان إن البيانات مش هتضيع أبداً (RPO 15min).
- وقت recovery أقل من 30 دقيقة لو حصل تلف في DB.
- اختبار شهري للـ restore drill بـ metrics.
- تشفير backups بـ customer-managed keys.
- Cross-region replication (Frankfurt + Singapore).
- ثقة بأن تسليم أي feature بيمر على ألف test قبل production.
- Coverage > 80% على critical paths.
- E2E tests على كل user journey أساسي.

**اللي هيتبني في المرحلة دي:**

## 🔧 تفاصيل التنفيذ

### 1) Backup strategy (Postgres)

```bash
# scripts/backup-daily.sh
#!/usr/bin/env bash
set -euo pipefail

DATE=$(date -u +%Y-%m-%d)
BACKUP_FILE="zenith-${DATE}.dump"
KMS_KEY="arn:aws:kms:eu-central-1:...:key/..."

# 1. Logical dump (custom format, parallel)
pg_dump --format=custom --compress=9 --jobs=4 \
  --no-owner --no-privileges \
  --exclude-table-data='analytics_events*' \
  --file="/tmp/${BACKUP_FILE}" \
  "$DATABASE_URL"

# 2. Encrypt بـ KMS data key
aws kms generate-data-key --key-id "$KMS_KEY" --key-spec AES_256 \
  > /tmp/dk.json
DATAKEY=$(jq -r .Plaintext /tmp/dk.json)
ENCKEY=$(jq -r .CiphertextBlob /tmp/dk.json)
openssl enc -aes-256-gcm -in "/tmp/${BACKUP_FILE}" \
  -out "/tmp/${BACKUP_FILE}.enc" -K "$DATAKEY" -iv "$(openssl rand -hex 16)"
shred -u "/tmp/${BACKUP_FILE}"

# 3. Upload لـ R2 (primary) + S3 Glacier (archive)
rclone copy "/tmp/${BACKUP_FILE}.enc" "r2:zenith-backups/daily/" \
  --metadata-set "encrypted-key=${ENCKEY}"
rclone copy "/tmp/${BACKUP_FILE}.enc" "s3:zenith-backups-archive/daily/"

# 4. Verify
SIZE=$(stat -c %s "/tmp/${BACKUP_FILE}.enc")
echo "backup_size_bytes ${SIZE}" | curl --data-binary @- \
  "$PROMETHEUS_PUSH_GATEWAY/metrics/job/backup-daily"
shred -u "/tmp/${BACKUP_FILE}.enc" /tmp/dk.json
```

### 3) Restore drill (أتمتة شهرية)

```yaml
# .github/workflows/restore-drill.yml
name: Monthly Restore Drill
on:
  schedule: [{ cron: '0 4 1 * *' }]    # أول يوم في كل شهر
  workflow_dispatch:
jobs:
  drill:
    runs-on: ubuntu-latest
    timeout-minutes: 60
    steps:
      - uses: actions/checkout@v4
      - name: Provision drill DB
        run: terraform -chdir=drill apply -auto-approve
      - name: Download latest backup
        run: aws s3 cp s3://zenith-backups-archive/daily/latest.dump.enc /tmp/
      - name: Decrypt + restore
        run: |
          ./scripts/decrypt-backup.sh /tmp/latest.dump.enc /tmp/dump
          time pg_restore --dbname=$DRILL_DATABASE_URL --jobs=8 /tmp/dump
      - name: Smoke tests
        run: pnpm test:smoke --target=$DRILL_DATABASE_URL
      - name: Record metrics
        run: |
          echo "restore_duration_seconds $RESTORE_TIME" | \
          curl --data-binary @- $PUSHGATEWAY/metrics/job/restore-drill
      - name: Destroy drill DB
        if: always()
        run: terraform -chdir=drill destroy -auto-approve
      - name: Notify Slack
        uses: slackapi/slack-github-action@v1
        with:
          payload: '{"text":"✅ Monthly restore drill passed in $ env.RESTORE_TIME s"}'
```

```jsx
// tests/perf/api-load.js
import http from 'k6/http'
import { check, sleep } from 'k6'

export const options = {
```

### 8) Test data factories

```tsx
// tests/factories/user.ts
import { faker } from '@faker-js/faker/locale/ar'
import { serviceSupabase } from '@/lib/supabase/service'

export async function makeUser(overrides: Partial<any> = {}) {
  const email = faker.internet.email().toLowerCase()
  const { data: user } = await serviceSupabase.auth.admin.createUser({
    email, password: 'TestPass123!', email_confirm: true,
  })
  await serviceSupabase.from('profiles').upsert({
    id: user.user!.id, full_name: faker.person.fullName(),
    locale: 'ar', timezone: 'Africa/Cairo', ...overrides,
  })
  return { id: user.user!.id, email }
}

export async function makeTask(userId: string, overrides: Partial<any> = {}) {
  const { data } = await serviceSupabase.from('tasks').insert({
    user_id: userId,
    title: faker.lorem.sentence(),
    priority: faker.helpers.arrayElement(['low','medium','high']),
    ...overrides,
  }).select().single()
  return data
}
```
