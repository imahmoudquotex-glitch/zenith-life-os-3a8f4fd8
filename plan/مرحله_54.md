## Phase 34 — Observability & Monitoring

### Architecture Rules

- Logs: structured JSON only (no console.log in production)
- Sensitive fields never logged: password, token, vault_key
- Metrics: server-side only (no client-side metric APIs exposed)
- Alert on: error rate > 1%, p95 latency > 2s, DB connections > 80%

### TypeScript: Structured Logger

```tsx
// src/lib/monitoring/logger.ts
import 'server-only'

type LogLevel = 'debug' | 'info' | 'warn' | 'error'

interface LogEntry {
  level:     LogLevel
  event:     string
  userId?:   string
  requestId?: string
  duration?:  number
  metadata?:  Record<string, unknown>
  error?:     string
  stack?:     string
  ts:         string
}

// Fields that must NEVER appear in logs
const REDACTED_KEYS = new Set([
  'password', 'token', 'accessToken', 'refreshToken',
  'vaultKey', 'secret', 'stripe_secret', 'vapid',
  'access_token_enc', 'refresh_token_enc',
])

function redact(obj: unknown): unknown {
  if (!obj || typeof obj !== 'object') return obj
  if (Array.isArray(obj)) return obj.map(redact)
  return Object.fromEntries(
    Object.entries(obj as Record<string, unknown>).map(([k, v]) => [
      k,
      REDACTED_KEYS.has(k) ? '[REDACTED]' : redact(v),
    ])
  )
}

function writeLog(entry: LogEntry): void {
  const clean = redact(entry)
  if (process.env.NODE_ENV === 'production') {
    // Axiom / CloudWatch — structured JSON
    process.stdout.write(JSON.stringify(clean) + '\n')
  } else {
    const { level, event, ...rest } = entry
    const prefix = { debug: '🔍', info: 'ℹ️', warn: '⚠️', error: '❌' }[level]
    console[level === 'error' ? 'error' : level === 'warn' ? 'warn' : 'log'](
      `${prefix} [${event}]`, JSON.stringify(rest, null, 2)
    )
  }
}

export const log = {
  debug: (event: string, meta?: Record<string, unknown>) =>
    process.env.LOG_LEVEL === 'debug' && writeLog({ level: 'debug', event, metadata: meta, ts: new Date().toISOString() }),

  info: (event: string, meta?: Record<string, unknown>) =>
    writeLog({ level: 'info', event, metadata: meta, ts: new Date().toISOString() }),

  warn: (event: string, meta?: Record<string, unknown>) =>
    writeLog({ level: 'warn', event, metadata: meta, ts: new Date().toISOString() }),

  error: (event: string, err: unknown, meta?: Record<string, unknown>) => {
    const error   = err instanceof Error ? err.message : String(err)
    const stack   = err instanceof Error ? err.stack : undefined
    writeLog({ level: 'error', event, error, stack, metadata: meta, ts: new Date().toISOString() })
  },
}
```

### TypeScript: Request Timing Middleware

```tsx
// src/lib/monitoring/request-timing.ts
import 'server-only'
import { log } from './logger'

export function withTiming<T>(
  label: string,
  fn: () => Promise<T>,
  meta?: Record<string, unknown>
): Promise<T> {
  const start = performance.now()
  return fn().then(
    result => {
      const ms = Math.round(performance.now() - start)
      if (ms > 2000) log.warn(`slow_${label}`, { ms, ...meta })
      else            log.debug(`${label}_ok`,  { ms, ...meta })
      return result
    },
    err => {
      const ms = Math.round(performance.now() - start)
      log.error(`${label}_failed`, err, { ms, ...meta })
      throw err
    }
  )
}
```

## Phase 35 — Error Tracking (Sentry)

### Architecture Rules

- Sentry DSN: server-side only (not in client bundle)
- Never send vault content to Sentry
- PII scrubbing: email, userId → hash before sending
- Source maps: uploaded to Sentry, deleted from public build
- Sentry tunnel: via `/api/monitoring/tunnel` to bypass ad-blockers

### TypeScript: Sentry Setup (Server)

```tsx
// src/lib/monitoring/sentry.server.ts
import 'server-only'
import * as Sentry from '@sentry/nextjs'
import { serverEnv } from '@/lib/env/server'

const VAULT_PATTERNS = [
  /vault/i, /encrypted/i, /aes/i, /pbkdf/i, /passphrase/i,
]

export function initSentryServer() {
  Sentry.init({
    dsn:         serverEnv.SENTRY_DSN,
    environment: process.env.NODE_ENV,
    release:     process.env.NEXT_PUBLIC_APP_VERSION,
    tracesSampleRate: 0.1,  // 10% of transactions

    beforeSend(event) {
      // Strip PII
      if (event.user?.email) {
        event.user.email = `[redacted-${event.user.email.length}chars]`
      }

      // Never send vault-related errors with their data
      const message = event.message ?? event.exception?.values?.[0]?.value ?? ''
      if (VAULT_PATTERNS.some(p => p.test(message))) {
        event.extra = { note: 'vault error — data redacted' }
        delete event.request?.data
        delete event.request?.cookies
      }

      // Strip sensitive headers
      if (event.request?.headers) {
        delete event.request.headers['authorization']
        delete event.request.headers['cookie']
        delete event.request.headers['x-supabase-auth']
      }

      return event
    },
  })
}
```

### TypeScript: Sentry Setup (Client)

```tsx
// src/lib/monitoring/sentry.client.ts
import * as Sentry from '@sentry/nextjs'

export function initSentryClient() {
  Sentry.init({
    dsn:         process.env.NEXT_PUBLIC_SENTRY_DSN,
    environment: process.env.NODE_ENV,
    tracesSampleRate: 0.05,   // 5% client transactions
    tunnel:      '/api/monitoring/tunnel',  // bypass ad-blockers

    beforeSend(event) {
      // Never send vault content from client
      if (event.request?.url?.includes('/vault')) {
        return null  // Drop entirely
      }
      return event
    },

    integrations: [
      Sentry.replayIntegration({
        maskAllText:    true,   // GDPR: mask all user text
        blockAllMedia:  true,
        maskAllInputs:  true,
      }),
    ],
    replaysSessionSampleRate: 0.01,   // 1% of sessions
    replaysOnErrorSampleRate:  0.5,   // 50% of error sessions
  })
}
```

### TypeScript: Error Boundary with Sentry

```tsx
// src/components/error/ErrorBoundary.tsx
'use client'
import * as Sentry from '@sentry/nextjs'
import { Component, type ReactNode, type ErrorInfo } from 'react'

interface Props { children: ReactNode; fallback?: ReactNode }
interface State { hasError: boolean; eventId?: string }

export class ErrorBoundary extends Component<Props, State> {
  state: State = { hasError: false }

  static getDerivedStateFromError(): State {
    return { hasError: true }
  }

  componentDidCatch(error: Error, info: ErrorInfo) {
    const eventId = Sentry.captureException(error, {
      contexts: { react: { componentStack: info.componentStack } },
    })
    this.setState({ eventId })
  }

  render() {
    if (this.state.hasError) {
      return this.props.fallback ?? (
        <div role="alert" className="flex flex-col items-center justify-center gap-4 p-8">
          <h2 className="text-xl font-bold text-[var(--fg-primary)]">حدث خطأ غير متوقع</h2>
          <p className="text-[var(--fg-muted)] text-sm">تم إبلاغ الفريق. حاول تحديث الصفحة.</p>
          <button
            onClick={() => this.setState({ hasError: false })}
            className="btn-primary"
          >
            إعادة المحاولة
          </button>
          {this.state.eventId && (
            <p className="text-xs text-[var(--fg-muted)] font-mono">{this.state.eventId}</p>
          )}
        </div>
      )
    }
    return this.props.children
  }
}
```

### TypeScript: Sentry Tunnel (bypass ad-blockers)

```tsx
// src/app/api/monitoring/tunnel/route.ts
import 'server-only'
import { serverEnv } from '@/lib/env/server'

const SENTRY_HOST    = 'sentry.io'
const SENTRY_PROJECT = serverEnv.SENTRY_PROJECT_ID

export async function POST(req: Request) {
  const body   = await req.text()
  const pieces = body.split('\n')
  const header = JSON.parse(pieces[0])
  const dsn    = new URL(header.dsn)

  if (dsn.hostname !== SENTRY_HOST) {
    return Response.json({ error: 'Invalid DSN' }, { status: 400 })
  }

  const projectId = dsn.pathname.replace('/', '')
  if (projectId !== SENTRY_PROJECT) {
    return Response.json({ error: 'Invalid project' }, { status: 400 })
  }

  const upstream = await fetch(`https://${SENTRY_HOST}/api/${projectId}/envelope/`, {
    method: 'POST', body,
    headers: { 'Content-Type': 'application/x-sentry-envelope' },
  })

  return new Response(upstream.body, { status: upstream.status })
}
```
