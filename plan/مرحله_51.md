### Privacy Acceptance

- opt-out يمنع ingestion قبل الشبكة.
- event catalog إلزامي؛ أي event عشوائي يفشل في CI.
- retention للـ raw events محدود ومعلن.

## 🧠 Master Architect Upgrade — Analytics Privacy Gate

- **Privacy by design:** analytics opt-out respected من أول request. لا cookies tracking، لا fingerprinting دائم، ولا raw IP storage؛ فقط daily-rotating session hash.
- **No sensitive payloads:** scrub PII/secrets/vault/finance transaction details قبل ingest. أي event properties تحتوي `password/token/vault/secret/email/phone` تفشل أو تُحذف.
- **Tenant + purpose:** product analytics للقرار التشغيلي فقط، لا feature gating ولا paywall ولا بيع بيانات. retention محدود ومعلن.
- **Experiment ethics:** A/B tests لا تغيّر security/privacy/pricing/free access بدون explicit review. Guardrail metrics إلزامية قبل إعلان winner.
- **Stats correctness:** لا توقف تجربة قبل minimum sample + duration. سجّل hypothesis/primary metric/guardrails قبل التشغيل لمنع p-hacking.

## 🎯 نظرة عامة

المرحلة دي بتبني محرك تحليلات منتج داخلي يحترم خصوصية المستخدم بدل الاعتماد على Google Analytics أو Mixpanel. يجمع events بدون cookies tracking، يحسب funnels و retention cohorts و KPIs، يشغّل تجارب A/B إحصائية حقيقية (frequentist + Bayesian)، ويربط feature flags بالتجارب عشان الفريق يتخذ قرارات مبنية على بيانات، مش تخمين. كل ده مجاني بالكامل للمستخدم ومفيش tier checks.

**الفريق هيكسب إيه:**

- يفهم إيه الميزات الأكثر استخداماً وإيه اللي مفوّت.
- يشوف funnels (signup → first task → first review → retained day 7) بـ drop-off rates.
- يحسب retention cohorts (يومي، أسبوعي، شهري).
- يشغّل A/B tests على أي feature flag بـ statistical significance حقيقية.
- يشوف KPI dashboards (DAU/MAU، retention، engagement) في real-time.
- يحلل user journeys عشان يشوف أنماط الاستخدام.
- يعمل segmentation (أول أسبوع، power users، churned).
- يغلق التجربة الفاسدة ويفعل الفائزة تلقائياً.

**المستخدم هيكسب إيه:**

- خصوصية حقيقية: مفيش Google Analytics، مفيش Facebook Pixel.
- خيار opt-out في settings.

### 2) Daily salt rotation

```tsx
// src/lib/analytics/session.ts
import 'server-only'
import { createHash, randomBytes } from 'crypto'
import { redis } from '@/lib/redis'

export async function getDailySalt(): Promise<string> {
  const today = new Date().toISOString().slice(0, 10)
  const key = `analytics:salt:${today}`
  let salt = await redis.get(key)
  if (!salt) {
    salt = randomBytes(32).toString('hex')
    await redis.set(key, salt, 'EX', 86_400 * 2)
  }
  return salt
}

export async function deriveSessionId(ip: string, ua: string): Promise<string> {
  const salt = await getDailySalt()
  return createHash('sha256').update(salt + ip + ua).digest('hex').slice(0, 32)
}
```

### 3) Event ingestion endpoint

```tsx
// src/app/api/analytics/track/route.ts
import 'server-only'
import { z } from 'zod'
import { headers } from 'next/headers'
import { serviceSupabase } from '@/lib/supabase/service'
import { deriveSessionId } from '@/lib/analytics/session'
import { rateLimit } from '@/lib/rate-limit'

const EventSchema = z.object({
  events: z.array(z.object({
    name:       z.string().min(1).max(100),
    properties: z.record(z.unknown()).default({}),
    occurredAt: z.string().datetime(),
    platform:   z.enum(['web','ios','android']).default('web'),
    appVersion: z.string().max(50).optional(),
  })).min(1).max(100),
  experimentAssignments: z.record(z.string()).optional(),
  userId: z.string().uuid().optional(),
})

export async function POST(req: Request) {
  const h  = await headers()
  const ip = h.get('x-forwarded-for')?.split(',')[0]?.trim() ?? ''
  const ua = h.get('user-agent') ?? ''
  const cc = h.get('x-vercel-ip-country') ?? null

  await rateLimit({ key: `analytics:${ip}`, limit: 1000, windowSec: 60 })

  const body = EventSchema.parse(await req.json())
  const sessionId = await deriveSessionId(ip, ua)

  const rows = body.events.map(e => ({
    event_name: e.name,
    properties: scrubPii(e.properties),
    session_id: sessionId,
    user_id:    body.userId ?? null,
    platform:   e.platform,
    country_code: cc,
    app_version: e.appVersion ?? null,
    experiment_assignments: body.experimentAssignments ?? null,
    occurred_at: e.occurredAt,
  }))

  await serviceSupabase.from('analytics_events').insert(rows)
  return Response.json({ ok: true, ingested: rows.length })
}

function scrubPii(props: Record<string, unknown>): Record<string, unknown> {
  const BLOCKED = ['email','phone','address','full_name','ssn','password']
  const safe: Record<string, unknown> = {}
  for (const [k, v] of Object.entries(props)) {
    if (BLOCKED.some(b => k.toLowerCase().includes(b))) continue
    safe[k] = v
  }
  return safe
}
```

### 4) Client SDK (<5KB)

```tsx
// src/lib/analytics/sdk.ts
class AnalyticsClient {
  private queue: Array<any> = []
  private flushTimer: NodeJS.Timeout | null = null
  private endpoint = '/api/analytics/track'
  private optOut = false

  constructor() {
    if (typeof window === 'undefined') return
    this.optOut = localStorage.getItem('analytics-opt-out') === '1'
    const offline = localStorage.getItem('analytics-queue')
    if (offline) {
      try { this.queue.push(...JSON.parse(offline)) } catch {}
    }
    window.addEventListener('beforeunload', () => this.flush(true))
    window.addEventListener('online', () => this.flush())
  }

  track(name: string, properties: Record<string, unknown> = {}) {
    if (this.optOut) return
    this.queue.push({
      name, properties,
      occurredAt: new Date().toISOString(),
      platform: 'web',
    })
    this.scheduleFlush()
  }

  setOptOut(value: boolean) {
    this.optOut = value
    localStorage.setItem('analytics-opt-out', value ? '1' : '0')
    if (value) { this.queue = []; localStorage.removeItem('analytics-queue') }
  }

  private scheduleFlush() {
    if (this.flushTimer) return
    this.flushTimer = setTimeout(() => this.flush(), 5000)
    if (this.queue.length >= 50) this.flush()
  }

  private async flush(sync = false) {
    if (this.flushTimer) { clearTimeout(this.flushTimer); this.flushTimer = null }
    if (this.queue.length === 0) return
    const batch = this.queue.splice(0, 100)
    const body = JSON.stringify({ events: batch })
    try {
      if (sync && navigator.sendBeacon) {
        navigator.sendBeacon(this.endpoint, body)
      } else {
        await fetch(this.endpoint, { method: 'POST', body, keepalive: true,
          headers: { 'Content-Type': 'application/json' } })
      }
    } catch {
      this.queue.unshift(...batch)
      localStorage.setItem('analytics-queue', JSON.stringify(this.queue.slice(0, 500)))
    }
  }
}

export const analytics = new AnalyticsClient()
```

### 5) Funnels engine

```sql
CREATE OR REPLACE FUNCTION public.compute_funnel(
  p_steps      TEXT[],
  p_start_date DATE,
  p_end_date   DATE
) RETURNS TABLE (
  step_name  TEXT, step_index INT, user_count INT, drop_pct FLOAT
) LANGUAGE plpgsql STABLE SECURITY DEFINER
SET search_path = public, pg_temp AS $$
DECLARE
  v_step       TEXT; v_index INT := 0;
  v_prev_count INT := NULL; v_curr_count INT;
BEGIN
  FOREACH v_step IN ARRAY p_steps LOOP
    SELECT COUNT(DISTINCT user_id) INTO v_curr_count
    FROM analytics_events
    WHERE event_name = v_step
      AND occurred_at::date BETWEEN p_start_date AND p_end_date
      AND user_id IS NOT NULL;
    step_name := v_step; step_index := v_index;
    user_count := COALESCE(v_curr_count, 0);
    drop_pct := CASE WHEN v_prev_count IS NULL OR v_prev_count = 0 THEN 0
                ELSE (1 - v_curr_count::float / v_prev_count) * 100 END;
    RETURN NEXT;
    v_prev_count := v_curr_count; v_index := v_index + 1;
  END LOOP;
END $$;
```

### 7) Deterministic variant assignment

```tsx
// src/lib/experiments/assign.ts
import 'server-only'
import { serviceSupabase } from '@/lib/supabase/service'
import { createHash } from 'crypto'

export async function getOrAssignVariant(
  experimentKey: string, userId: string,
): Promise<{ variant: string; config: unknown } | null> {
  const { data: exp } = await serviceSupabase.from('experiments')
    .select('*, experiment_variants(*)').eq('key', experimentKey).single()
  if (!exp || exp.status !== 'running') return null

  const { data: existing } = await serviceSupabase.from('experiment_assignments')
    .select('variant_key').eq('experiment_id', exp.id).eq('user_id', userId).maybeSingle()
  if (existing) {
    const v = exp.experiment_variants.find((x: any) => x.key === existing.variant_key)
    return v ? { variant: v.key, config: v.config } : null
  }

  const hash = parseInt(
    createHash('sha256').update(exp.id + userId).digest('hex').slice(0, 8), 16
  ) % 100
  let cumulative = 0
  let selected = exp.experiment_variants[0]
  for (const v of exp.experiment_variants) {
    cumulative += v.allocation_pct
    if (hash < cumulative) { selected = v; break }
  }

  await serviceSupabase.from('experiment_assignments').insert({
    experiment_id: exp.id, user_id: userId, variant_key: selected.key,
  })
  return { variant: selected.key, config: selected.config }
}
```

### 8) Statistical engine

```tsx
// src/lib/experiments/stats.ts
export interface VariantStats {
  variant: string; conversions: number; total: number; rate: number
}

export function ztestTwoProportions(c: VariantStats, v: VariantStats) {
  const p1 = c.rate, p2 = v.rate
  const n1 = c.total, n2 = v.total
  const pooled = (c.conversions + v.conversions) / (n1 + n2)
  const se = Math.sqrt(pooled * (1 - pooled) * (1/n1 + 1/n2))
  const z  = (p2 - p1) / se
  const pValue = 2 * (1 - normalCdf(Math.abs(z)))
  return {
    zScore: z, pValue,
    isSignificant: pValue < 0.05,
    lift: ((p2 - p1) / p1) * 100,
  }
}

function normalCdf(x: number): number {
  const t = 1 / (1 + 0.2316419 * x)
  const d = 0.3989423 * Math.exp(-x * x / 2)
  return 1 - d * t * (0.3193815 + t * (-0.3565638 + t * (1.781478 + t * (-1.821256 + t * 1.330274))))
}

export function bayesianBeta(c: VariantStats, v: VariantStats) {
  const samples = 50_000
  let countBetter = 0, totalLift = 0
  for (let i = 0; i < samples; i++) {
    const cs = sampleBeta(1 + c.conversions, 1 + c.total - c.conversions)
    const vs = sampleBeta(1 + v.conversions, 1 + v.total - v.conversions)
    if (vs > cs) countBetter++
    totalLift += ((vs - cs) / cs) * 100
  }
  return {
    probVariantBetter: countBetter / samples,
    expectedLift: totalLift / samples,
  }
}

function sampleBeta(a: number, b: number): number {
  // مبسّط — في الإنتاج استخدم jstat
  let x = 0, y = 0
  for (let i = 0; i < a; i++) x -= Math.log(Math.random())
  for (let i = 0; i < b; i++) y -= Math.log(Math.random())
  return x / (x + y)
}
```
