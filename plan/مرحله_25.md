### Migration ~~0023~~ → 2500–2504 — immutable_xp_ledger + achievements

> 🔧 **تصحيح Audit:** رقم الـ migration اتغير من `0023` إلى `2500–2504` عشان يتوافق مع نظام الترقيم المعتمد في Phase 00.
> كل الـ UUIDs اتحولت لـ TEXT ULID. اتضاف workspace_id + RLS + FORCE RLS + CHECK constraints.

```sql
-- 2500_experience_points.sql
-- Immutable XP Ledger (No updates, only inserts — enforced by trigger)
CREATE TABLE experience_points (
  id TEXT PRIMARY KEY CHECK (id ~ '^[0-9A-HJKMNP-TV-Z]{26}$'),
  workspace_id TEXT NOT NULL REFERENCES workspaces(id) ON DELETE CASCADE,
  user_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,

  amount INT NOT NULL CHECK (amount > 0), -- positive-only (no XP deductions ever)
  reason_code TEXT NOT NULL CHECK (reason_code IN (
    'task_done','habit_streak_3','habit_streak_7','habit_streak_30',
    'deep_work_session','goal_milestone','weekly_review','daily_note',
    'achievement_bonus','learning_complete','focus_30min','challenge_done'
  )),
  metadata JSONB, -- { "task_id": "...", "duration": 25 }

  created_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX idx_xp_user_date ON experience_points(workspace_id, user_id, created_at DESC);

-- Immutability: append-only (no UPDATE or DELETE allowed)
CREATE OR REPLACE FUNCTION prevent_xp_mutation() RETURNS trigger AS $$
BEGIN RAISE EXCEPTION 'XP ledger is append-only. No updates or deletes allowed.'; END;
$$ LANGUAGE plpgsql;
CREATE TRIGGER no_update_xp BEFORE UPDATE OR DELETE ON experience_points
FOR EACH ROW EXECUTE FUNCTION prevent_xp_mutation();

ALTER TABLE experience_points ENABLE ROW LEVEL SECURITY;
ALTER TABLE experience_points FORCE ROW LEVEL SECURITY;
CREATE POLICY xp_isolation ON experience_points
  USING (workspace_id = current_workspace_id() AND user_id = current_user_id());

-- 2501_achievement_types.sql
-- Achievement definitions (admin-managed, workspace-scoped)
CREATE TABLE achievement_types (
  id TEXT PRIMARY KEY CHECK (id ~ '^[0-9A-HJKMNP-TV-Z]{26}$'),
  workspace_id TEXT NOT NULL REFERENCES workspaces(id) ON DELETE CASCADE,
  name_ar TEXT NOT NULL,
  name_en TEXT NOT NULL,
  description_ar TEXT,
  description_en TEXT,
  icon_slug TEXT,
  xp_reward INT NOT NULL DEFAULT 0 CHECK (xp_reward >= 0),
  secret BOOLEAN NOT NULL DEFAULT FALSE,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

ALTER TABLE achievement_types ENABLE ROW LEVEL SECURITY;
ALTER TABLE achievement_types FORCE ROW LEVEL SECURITY;
CREATE POLICY achievement_types_isolation ON achievement_types
  USING (workspace_id = current_workspace_id());

-- 2502_user_achievements.sql
-- User unlocked achievements
CREATE TABLE user_achievements (
  workspace_id TEXT NOT NULL REFERENCES workspaces(id) ON DELETE CASCADE,
  user_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  achievement_id TEXT NOT NULL REFERENCES achievement_types(id) ON DELETE CASCADE,
  unlocked_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  PRIMARY KEY (workspace_id, user_id, achievement_id)
);

ALTER TABLE user_achievements ENABLE ROW LEVEL SECURITY;
ALTER TABLE user_achievements FORCE ROW LEVEL SECURITY;
CREATE POLICY user_achievements_isolation ON user_achievements
  USING (workspace_id = current_workspace_id() AND user_id = current_user_id());

-- 2503_user_levels_view.sql
-- Dynamic Level View (read-only aggregate)
CREATE VIEW user_levels AS
  SELECT
    workspace_id,
    user_id,
    SUM(amount) as total_xp,
    floor(sqrt(SUM(amount) / 100)) + 1 as level
  FROM experience_points
  GROUP BY workspace_id, user_id;

-- 2504_daily_xp_cap.sql
-- Anti-grind: daily cap configuration per workspace
CREATE TABLE xp_daily_caps (
  workspace_id TEXT PRIMARY KEY REFERENCES workspaces(id) ON DELETE CASCADE,
  max_xp_per_day INT NOT NULL DEFAULT 500 CHECK (max_xp_per_day > 0),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

ALTER TABLE xp_daily_caps ENABLE ROW LEVEL SECURITY;
ALTER TABLE xp_daily_caps FORCE ROW LEVEL SECURITY;
CREATE POLICY xp_caps_isolation ON xp_daily_caps
  USING (workspace_id = current_workspace_id());
```

### The Ethical Motivation Engine

1.  **Intentional Gravity:** الـ XP مش بيتحسب أوتوماتيك لكل حاجة. المستخدم لازم يعمل "Claim" لبعض الإنجازات عشان يحس بقيمتها (Micro-moment of reflection).
2.  **Anti-Grind Limits:** حد أقصى (Daily Cap) للـ XP من المهام المتكررة لمنع إدمان الاستخدام (Digital Wellbeing).
3.  **Surprise & Delight:** إنجازات "Secret" بتظهر فقط لما تحققها (زي "Focused Monk" لما تعمل 5 Deep Work sessions في أسبوع واحد).

### UI/UX Specs (Zenith Gamification)

- **The Level Bar:** يظهر فقط في الـ Profile أو الـ Sidebar بلمسة Glassmorphism هادية.
- **Celebration Component:** لما يفتح Achievement، يظهر Toast شيك بـ `canvas-confetti` خفيف، مش بوب-أب مزعج.
- **Progression Tab:** صفحة بتعرض الـ History بتاعك كـ "Journey" مش مجرد أرقام.
