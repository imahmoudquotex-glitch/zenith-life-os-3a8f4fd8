## 🧠 Master Architect Upgrade — Search Isolation Gate

- **Mandatory filters:** كل Typesense collection/document يحمل `workspace_id`, visibility, resource_type، وquery layer يحقن tenant filter server-side. client لا يرسل filter كحقيقة.
- **No vault indexing:** vault/E2EE/password/plain secret fields ممنوعة من indexer/outbox/search snippets. أضف test يفشل لو ظهر `vault` في searchable payload.
- **Scoped search keys:** استخدم search-only keys قصيرة العمر per workspace/user/capability، ولا تعرض admin key للclient نهائيًا.
- **Indexer resilience:** outbox leases + idempotency + ordered per resource + tombstones delete. أي failure يدخل dead-letter مع replay command.
- **Privacy snippets:** snippets تعمل redaction للأيميلات/tokens/finance sensitive fields حسب policy قبل الإرجاع للـ UI أو AI.
    
    **تصحيح معماري إلزامي — W15-A:** Typesense Search لازم يكون zero-trust. ممنوع query بدون tenant/user filter، ممنوع vault docs، وممنوع sync worker بدون lease/dead-letter. أي `UUID` references في snippets تتعدل لـ TEXT ULID canonical.

```sql
-- supabase/migrations/1600_search_outbox.sql
BEGIN;

CREATE TABLE IF NOT EXISTS public.search_outbox (
  id            BIGSERIAL PRIMARY KEY,
  user_id       UUID NOT NULL,
  entity_type   TEXT NOT NULL,
  entity_id     UUID NOT NULL,
  operation     TEXT NOT NULL CHECK (operation IN ('upsert','delete')),
  payload       JSONB,
  attempts      INT NOT NULL DEFAULT 0,
  last_error    TEXT,
  status        TEXT NOT NULL DEFAULT 'pending'
                  CHECK (status IN ('pending','processing','completed','failed','dead')),
  enqueued_at   TIMESTAMPTZ NOT NULL DEFAULT now(),
  processed_at  TIMESTAMPTZ
);

CREATE INDEX outbox_pending ON public.search_outbox(enqueued_at)
  WHERE status = 'pending';
CREATE INDEX outbox_dead ON public.search_outbox(enqueued_at) WHERE status = 'dead';

-- Generic hook: enqueue outbox event from any source table
CREATE OR REPLACE FUNCTION public.enqueue_search_sync()
RETURNS TRIGGER LANGUAGE plpgsql SECURITY DEFINER SET search_path = public, pg_temp AS $$
DECLARE
  v_entity_type TEXT := TG_ARGV[0];
  v_op TEXT := CASE TG_OP WHEN 'DELETE' THEN 'delete' ELSE 'upsert' END;
  v_id UUID := COALESCE(NEW.id, OLD.id);
  v_user UUID := COALESCE(NEW.user_id, OLD.user_id);
BEGIN
  -- Skip if vault note
  IF v_entity_type = 'note' AND (NEW.is_vault = TRUE OR OLD.is_vault = TRUE) THEN
    INSERT INTO public.search_outbox(user_id, entity_type, entity_id, operation)
    VALUES (v_user, v_entity_type, v_id, 'delete');
    RETURN COALESCE(NEW, OLD);
  END IF;
  INSERT INTO public.search_outbox(user_id, entity_type, entity_id, operation,
    payload)
  VALUES (v_user, v_entity_type, v_id, v_op, to_jsonb(NEW));
  RETURN COALESCE(NEW, OLD);
END $$;

-- Hook on every searchable table
CREATE TRIGGER trg_sync_notes
  AFTER INSERT OR UPDATE OR DELETE ON public.notes
  FOR EACH ROW EXECUTE FUNCTION public.enqueue_search_sync('note');

CREATE TRIGGER trg_sync_tasks
  AFTER INSERT OR UPDATE OR DELETE ON public.tasks
  FOR EACH ROW EXECUTE FUNCTION public.enqueue_search_sync('task');

CREATE TRIGGER trg_sync_events
  AFTER INSERT OR UPDATE OR DELETE ON public.calendar_events
  FOR EACH ROW EXECUTE FUNCTION public.enqueue_search_sync('event');

CREATE TRIGGER trg_sync_goals
  AFTER INSERT OR UPDATE OR DELETE ON public.goals
  FOR EACH ROW EXECUTE FUNCTION public.enqueue_search_sync('goal');

CREATE TRIGGER trg_sync_projects

CREATE TRIGGER trg_sync_habits

COMMIT;
```

### TypeScript: Sync Worker (Cron every minute)

```tsx
// src/features/search/sync/worker.ts
import 'server-only'
import Typesense from 'typesense'
import { serviceSupabase } from '@/lib/supabase/service'

const client = new Typesense.Client({
  nodes: [{ host: process.env.TYPESENSE_HOST!, port: 443, protocol: 'https' }],
  apiKey: process.env.TYPESENSE_ADMIN_API_KEY!,
  connectionTimeoutSeconds: 10,
})

export async function syncSearchOutbox(batchSize = 100): Promise<{ processed: number }> {
  const { data: events } = await serviceSupabase
    .from('search_outbox')
    .select('*')
    .eq('status', 'pending')
    .order('enqueued_at', { ascending: true })
    .limit(batchSize)

  if (!events?.length) return { processed: 0 }

  // Group by operation for batch endpoints
  const upserts: any[] = []
  const deletes: string[] = []
  const eventIds: number[] = []

  for (const ev of events) {
    eventIds.push(ev.id)
    if (ev.operation === 'delete') {
      deletes.push(`${ev.entity_type}:${ev.entity_id}`)
    } else {
      upserts.push(buildTypesenseDoc(ev))
    }
  }

  // Mark in progress
  await serviceSupabase.from('search_outbox')
    .update({ status: 'processing' })
    .in('id', eventIds)

  try {
    if (upserts.length) {
      await client.collections('zenith_content').documents().import(upserts, { action: 'upsert' })
    }
    if (deletes.length) {
      await Promise.all(deletes.map(id =>
        client.collections('zenith_content').documents(id).delete().catch(() => null)
      ))
    }
    await serviceSupabase.from('search_outbox')
      .update({ status: 'completed', processed_at: new Date().toISOString() })
      .in('id', eventIds)
    return { processed: eventIds.length }
  } catch (err: any) {
    await serviceSupabase.from('search_outbox')
      .update({ status: 'pending', attempts: events[0].attempts + 1,
        last_error: err.message })
      .in('id', eventIds)
    // Dead-letter after 5 attempts
    await serviceSupabase.from('search_outbox')
      .update({ status: 'dead' })
      .gte('attempts', 5)
      .in('id', eventIds)
    throw err
  }
}

function buildTypesenseDoc(event: any) {
  const p = event.payload
  return {
    id: `${event.entity_type}:${event.entity_id}`,
    user_id: event.user_id,
    entity_type: event.entity_type,
    title: p?.title ?? p?.name ?? 'Untitled',
    content: (p?.description ?? p?.content)?.slice(0, 5000) ?? '',
    tags: p?.labels ?? p?.tags ?? [],
    updated_at: new Date(p?.updated_at ?? Date.now()).getTime(),
    created_at: new Date(p?.created_at ?? Date.now()).getTime(),
    is_vault: p?.is_vault ?? false,
    is_deleted: p?.is_deleted ?? false,
    priority: p?.priority,
    status: p?.status,
    category: p?.category,
    icon: p?.icon,
    color: p?.color ?? p?.cover_color,
    url: buildEntityUrl(event.entity_type, event.entity_id),
  }
}
```

### Search API

```tsx
// src/app/api/search/v2/route.ts
import 'server-only'
import { z } from 'zod'
import { requireUser } from '@/lib/auth/require-user'
import { withApiErrorHandling, parseJsonBody } from '@/lib/route'
import { assertSearchRateLimit } from '@/lib/notifications/rate-limit'
import Typesense from 'typesense'

const client = new Typesense.Client({
  nodes: [{ host: process.env.TYPESENSE_HOST!, port: 443, protocol: 'https' }],
  apiKey: process.env.TYPESENSE_SEARCH_API_KEY!,
  connectionTimeoutSeconds: 3,
})

const schema = z.object({
  query:        z.string().min(1).max(200),
  entity_types: z.array(z.string()).optional(),
  tag_paths:    z.array(z.string()).optional(),
  limit:        z.number().int().min(1).max(50).default(15),
  page:         z.number().int().min(1).default(1),
  facets:       z.boolean().default(true),
})

export const POST = withApiErrorHandling(async (req) => {
  const user = await requireUser()
  await assertSearchRateLimit(user.id, 'typesense-search', 120) // 120/min
  const params = await parseJsonBody(req, schema)

  const filters: string[] = [`user_id:=${user.id}`, 'is_vault:=false', 'is_deleted:=false']
  if (params.entity_types?.length)
    filters.push(`entity_type:=[${params.entity_types.join(',')}]`)
  if (params.tag_paths?.length)
    filters.push(`tag_paths:=[${params.tag_paths.join(',')}]`)

  try {
    const result = await client.collections('zenith_content').documents().search({
      q: params.query,
      query_by: 'title,content,tags',
      query_by_weights: '3,1,2',
      filter_by: filters.join(' && '),
      sort_by: '_text_match:desc,backlink_count:desc,updated_at:desc',
      per_page: params.limit,
      page: params.page,
      facet_by: params.facets ? 'entity_type,tag_paths,priority,status' : '',
      max_facet_values: 10,
      use_cache: true,
      cache_ttl: 30,
      highlight_fields: 'title,content',
      highlight_full_fields: 'title',
      snippet_threshold: 30,
      num_typos: 2,
      typo_tokens_threshold: 1,
    })

    // Log query for analytics
    await logSearch(user.id, params.query, result.hits?.length ?? 0)

    return Response.json({
      hits: result.hits,
      facet_counts: result.facet_counts,
      found: result.found,
      page: result.page,
      took: result.search_time_ms,
      engine: 'typesense',
    })
  } catch (err) {
    // Fallback to FTS (W12)
    const ftsResults = await fallbackToFTS(user.id, params.query, params.limit)
    return Response.json({ ...ftsResults, engine: 'fts-fallback' })
  }
})
```

```tsx
// src/features/quick-switcher/QuickSwitcher.tsx
'use client'
import { useEffect, useState } from 'react'
import { Command } from 'cmdk'
import { useRouter } from 'next/navigation'
import { searchAction } from '../search/actions/search'

export function QuickSwitcher() {
  const [open, setOpen] = useState(false)
  const [query, setQuery] = useState('')
  const [results, setResults] = useState<any[]>([])
  const router = useRouter()

  useEffect(() => {
    function down(e: KeyboardEvent) {
      if ((e.key === 'k' || e.key === 'K') && (e.metaKey || e.ctrlKey)) {
        e.preventDefault()
        setOpen(o => !o)
      }
    }
    document.addEventListener('keydown', down)
    return () => document.removeEventListener('keydown', down)
  }, [])

  useEffect(() => {
    if (!query.trim()) {
      setResults(buildDefaultActions())
      return
    }
    const id = setTimeout(async () => {
      const res = await searchAction({ query, limit: 8 })
      setResults([...buildDefaultActions(query), ...(res.hits ?? [])])
    }, 100)
    return () => clearTimeout(id)
  }, [query])

  return (
    <Command.Dialog open={open} onOpenChange={setOpen} label="Search"
      className="fixed inset-0 z-50 flex items-start justify-center bg-black/60 backdrop-blur-sm p-24">
      <div className="w-full max-w-2xl rounded-2xl border border-white/10 bg-[var(--bg-elevated)] shadow-2xl">
        <Command.Input value={query} onValueChange={setQuery}
          placeholder="ابحث في Zenith..."
          className="w-full bg-transparent px-5 py-4 text-lg outline-none" />
        <Command.List className="max-h-96 overflow-y-auto p-2">
          <Command.Empty className="p-8 text-center text-[var(--fg-muted)]">
            لا توجد نتائج
          </Command.Empty>
          {results.map(r => (
            <Command.Item
              key={r.id ?? r.action}
              onSelect={() => { router.push(r.url); setOpen(false) }}
              className="flex items-center gap-3 rounded-lg px-3 py-2 hover:bg-white/5 cursor-pointer"
            >
              <span>{r.icon ?? typeIcon(r.entity_type)}</span>
              <div className="flex-1">
                <div className="text-sm font-medium">{r.title}</div>
                {r.snippet && (

                )}
              </div>
              <span className="text-xs text-[var(--fg-muted)]">{r.entity_type}</span>
            </Command.Item>
          ))}
        </Command.List>
      </div>
    </Command.Dialog>
  )
}
```

```sql
CREATE TABLE IF NOT EXISTS public.search_log (

  user_id       UUID NOT NULL REFERENCES public.profiles(id) ON DELETE CASCADE,
  query         TEXT NOT NULL,

  clicked_result_id TEXT,
  filters       JSONB,
  created_at    TIMESTAMPTZ NOT NULL DEFAULT now()
);
ALTER TABLE public.search_log ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.search_log FORCE ROW LEVEL SECURITY;
CREATE POLICY sl_owner ON public.search_log FOR ALL
  USING (user_id = (SELECT auth.uid()))
  WITH CHECK (user_id = (SELECT auth.uid()));

CREATE INDEX search_log_user_recent ON public.search_log(user_id, created_at DESC);
CREATE INDEX search_log_no_results ON public.search_log(query, created_at)
  WHERE result_count = 0;

-- Retention: 60 days
CREATE OR REPLACE FUNCTION public.cleanup_old_search_log()
RETURNS VOID LANGUAGE plpgsql SECURITY DEFINER SET search_path = public, pg_temp AS $$
BEGIN
  DELETE FROM public.search_log WHERE created_at < now() - INTERVAL '60 days';
END $$;

COMMIT;
```

### No-Result Suggestions

```tsx
// src/features/search/lib/no-result-suggestions.ts
import 'server-only'
import { serviceSupabase } from '@/lib/supabase/service'

export async function suggestForNoResults(userId: string, query: string): Promise<string[]> {
  // Typo correction: query Typesense with num_typos=3
  // Synonyms: lookup synonym set
  // Recent successful queries: pull from search_log where result_count > 0
  const { data } = await serviceSupabase
    .from('search_log')
    .select('query')
    .eq('user_id', userId)
    .gt('result_count', 0)
    .ilike('query', `%${query.slice(0, 3)}%`)
    .order('created_at', { ascending: false })
    .limit(5)
  return (data ?? []).map(r => r.query)
}
```
