```tsx
export const ROUTE_BUDGETS = {
  '/': { jsKb: 160, lcpMs: 1500, inpMs: 180 },

  '/editor/[id]': { jsKb: 260, lcpMs: 1800, inpMs: 180 },
  '/databases/[id]': { jsKb: 300, lcpMs: 2000, inpMs: 200 },
} as const

export function assertBudget(route: keyof typeof ROUTE_BUDGETS, m: PerfMetrics) {
  const b = ROUTE_BUDGETS[route]
  if (m.jsKb > b.jsKb) throw new Error(`BUDGET_JS:${route}`)
  if (m.lcpMs > b.lcpMs) throw new Error(`BUDGET_LCP:${route}`)
  if (m.inpMs > b.inpMs) throw new Error(`BUDGET_INP:${route}`)
}
```

## 🧠 Master Architect Upgrade — Desktop PWA Performance Gate

- **Desktop app budgets:** shell boot, command palette، dashboard، editor، وdatabase views لهم budgets منفصلة. فشل أي budget في CI يمنع merge.
- **Service worker safety:** لا cache لأي `/api`, auth, vault, AI, webhooks, exports، أو user-sensitive routes. offline shell فقط + explicit allowlist.
- **Dark-only visual perf:** snapshots تقيس dark theme فقط، وتمنع CLS من font/theme swaps أو late color token loads.
- **RUM + SLO:** Web Vitals تدخل SLO dashboards؛ LCP/INP/CLS alerts مرتبطة بowner/runbook وليس مجرد analytics.
- **SEO boundary:** SEO للـ marketing/public pages فقط. private workspace/share protected pages noindex/noarchive حسب policy.

## 🎯 نظرة عامة

### 8) Redis cache wrapper

```tsx
// src/lib/cache/redis-cache.ts
import 'server-only'
import { redis } from '@/lib/redis'

export async function cached<T>(
  key: string, ttlSec: number, fetcher: () => Promise<T>,
): Promise<T> {
  const hit = await redis.get(key)
  if (hit) return JSON.parse(hit) as T
  const fresh = await fetcher()
  await redis.set(key, JSON.stringify(fresh), 'EX', ttlSec)
  return fresh
}

export async function invalidate(pattern: string) {
  const keys = await redis.keys(pattern)
  if (keys.length) await redis.del(...keys)
}
```

### 9) Slow query monitor

```sql
-- supabase/migrations/0055_slow_queries.sql
CREATE OR REPLACE FUNCTION public.pg_slow_queries_top10()
RETURNS TABLE (query TEXT, calls BIGINT, mean_ms FLOAT, max_ms FLOAT)
LANGUAGE sql SECURITY DEFINER
SET search_path = public, pg_temp AS $$
  SELECT query, calls, mean_exec_time, max_exec_time
  FROM pg_stat_statements
  WHERE query NOT ILIKE '%pg_stat_statements%'
  ORDER BY mean_exec_time DESC
  LIMIT 10;
$$;
```

### 11) Smart 404 page

```tsx
// src/app/not-found.tsx
import Link from 'next/link'
import { headers } from 'next/headers'
import { unifiedSearch } from '@/lib/search'

export default async function NotFound() {
  const h = await headers()
  const referer = h.get('referer') ?? ''
  const guessQuery = referer.split('/').pop()?.replace(/-/g, ' ') ?? ''
  const suggestions = guessQuery ? await unifiedSearch(guessQuery, 5) : []
  return (
    <main className="py-20 text-center max-w-2xl mx-auto">
      <h1 className="text-6xl font-bold">404</h1>
      <p className="mt-4 text-xl">الصفحة غير موجودة</p>
      {suggestions.length > 0 && (
        <div className="mt-8 text-left">
          <h2 className="font-semibold mb-3">يمكن تقصد:</h2>
          <ul className="space-y-2">
            {suggestions.map(s => (
              <li key={s.url}>
                <Link href={s.url} className="underline">{s.title}</Link>
              </li>
            ))}
          </ul>
        </div>
      )}
      <Link href="/" className="btn-primary mt-10 inline-block">العودة للرئيسية</Link>
    </main>
  )
}
```
