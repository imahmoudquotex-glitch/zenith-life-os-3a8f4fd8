### Launch Gate Evaluator

```tsx
export type LaunchGateSnapshot = {
  sloBurnClean: boolean
  restoreDrillPassed: boolean
  criticalSecurityFindings: number
  highSecurityFindings: number
  loadTestMultiplierPassed: number
  rollbackRehearsed: boolean
  supportStaffReady: boolean
}

export function assertLaunchReady(s: LaunchGateSnapshot) {
  if (!s.sloBurnClean) throw new Error('LAUNCH_BLOCKED_SLO_BURN')
  if (!s.restoreDrillPassed) throw new Error('LAUNCH_BLOCKED_RESTORE_DRILL')
  if (s.criticalSecurityFindings > 0 || s.highSecurityFindings > 0) throw new Error('LAUNCH_BLOCKED_SECURITY')
  if (s.loadTestMultiplierPassed < 10) throw new Error('LAUNCH_BLOCKED_LOAD_TEST')
  if (!s.rollbackRehearsed) throw new Error('LAUNCH_BLOCKED_ROLLBACK_NOT_REHEARSED')
  if (!s.supportStaffReady) throw new Error('LAUNCH_BLOCKED_SUPPORT_NOT_READY')
}
```

## 🧠 Master Architect Upgrade — Launch Readiness as Engineering Gate

- **No launch by checklist only:** launch يحتاج SLO burn-rate clean، restore drill passed، security high/critical = 0، load test 10x، وrollback rehearsal.
- **Chaos blast radius:** chaos يبدأ staging ثم canary production بإذن، مع abort criteria واضحة وfeature kill-switch جاهز.
- **Incident readiness:** on-call rotation، severity matrix، runbooks، status page، comms templates عربي/إنجليزي، وpostmortem template كلها blockers.
- **Dependency degradation:** AI/email/webhooks/search/integrations عند سقوطهم يقدموا graceful fallback ولا يكسروا core app.
- **War-room metrics:** launch dashboard يراقب activation, error rate, latency, DB connections, queue depth, AI quota, delivery failures، وsupport load.

### 3) Circuit breaker pattern

```tsx
// src/lib/resilience/circuit-breaker.ts
type State = 'closed' | 'open' | 'half-open'

export class CircuitBreaker {
  private state: State = 'closed'
  private failures = 0
  private nextAttempt = 0

  constructor(private opts: {
    threshold: number; resetTimeoutMs: number; name: string;
  }) {}

  async execute<T>(fn: () => Promise<T>, fallback: () => Promise<T>): Promise<T> {
    if (this.state === 'open') {
      if (Date.now() < this.nextAttempt) return fallback()
      this.state = 'half-open'
    }
    try {
      const result = await fn()
      this.onSuccess()
      return result
    } catch (err) {
      this.onFailure()
      return fallback()
    }
  }

  private onSuccess() {
    this.failures = 0
    this.state = 'closed'
  }
  private onFailure() {
    this.failures++
    if (this.failures >= this.opts.threshold) {
      this.state = 'open'
      this.nextAttempt = Date.now() + this.opts.resetTimeoutMs
      log.warn(`Circuit OPEN: ${this.opts.name}`)
    }
  }
}

// Usage:
export const openaiBreaker = new CircuitBreaker({
  threshold: 5, resetTimeoutMs: 30_000, name: 'openai',
})
```

### 4) Feature kill switches

```tsx
// src/lib/kill-switches.ts
import { redis } from '@/lib/redis'

const KILL_SWITCHES = [
  'ai_summarization', 'public_sharing', 'webhooks', 'integrations',
  'automation_runs', 'analytics_tracking', 'pdf_export',
] as const

export type KillSwitchKey = (typeof KILL_SWITCHES)[number]

export async function isFeatureKilled(key: KillSwitchKey): Promise<boolean> {
  const v = await redis.get(`kill-switch:${key}`)
  return v === '1'
}

export async function killFeature(key: KillSwitchKey, reason: string, by: string) {
  await redis.set(`kill-switch:${key}`, '1')
  await redis.set(`kill-switch-meta:${key}`, JSON.stringify({
    reason, by, at: new Date().toISOString(),
  }))
  // alert
  await sendSlack(`#incidents`, `🚨 Feature KILLED: ${key} by ${by} (${reason})`)
}
```

### 7) Game day script

```tsx
// scripts/game-day.ts
import { EXPERIMENTS } from '@/chaos/experiments'
import { sendSlack } from '@/lib/slack'

async function runGameDay() {
  const today = new Date().toISOString().slice(0,10)
  await sendSlack('#incidents', `🎮 Game Day starting ${today}`)

  for (const exp of EXPERIMENTS) {
    await sendSlack('#incidents', `Starting: ${exp.name}`)
    const cleanup = await exp.inject()
    await sleep(60_000)  // 1 min observation
    const before = await exp.observe()
    await sleep(4 * 60_000)  // 4 min more
    const after = await exp.observe()
    await cleanup()
    await sleep(60_000)  // recovery
    const recovered = await exp.observe()
    await sendSlack('#incidents', `✅ ${exp.name}: healthy=${recovered.healthy}`)
  }
  await sendSlack('#incidents', `🎬 Game Day complete`)
}
```
