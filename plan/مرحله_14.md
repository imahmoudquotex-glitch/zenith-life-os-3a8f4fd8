```tsx
export async function leaseGraphJobs(workerId: string, limit = 100) {
  return db.manyOrNone(`
    UPDATE graph_index_outbox
    SET locked_at = now(), locked_by = $1
    WHERE id IN (
      SELECT id FROM graph_index_outbox
      WHERE dead_reason IS NULL
        AND (locked_at IS NULL OR locked_at < now() - interval '5 minutes')
        AND next_attempt_at <= now()
      ORDER BY id
      LIMIT $2
      FOR UPDATE SKIP LOCKED
    )
    RETURNING *`, [workerId, limit])
}
```

### Realtime Subscriber (Backlinks Panel)

```tsx
export function useBacklinksRealtime(targetId: string) {
  const [links, setLinks] = useState<BacklinkRow[]>([])

  useEffect(() => {
    void fetchInitial(targetId).then(setLinks)

    const channel = supabase.channel(`bli:${targetId}`)
      .on('postgres_changes', {
        event: '*', schema: 'public', table: 'bidirectional_link_index',
        filter: `target_entity_id=eq.${targetId}`,
      }, (payload) => {
        if (payload.eventType === 'INSERT') {
          setLinks(prev => [...prev, payload.new as BacklinkRow])
        } else if (payload.eventType === 'DELETE') {
          setLinks(prev => prev.filter(l => l.id !== payload.old.id))
        }
      })
      .subscribe()

    return () => { void channel.unsubscribe() }
  }, [targetId])

  return links
}
```

## Phase 50 — Knowledge Graph Engine

### Architecture Rules

- **Materialized view** `graph_edges` aggregates `(from_entity, to_entity, edge_type, weight)` from notes/backlinks/tags/dependencies
- Refresh strategy: **incremental** via INSERT/UPDATE triggers on source tables
- Full rebuild: nightly cron (W14 scheduled_jobs) at 03:00 UTC
- Max graph size per user: 10k nodes · 50k edges (truncate oldest beyond)
- **Vault content** never appears in graph (vault_excluded = true in source filter)
- All graph queries use SECURITY DEFINER RLS-compatible functions
- Graph layout computed in **Web Worker** (d3-force) — never blocks main thread

### Migration 1400 — graph_cache

```sql
-- supabase/migrations/1400_graph_cache.sql
BEGIN;

CREATE MATERIALIZED VIEW public.graph_edges AS
  -- backlinks (note → note)
  SELECT user_id, source_id AS from_entity, source_type AS from_type,
         target_id AS to_entity, target_type AS to_type,
         'backlink' AS edge_type, 1 AS weight, created_at
  FROM public.backlinks WHERE NOT is_deleted
UNION ALL
  -- task → goal
  SELECT user_id, id AS from_entity, 'task' AS from_type,
         goal_id AS to_entity, 'goal' AS to_type,
         'task_of_goal' AS edge_type, 2 AS weight, created_at
  FROM public.tasks WHERE goal_id IS NOT NULL AND NOT is_deleted
UNION ALL
  -- task → project
  SELECT user_id, id, 'task', project_id, 'project',
         'task_of_project', 2, created_at
  FROM public.tasks WHERE project_id IS NOT NULL AND NOT is_deleted
UNION ALL
  -- project → goal
  SELECT user_id, id, 'project', goal_id, 'goal',
         'project_of_goal', 3, created_at
  FROM public.projects WHERE goal_id IS NOT NULL AND NOT is_deleted
UNION ALL
  -- habit → goal (W10.5 + W11 bridge)
  SELECT user_id, habit_id AS from_entity, 'habit' AS from_type,
         goal_id AS to_entity, 'goal' AS to_type,
         'habit_supports_goal' AS edge_type, weight, created_at
  FROM public.habit_goal_links;

CREATE UNIQUE INDEX graph_edges_idx ON public.graph_edges(
  user_id, from_entity, to_entity, edge_type
);
CREATE INDEX graph_edges_user_from ON public.graph_edges(user_id, from_entity);
CREATE INDEX graph_edges_user_to   ON public.graph_edges(user_id, to_entity);

-- Refresh function (incremental-friendly)
CREATE OR REPLACE FUNCTION public.refresh_graph_edges()
RETURNS VOID LANGUAGE plpgsql SECURITY DEFINER SET search_path = public, pg_temp AS $$
BEGIN
  REFRESH MATERIALIZED VIEW CONCURRENTLY public.graph_edges;
END $$;

-- Cron job 'graph_edges_refresh' (0 3 * * *) is registered by W14 (scheduled_jobs owner).

COMMIT;
```

### TypeScript: getGraphForUser

```tsx
// src/features/graph/lib/get-graph.ts
import 'server-only'
import { unstable_cache } from 'next/cache'
import { serviceSupabase } from '@/lib/supabase/service'

export interface GraphNode {
  id: string; type: string; label: string
  group: 'note'|'task'|'goal'|'project'|'habit'|'event'|'tag'
  size?: number; color?: string
}
export interface GraphEdge {
  source: string; target: string; type: string; weight: number
}

export const getGraphForUser = unstable_cache(
  async (userId: string, opts: {
    rootEntityId?: string; depth?: number; nodeLimit?: number
  } = {}): Promise<{ nodes: GraphNode[]; edges: GraphEdge[] }> => {
    const { rootEntityId, depth = 2, nodeLimit = 500 } = opts

    let edges: any[]
    if (rootEntityId) {
      // BFS from root, up to depth
      const { data } = await serviceSupabase.rpc('bfs_graph', {
        p_user: userId, p_root: rootEntityId, p_depth: depth, p_limit: nodeLimit,
      })
      edges = data ?? []
    } else {
      const { data } = await serviceSupabase
        .from('graph_edges')
        .select('from_entity, from_type, to_entity, to_type, edge_type, weight')
        .eq('user_id', userId)
        .limit(nodeLimit * 4)
      edges = data ?? []
    }

    const nodeSet = new Map<string, GraphNode>()
    const edgeList: GraphEdge[] = []

    for (const e of edges) {
      addNode(nodeSet, e.from_entity, e.from_type)
      addNode(nodeSet, e.to_entity, e.to_type)
      edgeList.push({
        source: e.from_entity, target: e.to_entity,
        type: e.edge_type, weight: e.weight,
      })
    }
    // Resolve labels in batch
    await resolveLabels(nodeSet, userId)
    return { nodes: Array.from(nodeSet.values()), edges: edgeList }
  },
  ['user-graph'],
  { revalidate: 300, tags: ['graph'] }
)
```

```tsx
// src/features/graph/workers/layout.worker.ts
import { forceSimulation, forceManyBody, forceLink, forceCenter, forceCollide } from 'd3-force'

self.addEventListener('message', (e) => {
  const { nodes, edges, width, height } = e.data
  const sim = forceSimulation(nodes)
    .force('charge', forceManyBody().strength(-150))
    .force('link', forceLink(edges).id((d: any) => d.id).distance(60).strength(0.6))
    .force('center', forceCenter(width / 2, height / 2))
    .force('collide', forceCollide().radius(20))
    .stop()

  for (let i = 0; i < 300; i++) {
    sim.tick()
    if (i % 30 === 0) {
      self.postMessage({ type: 'progress', tick: i, nodes })
    }
  }
  self.postMessage({ type: 'done', nodes, edges })
})
```

### Migration 1402 — extract_backlinks Trigger

```sql
-- supabase/migrations/1402_backlinks_trigger.sql
BEGIN;

CREATE OR REPLACE FUNCTION public.extract_backlinks_from_note()
RETURNS TRIGGER LANGUAGE plpgsql SECURITY DEFINER SET search_path = public, pg_temp AS $$
DECLARE
  v_match RECORD;
  v_target_id UUID;
  v_pattern TEXT := '\[\[([^\]|]+)(?:\|([^\]]+))?\]\]';  -- [[Note Title]] or [[id|text]]
BEGIN
  IF NEW.is_vault THEN
    -- Skip vault notes
    DELETE FROM public.backlinks WHERE source_id = NEW.id AND source_type = 'note';
    RETURN NEW;
  END IF;

  -- Delete old backlinks for this source
  DELETE FROM public.backlinks
  WHERE source_id = NEW.id AND source_type = 'note' AND user_id = NEW.user_id;

  -- Extract matches
  FOR v_match IN
    SELECT (regexp_matches(NEW.content, v_pattern, 'g'))[1] AS target_ref,
           (regexp_matches(NEW.content, v_pattern, 'g'))[2] AS display_text
  LOOP
    -- Try resolve target: by UUID if looks like one, else by title match
    IF v_match.target_ref ~ '^[0-9a-f-]{36}$' THEN
      v_target_id := v_match.target_ref::UUID;
    ELSE
      SELECT id INTO v_target_id FROM public.notes
      WHERE user_id = NEW.user_id AND title = v_match.target_ref AND NOT is_vault
      LIMIT 1;
    END IF;

    IF v_target_id IS NOT NULL AND v_target_id <> NEW.id THEN
      INSERT INTO public.backlinks(user_id, source_id, source_type, target_id, target_type, display_text)
      VALUES (NEW.user_id, NEW.id, 'note', v_target_id, 'note', v_match.display_text)
      ON CONFLICT DO NOTHING;
    END IF;
  END LOOP;

  RETURN NEW;
END $$;

CREATE TRIGGER trg_extract_backlinks
  AFTER INSERT OR UPDATE OF content ON public.notes
  FOR EACH ROW EXECUTE FUNCTION public.extract_backlinks_from_note();

COMMIT;
```

```sql
  user_id     UUID NOT NULL REFERENCES public.profiles(id) ON DELETE CASCADE,
  tag_id      UUID NOT NULL REFERENCES public.tags(id) ON DELETE CASCADE,
  entity_type TEXT NOT NULL CHECK (entity_type IN (
    'note','task','event','goal','project','habit'
  )),
  entity_id   UUID NOT NULL,
  created_at  TIMESTAMPTZ NOT NULL DEFAULT now(),
  UNIQUE (user_id, tag_id, entity_type, entity_id)
);
ALTER TABLE public.tag_assignments ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.tag_assignments FORCE ROW LEVEL SECURITY;
CREATE POLICY ta_owner ON public.tag_assignments FOR ALL
  USING (user_id = (SELECT auth.uid()))
  WITH CHECK (user_id = (SELECT auth.uid()));
CREATE INDEX ta_entity ON public.tag_assignments(entity_type, entity_id);
CREATE INDEX ta_tag ON public.tag_assignments(tag_id);

-- Auto-derive parent_path before INSERT/UPDATE
CREATE OR REPLACE FUNCTION public.set_tag_parent_path()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
BEGIN
  IF position('/' IN NEW.path) > 0 THEN
    NEW.parent_path := substring(NEW.path FROM 1 FOR length(NEW.path) - position('/' IN reverse(NEW.path)));
  ELSE
    NEW.parent_path := NULL;
  END IF;
  RETURN NEW;
END $$;
CREATE TRIGGER trg_set_tag_parent BEFORE INSERT OR UPDATE OF path ON public.tags
  FOR EACH ROW EXECUTE FUNCTION public.set_tag_parent_path();

-- Bump use_count on assignment
CREATE OR REPLACE FUNCTION public.bump_tag_use_count()
RETURNS TRIGGER LANGUAGE plpgsql SECURITY DEFINER SET search_path = public, pg_temp AS $$
BEGIN
  IF TG_OP = 'INSERT' THEN
    UPDATE public.tags SET use_count = use_count + 1 WHERE id = NEW.tag_id;
  ELSE
    UPDATE public.tags SET use_count = GREATEST(0, use_count - 1) WHERE id = OLD.tag_id;
  END IF;
  RETURN COALESCE(NEW, OLD);
END $$;
CREATE TRIGGER trg_bump_tag_use
  AFTER INSERT OR DELETE ON public.tag_assignments
  FOR EACH ROW EXECUTE FUNCTION public.bump_tag_use_count();

-- Cascade rename: when path changes, update all descendants
CREATE OR REPLACE FUNCTION public.cascade_tag_rename()
RETURNS TRIGGER LANGUAGE plpgsql SECURITY DEFINER SET search_path = public, pg_temp AS $$
BEGIN
  IF NEW.path <> OLD.path THEN
    UPDATE public.tags
    SET path = NEW.path || substring(path, length(OLD.path) + 1)
    WHERE user_id = NEW.user_id
      AND path LIKE OLD.path || '/%'
      AND id <> NEW.id;
  END IF;
  RETURN NEW;
END $$;
CREATE TRIGGER trg_cascade_rename AFTER UPDATE OF path ON public.tags
  FOR EACH ROW EXECUTE FUNCTION public.cascade_tag_rename();

COMMIT;
```

### TypeScript: mergeTagsAction

```tsx
// src/features/tags/actions/merge-tags.ts
'use server'
import { z } from 'zod'
import { requireUser } from '@/lib/auth/require-user'
import { serviceSupabase } from '@/lib/supabase/service'
import { audit } from '@/lib/audit'

const schema = z.object({
  sourceTagId: z.string().uuid(),
  targetTagId: z.string().uuid(),
})

export async function mergeTagsAction(input: unknown) {
  const { sourceTagId, targetTagId } = schema.parse(input)
  const user = await requireUser()
  if (sourceTagId === targetTagId) throw new Error('cannot_merge_to_self')

  // Reassign all assignments from source to target (ignore conflicts)
  await serviceSupabase
    .from('tag_assignments')
    .update({ tag_id: targetTagId })
    .eq('tag_id', sourceTagId)
    .eq('user_id', user.id)

  // Delete source tag
  await serviceSupabase
    .from('tags')
    .delete()
    .eq('id', sourceTagId)
    .eq('user_id', user.id)

  await audit({ actorId: user.id, action: 'tag.merged',
    targetId: targetTagId, targetType: 'tag',
    metadata: { from: sourceTagId, to: targetTagId } })

  return { mergedInto: targetTagId }
}
```

```tsx
import { Mark, mergeAttributes } from '@tiptap/core'
```

```tsx
// src/app/api/notes/suggest/route.ts
import 'server-only'
import { z } from 'zod'
import { requireUser } from '@/lib/auth/require-user'
import { withApiErrorHandling } from '@/lib/route'
import { serviceSupabase } from '@/lib/supabase/service'

export const GET = withApiErrorHandling(async (req) => {
  const user = await requireUser()
  const q = new URL(req.url).searchParams.get('q')?.slice(0, 100) ?? ''

  const { data } = await serviceSupabase
    .from('notes')
    .select('id, title')
    .eq('user_id', user.id)
    .eq('is_deleted', false)
    .eq('is_vault', false)  // never suggest vault
    .ilike('title', `%${q}%`)
    .order('updated_at', { ascending: false })
    .limit(10)

  return Response.json(data ?? [])
})
```

## Phase 52.9 — Knowledge Stats + XP + Life Score Integration

### Migration 1404 — knowledge_stats

```sql
-- supabase/migrations/1404_knowledge_stats.sql
BEGIN;

CREATE OR REPLACE VIEW public.user_knowledge_stats AS
  SELECT
    user_id,
    COUNT(DISTINCT from_entity) FILTER (WHERE from_type = 'note')   AS linked_notes,
    COUNT(*)                                                         AS total_edges,
    COUNT(*) FILTER (WHERE edge_type = 'backlink')                   AS backlink_count,
    COUNT(DISTINCT to_entity) FILTER (WHERE to_type = 'tag')         AS distinct_tags,
    AVG((SELECT COUNT(*) FROM public.graph_edges e2
         WHERE e2.user_id = e.user_id AND e2.from_entity = e.from_entity))::INT AS avg_outdegree
  FROM public.graph_edges e
  GROUP BY user_id;

COMMIT;
```
