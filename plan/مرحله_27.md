```sql
-- prune trigger ≥ 100 snapshots per page
CREATE OR REPLACE FUNCTION prune_page_snapshots() RETURNS TRIGGER LANGUAGE plpgsql AS $$
BEGIN
  DELETE FROM page_snapshots WHERE page_id = NEW.page_id
    AND id NOT IN (SELECT id FROM page_snapshots WHERE page_id = NEW.page_id
                   ORDER BY created_at DESC LIMIT 100);
  RETURN NEW;
END $$;
CREATE TRIGGER trg_prune_ps AFTER INSERT ON page_snapshots
  FOR EACH ROW EXECUTE FUNCTION prune_page_snapshots();

-- Lock guard على كل mutations (blocks + page)
CREATE OR REPLACE FUNCTION assert_page_not_locked() RETURNS TRIGGER LANGUAGE plpgsql AS $$
DECLARE v_locked BOOLEAN; v_locker UUID;
BEGIN
  SELECT is_locked, locked_by_user_id INTO v_locked, v_locker
  FROM pages WHERE id = COALESCE(NEW.page_id, OLD.page_id);
  IF v_locked AND v_locker IS DISTINCT FROM (SELECT auth.uid()) THEN
    RAISE EXCEPTION 'PAGE_LOCKED';
  END IF;
  RETURN COALESCE(NEW, OLD);
END $$;
CREATE TRIGGER trg_blocks_lock_guard BEFORE INSERT OR UPDATE OR DELETE ON blocks
  FOR EACH ROW EXECUTE FUNCTION assert_page_not_locked();
```

### Server Actions: lock + restore

```tsx
export async function lockPageAction(input: { pageId: string; reason?: string }) {
  const user = await requireUser()
  await assertPageRole(user.id, input.pageId, ['owner','admin'])
  await serviceSupabase.from('pages').update({
    is_locked: true, locked_by_user_id: user.id,
    locked_at: new Date().toISOString(), lock_reason: input.reason ?? null,
  }).eq('id', input.pageId)
  await audit({ actorId: user.id, action: 'page.locked', targetId: input.pageId, targetType: 'page' })
}

export async function restorePageVersionAction(input: { pageId: string; snapshotId: string }) {
  const user = await requireUser()
  await assertPageRole(user.id, input.pageId, ['owner','admin'])

  // 1. Take pre_restore snapshot first
  const current = await loadFullPageSnapshot(input.pageId)
  await serviceSupabase.from('page_snapshots').insert({
    workspace_id: current.workspace_id, page_id: input.pageId,
    taken_by: user.id, reason: 'pre_restore', snapshot: current,
    block_count: current.blocks.length,
  })

  // 2. Atomic restore via RPC
  await serviceSupabase.rpc('restore_page_from_snapshot', {
    p_page_id: input.pageId, p_snapshot_id: input.snapshotId,
  })
  await audit({ actorId: user.id, action: 'page.restored',
    targetId: input.pageId, targetType: 'page', metadata: { snapshotId: input.snapshotId } })
}
```

### Acceptance — Page Lock + Restore

- [ ]  migration 0025 (locks + snapshots + guards) متطبقة.
- [ ]  locked page: أي block mutation من غير الـ locker → `PAGE_LOCKED` (403).
- [ ]  locker نفسه يقدر يعدّل (للـ owner editing during lock period).
- [ ]  manual snapshot من UI: page menu → "حفظ نسخة".
- [ ]  auto snapshot قبل أي restore (`pre_restore` reason).
- [ ]  restore atomic: كل blocks تترجع لـ snapshot state في transaction واحدة.
- [ ]  vault page: lock مسموح، restore يحترم E2EE (snapshot encrypted blob).
- [ ]  Playwright: lock → user2 mutation → blocked. unlock → mutation → allowed. restore → blocks reverted.

## 🧠 Master Architect Upgrade — Public Share & Canvas Canonical Gate

- **Canonical IDs:** قبل التنفيذ، بدّل أي `UUID/gen_random_uuid/profiles/auth.uid()` في tenant business tables إلى `TEXT ULID + workspace_id + users + current_workspace_id()/current_user_id()` حسب W00/W01.
- **Share tokens:** token/slug checks لا تكشف هل المورد موجود. كل response عام noindex، rate-limited، cache-control private/no-store عند protected shares، وبدون PII.
- **Vault hard wall:** Vault/E2EE content ممنوع public share وممنوع canvas node وممنوع OG preview وممنوع search snippet.
- **Canvas persistence:** positions centipixels INT، mutations idempotent، history snapshots capped، وserver-side max nodes atomic داخل transaction/RPC.
- **Desktop canvas UX:** pan/zoom/selection/undo/redo لازم keyboard accessible، high contrast dark-only، وrendering virtualized/worker-backed للخرائط الكبيرة.

## 🎯 نظرة عامة

المرحلة دي بتدّي المستخدم قدرتين أساسيين: إنه يشارك صفحاته أو مهامه مع أي حد بره الـ workspace عن طريق رابط عام آمن، وإنه يبني خرائط ذهنية ولوحات بصرية (Canvas) يربط فيها الأفكار والملاحظات والمهام بصرياً بدل النصوص المتسلسلة.

**المستخدم هيكسب إيه:**

- يقدر يشارك ملاحظة أو مهمة برابط عام بصلاحية زمنية (يوم، أسبوع، شهر، أو دائم) وبعدد مشاهدات محدود لو حب.
- يقدر يلغي الشير في أي وقت ويتأكد إن الرابط ميشتغلش حتى لو الناس عملتله cache.
- يضمن إن أي محتوى Vault (المحمي بكلمة سر) ميتشاركش أبداً، حتى لو غلط.
- يقدر يبني Canvas بنود (Nodes) وروابط (Edges) عشان يخطط مشروع أو يعمل mindmap بصري بسرعة.
- يتنقل بين الصفحات والمهام والملاحظات داخل اللوحة بمساحة لا نهائية (infinite canvas) مع zoom/pan سلس.
- يعمل Undo/Redo بدون ما يفقد شغله.

**اللي هيتبني في المرحلة دي:**

- جدول `public_shares` فيه token عشوائي 32 بايت + expiry + view counter + revocation flag + custom slug.
- صفحة `/share/[token]` و `/s/[slug]` بـ `robots: noindex` وبتعرض المحتوى بدون ما تسرّب user_id أو email.
- RPC اسمه `record_share_view` بيشتغل بدون auth ويحدّث counter ويرجع المحتوى.
- Edge Function `cleanup_expired_shares` بيشتغل بـ pg_cron كل ساعة عشان يعمل revoke تلقائي للروابط المنتهية.
- جداول `canvases` + `canvas_nodes` + `canvas_edges` بمواقع integer centipixels (مفيش float drift).
- جدول `canvas_history` بـ trigger بيقص الـ snapshots لآخر 50 حالة عشان مياخدش مساحة كبيرة.
- حد أقصى 500 node لكل canvas (server-enforced) وحد مفيش Vault notes تتضاف للوحة.
- Hook `useUndoStack` للـ undo/redo client-side مع batch debouncing.
- Rate limit: 10 shares في الساعة لكل مستخدم.

### 2) RPC: `record_share_view` (للزوار بدون auth)

```sql
CREATE OR REPLACE FUNCTION public.record_share_view(p_token TEXT)
RETURNS TABLE(found BOOLEAN, entity_type TEXT, entity_data JSONB)
LANGUAGE plpgsql SECURITY DEFINER SET search_path = public, pg_temp AS $$
DECLARE v_share public.public_shares%ROWTYPE;
BEGIN
  SELECT * INTO v_share FROM public.public_shares
  WHERE (token = p_token OR custom_slug = p_token)
    AND NOT is_revoked
    AND (expires_at IS NULL OR expires_at > now())
    AND (max_views IS NULL OR view_count < max_views);
  IF NOT FOUND THEN
    RETURN QUERY SELECT FALSE, NULL::TEXT, NULL::JSONB; RETURN;
  END IF;
  UPDATE public.public_shares
    SET view_count = view_count + 1, last_viewed = now()
  WHERE id = v_share.id;
  CASE v_share.entity_type
    WHEN 'note' THEN
      RETURN QUERY SELECT TRUE, 'note'::TEXT,
        jsonb_build_object('title', n.title, 'content', n.content)
      FROM public.notes n
      WHERE n.id = v_share.entity_id AND NOT n.is_vault AND NOT n.is_deleted;
    WHEN 'task' THEN
      RETURN QUERY SELECT TRUE, 'task'::TEXT,
        jsonb_build_object('title', t.title, 'status', t.status, 'due', t.due_date)
      FROM public.tasks t
      WHERE t.id = v_share.entity_id AND NOT t.is_deleted;
    ELSE
      RETURN QUERY SELECT TRUE, v_share.entity_type, '{}'::JSONB;
  END CASE;
END $$;
REVOKE ALL ON FUNCTION public.record_share_view FROM authenticated;
GRANT EXECUTE ON FUNCTION public.record_share_view TO anon;
```

### 3) Server action: `createShareAction`

```tsx
// src/features/sharing/actions/create-share.ts
'use server'
import { z } from 'zod'
import { requireUser } from '@/lib/auth/require-user'
import { serviceSupabase } from '@/lib/supabase/service'
import { assertNotificationLimit } from '@/lib/notifications/rate-limit'
import { AppError } from '@/lib/result'
import { audit } from '@/lib/audit'

const schema = z.object({
  entityType: z.enum(['note','task','canvas','goal']),
  entityId:   z.string().uuid(),
  expiresIn:  z.enum(['1d','7d','30d','never']).default('7d'),
  maxViews:   z.number().int().positive().max(10_000).optional(),
  customSlug: z.string().regex(/^[a-z0-9-]{3,50}$/).optional(),
})

const MS = { '1d': 86_400_000, '7d': 7*86_400_000, '30d': 30*86_400_000, never: null }

export async function createShareAction(input: unknown) {
  const parsed = schema.parse(input)
  const user   = await requireUser()
  await assertNotificationLimit(user.id, 'note_share', 10)

  // Vault guard: لو note محمي بـ Vault، ممنوع الشير
  if (parsed.entityType === 'note') {
    const { data } = await serviceSupabase
      .from('notes').select('is_vault')
      .eq('id', parsed.entityId).eq('user_id', user.id).single()
    if (!data) throw new AppError('NOTE_NOT_FOUND','not found', 404)
    if (data.is_vault) throw new AppError('VAULT_CANNOT_SHARE','Vault notes cannot be shared', 403)
  }

  const ms = MS[parsed.expiresIn]
  const expiresAt = ms ? new Date(Date.now() + ms).toISOString() : null

  const { data, error } = await serviceSupabase.from('public_shares').insert({
    user_id:     user.id,
    entity_type: parsed.entityType,
    entity_id:   parsed.entityId,
    expires_at:  expiresAt,
    max_views:   parsed.maxViews ?? null,
    custom_slug: parsed.customSlug ?? null,
  }).select('token, custom_slug, expires_at').single()

  if (error) {
    if (error.code === '23505') throw new AppError('SLUG_TAKEN','slug taken', 409)
    throw new AppError('SHARE_FAILED', error.message, 500)
  }

  await audit({ actorId: user.id, action: 'share.created',
    targetId: parsed.entityId, targetType: parsed.entityType })

  return {
    token: data.token, slug: data.custom_slug, expiresAt: data.expires_at,
    url: data.custom_slug
      ? `${process.env.APP_URL}/s/${data.custom_slug}`
      : `${process.env.APP_URL}/share/${data.token}`,
  }
}
```

```tsx
// src/features/canvas/actions/add-node.ts
'use server'
import { z } from 'zod'
import { requireUser } from '@/lib/auth/require-user'
import { serviceSupabase } from '@/lib/supabase/service'
import { AppError } from '@/lib/result'

const MAX_NODES = 500

const schema = z.object({
  canvasId: z.string().uuid(),
  kind:     z.enum(['note','task','shape','text','image','link']),
  xCp:      z.number().int(),
  yCp:      z.number().int(),
  content:  z.string().max(5000).optional(),
  noteId:   z.string().uuid().optional(),
  taskId:   z.string().uuid().optional(),
})

export async function addCanvasNode(input: unknown) {
  const parsed = schema.parse(input)
  const user   = await requireUser()

  const { data: canvas } = await serviceSupabase
    .from('canvases').select('id')
    .eq('id', parsed.canvasId).eq('user_id', user.id).single()
  if (!canvas) throw new AppError('CANVAS_NOT_FOUND','not found', 404)

  const { count } = await serviceSupabase
    .from('canvas_nodes').select('*', { count: 'exact', head: true })
    .eq('canvas_id', parsed.canvasId)
  if ((count ?? 0) >= MAX_NODES)
    throw new AppError('CANVAS_FULL', `Max ${MAX_NODES} nodes per canvas`, 422)

  if (parsed.noteId) {
    const { data: note } = await serviceSupabase
      .from('notes').select('is_vault')
      .eq('id', parsed.noteId).eq('user_id', user.id).single()
    if (!note) throw new AppError('NOTE_NOT_FOUND','not found', 404)
    if (note.is_vault) throw new AppError('VAULT_IN_CANVAS','Vault notes cannot be in canvas', 403)
  }

  const { data, error } = await serviceSupabase.from('canvas_nodes').insert({
    canvas_id: parsed.canvasId,
    user_id:   user.id,
    kind:      parsed.kind,
    x_cp:      parsed.xCp, y_cp: parsed.yCp,
    content:   parsed.content ?? null,
    note_id:   parsed.noteId ?? null,
    task_id:   parsed.taskId ?? null,
  }).select().single()

  if (error) throw new AppError('NODE_CREATE_FAILED', error.message, 500)
  return data
}
```

### 7) Hook الـ Undo/Redo (client-side stack)

```tsx
// src/features/canvas/hooks/useUndoStack.ts
'use client'
import { useCallback, useRef, useState } from 'react'

const MAX_STACK = 50

interface Snapshot<T> { state: T; description: string; ts: number }

export function useUndoStack<T>(initial: T) {
  const [current, setCurrent] = useState<T>(initial)
  const past   = useRef<Snapshot<T>[]>([])
  const future = useRef<Snapshot<T>[]>([])

  const push = useCallback((next: T, description: string) => {
    past.current = [
      ...past.current.slice(-MAX_STACK + 1),
      { state: current, description, ts: Date.now() },
    ]
    future.current = []
    setCurrent(next)
  }, [current])

  const undo = useCallback(() => {
    if (past.current.length === 0) return
    const prev = past.current.at(-1)!
    future.current = [{ state: current, description: 'undo', ts: Date.now() }, ...future.current.slice(0, MAX_STACK - 1)]
    past.current = past.current.slice(0, -1)
    setCurrent(prev.state)
  }, [current])

  const redo = useCallback(() => {
    if (future.current.length === 0) return
    const next = future.current[0]
    past.current = [...past.current.slice(-MAX_STACK + 1), { state: current, description: 'redo', ts: Date.now() }]
    future.current = future.current.slice(1)
    setCurrent(next.state)
  }, [current])

  return { state: current, push, undo, redo,
    canUndo: past.current.length > 0,
    canRedo: future.current.length > 0 }
}
```

```sql
CREATE OR REPLACE FUNCTION public.cleanup_expired_shares()
RETURNS INT LANGUAGE plpgsql SECURITY DEFINER SET search_path = public, pg_temp AS $$
DECLARE v_count INT;
BEGIN
  UPDATE public.public_shares
  SET is_revoked = TRUE
  WHERE NOT is_revoked
    AND expires_at IS NOT NULL AND expires_at < now();
  GET DIAGNOSTICS v_count = ROW_COUNT;
  RETURN v_count;
END $$;

-- pg_cron schedule
SELECT cron.schedule('cleanup_expired_shares', '0 * * * *',
  $$SELECT public.cleanup_expired_shares()$$);
```
