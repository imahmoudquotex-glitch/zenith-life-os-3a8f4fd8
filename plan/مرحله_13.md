### Timezone-Safe Focus Slots

```sql
CREATE OR REPLACE FUNCTION suggest_focus_slots(p_user TEXT, p_workspace TEXT, p_min_minutes INT)
RETURNS TABLE(slot_start TIMESTAMPTZ, slot_end TIMESTAMPTZ, energy_score INT)
LANGUAGE plpgsql SECURITY DEFINER SET search_path = public, pg_temp AS $$
DECLARE v_tz TEXT;
BEGIN
  SELECT COALESCE(timezone, 'UTC') INTO v_tz FROM users WHERE id = p_user;
  RETURN QUERY
  SELECT s.slot, s.slot + make_interval(mins => p_min_minutes), COALESCE(e.energy, 50)
  FROM generate_series(now(), now() + interval '7 days', interval '30 minutes') s(slot)
  LEFT JOIN energy_curve_hours e
    ON e.user_id = p_user
   AND e.local_hour = EXTRACT(HOUR FROM s.slot AT TIME ZONE v_tz)
  WHERE NOT EXISTS (
    SELECT 1 FROM calendar_events c
    WHERE c.workspace_id = p_workspace
      AND c.start_at < s.slot + make_interval(mins => p_min_minutes)
      AND c.end_at > s.slot
      AND NOT c.is_cancelled
  );
END $$;
```

```tsx
import 'server-only'

export function generateICal(events: CalEvent[], userTimezone: string): string {
  const lines: string[] = [
    'BEGIN:VCALENDAR',
    'VERSION:2.0',
    'PRODID:-//Zenith Life OS//AR',
    'CALSCALE:GREGORIAN',
    'METHOD:PUBLISH',
    `X-WR-TIMEZONE:${userTimezone}`,
    'BEGIN:VTIMEZONE',
    `TZID:${userTimezone}`,
    'END:VTIMEZONE',
  ]
  for (const ev of events) {
    lines.push(
      'BEGIN:VEVENT',
      `UID:${ev.id}@zenith.app`,
      `DTSTART;TZID=${ev.timezone}:${formatICal(ev.startAt)}`,
      `DTEND;TZID=${ev.timezone}:${formatICal(ev.endAt)}`,
      `SUMMARY:${foldLine(ev.title)}`,
      ev.recurrence ? `RRULE:${ev.recurrence}` : '',
      ...(ev.description ? [`DESCRIPTION:${foldLine(ev.description)}`] : []),
      `DTSTAMP:${formatICal(new Date().toISOString())}`,
      'END:VEVENT'
    )
  }
  lines.push('END:VCALENDAR')
  return lines.filter(Boolean).join('\r\n')
}
```

```tsx
import { getEnergyCurve } from '@/features/energy/lib/get-energy-curve'

interface TimelineBlock {
  id: string; startsAt: string; endsAt: string
  title: string; label: string
  source: 'calendar' | 'focus' | 'habit' | 'task'
  color?: string
}

export const getDayTimeline = unstable_cache(
  async (userId: string, date: string): Promise<{
    blocks: TimelineBlock[]
    energy: Array<{ hour: number; energy: number }>
    densityPct: number
  }> => {

      fetchCalendarEvents(userId, date),
      fetchFocusSessions(userId, date),
      fetchHabitCheckins(userId, date),
      fetchScheduledTasks(userId, date),  // tasks with planned_start
      getEnergyCurve(userId, date),       // W16
    ])
    const blocks: TimelineBlock[] = [

      ...focus.map(f => ({ ...f, source: 'focus' as const })),
```

## Phase 14.7 — Smart Scheduling (Energy + Focus Window aware)

### Migration 1301 — suggestion_slots view

```sql
-- supabase/migrations/1301_smart_schedule.sql
BEGIN;

-- View: free hour slots in next 7 days, joined with energy curve
CREATE OR REPLACE FUNCTION public.suggest_focus_slots(
  p_user UUID,
  p_min_minutes INT DEFAULT 45,
  p_lookahead_days INT DEFAULT 7
) RETURNS TABLE (
  slot_start TIMESTAMPTZ,
  slot_end   TIMESTAMPTZ,
  energy_score INT,
  reason TEXT
) LANGUAGE plpgsql SECURITY DEFINER SET search_path = public, pg_temp AS $$
BEGIN
  RETURN QUERY
  WITH busy AS (
    SELECT start_at, end_at FROM public.calendar_events
    WHERE user_id = p_user AND NOT is_cancelled
      AND start_at BETWEEN now() AND now() + (p_lookahead_days || ' days')::interval
  ),
  hours AS (
    SELECT generate_series(
      date_trunc('hour', now() + interval '1 hour'),
      now() + (p_lookahead_days || ' days')::interval,
      interval '30 minutes'
    ) AS slot
  )
  SELECT
    h.slot AS slot_start,
    h.slot + (p_min_minutes || ' minutes')::interval AS slot_end,
    COALESCE((
      SELECT energy FROM public.energy_curve_hours
      WHERE user_id = p_user AND hour = EXTRACT(HOUR FROM h.slot AT TIME ZONE 'Africa/Cairo')
    ), 50)::INT AS energy_score,
    'وقت تركيز عالي' AS reason
  FROM hours h
  WHERE NOT EXISTS (
    SELECT 1 FROM busy b
    WHERE b.start_at < h.slot + (p_min_minutes || ' minutes')::interval
      AND b.end_at   > h.slot
  )
  AND EXTRACT(HOUR FROM h.slot AT TIME ZONE 'Africa/Cairo') BETWEEN 7 AND 22
  ORDER BY energy_score DESC, slot_start
  LIMIT 20;
END $$;

COMMIT;
```

```tsx
// src/features/calendar/lib/check-conflicts.ts
import 'server-only'
import { serviceSupabase } from '@/lib/supabase/service'

interface ConflictResult {
  hasConflict: boolean
  conflictingEvents: Array<{ id: string; title: string; startAt: string; endAt: string }>
}

export async function checkEventConflicts(
  userId: string, startAt: string, endAt: string, excludeId?: string
): Promise<ConflictResult> {
  let query = serviceSupabase
    .from('calendar_events')
    .select('id, title, start_at, end_at')
    .eq('user_id', userId)
    .eq('is_cancelled', false)
    .lt('start_at', endAt)
    .gt('end_at', startAt)

    hasConflict: (data?.length ?? 0) > 0,
    conflictingEvents: data ?? [],
  }
```

```tsx
// src/app/api/search/route.ts
import 'server-only'
import { z } from 'zod'
import { requireUser } from '@/lib/auth/require-user'
import { withApiErrorHandling, parseJsonBody } from '@/lib/route'
import { assertSearchRateLimit } from '@/lib/notifications/rate-limit'
import { createSupabaseServerClient } from '@/lib/supabase/server'

const schema = z.object({
  query: z.string().min(2).max(200),
  types: z.array(z.enum(['task', 'note', 'habit', 'event', 'goal', 'project'])).default([
    'task', 'note', 'habit', 'event', 'goal', 'project'
  ]),
  limit: z.number().int().min(1).max(50).default(20),
})

export const POST = withApiErrorHandling(async (req) => {
  const user = await requireUser()
  await assertSearchRateLimit(user.id, 'search', 60)  // 60/min

  const { query, types, limit } = await parseJsonBody(req, schema)
  const supabase = await createSupabaseServerClient()
  const results: Array<{ id: string; type: string; title: string; snippet?: string }> = []

  for (const type of types) {
    const config = TABLE_CONFIG[type]
    const { data } = await supabase.from(config.table)
      .select(`id, ${config.titleField}, ${config.snippetField ?? ''}`)
      .textSearch(config.titleField, query, { type: 'websearch' })
      .eq('user_id', user.id)
      .eq(config.deletedField, false)
      .limit(Math.ceil(limit / types.length))

    results.push(...(data ?? []).map((r: any) => ({
      id: r.id,
      type,
      title: r[config.titleField],
      snippet: r[config.snippetField ?? '']?.slice(0, 200),
    })))
  }

  return Response.json({ data: results.slice(0, limit) })
})
```

```tsx
import { Metadata } from 'next'
import { serviceSupabase } from '@/lib/supabase/service'
import { notFound } from 'next/navigation'

export async function generateMetadata({ params }: { params: { token: string } }): Promise<Metadata> {
  const { data } = await getShareData(params.token)
  return {
    title: data?.noteTitle ?? 'ملاحظة مشتركة — Zenith',
    robots: 'noindex, nofollow',
  }
}

export default async function SharedPage({ params }: { params: { token: string } }) {
  const { data } = await getShareData(params.token)
  if (!data) notFound()
  return (
    <article dir="auto" className="max-w-2xl mx-auto p-8">
      <h1 className="text-2xl font-bold mb-6">{data.noteTitle}</h1>
      <div className="prose prose-invert">{data.noteContent}</div>
    </article>
  )
}

async function getShareData(token: string) {
  const { data } = await serviceSupabase
    .from('public_shares')
    .select('note_id, expires_at, is_revoked, view_count, notes(title, content, is_vault)')
    .eq('token', token).single()

  if (!data || data.is_revoked) return { data: null }
  if (new Date(data.expires_at) < new Date()) return { data: null }

  const note = Array.isArray(data.notes) ? data.notes[0] : data.notes
  if (!note || note.is_vault) return { data: null }

  // Increment view counter
  await serviceSupabase
    .from('public_shares')
    .update({ view_count: (data.view_count ?? 0) + 1 })
    .eq('token', token)

  return { data: { noteTitle: note.title, noteContent: note.content } }
}
```

### Max Nodes Guard

```tsx
// src/features/canvas/actions/add-node.ts
'use server'
import { requireUser } from '@/lib/auth/require-user'
import { serviceSupabase } from '@/lib/supabase/service'
import { AppError } from '@/lib/result'

const MAX_NODES_PER_CANVAS = 500

export async function addCanvasNode(canvasId: string, nodeData: any) {
  const user = await requireUser()
  const { count } = await serviceSupabase
    .from('canvas_nodes')
    .select('*', { count: 'exact', head: true })
    .eq('canvas_id', canvasId)
    .eq('user_id', user.id)
  if ((count ?? 0) >= MAX_NODES_PER_CANVAS) {
    throw new AppError('CANVAS_NODE_LIMIT', `Max ${MAX_NODES_PER_CANVAS} nodes per canvas`, 422)
  }
  // Reject linking to vault
  if (nodeData.linkedEntityType === 'note') {
    const { data: note } = await serviceSupabase
      .from('notes').select('is_vault').eq('id', nodeData.linkedEntityId).single()
    if (note?.is_vault) throw new AppError('VAULT_BLOCKED', 'Cannot link vault note', 403)
  }
  const { data, error } = await serviceSupabase
    .from('canvas_nodes')
    .insert({ ...nodeData, canvas_id: canvasId, user_id: user.id })
    .select().single()
  if (error) throw error
  return data
}
```
