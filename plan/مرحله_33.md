### Migration 057 — comments_threading

```sql
ALTER TABLE comments
  ADD COLUMN IF NOT EXISTS thread_root_id UUID REFERENCES comments(id),
  ADD COLUMN IF NOT EXISTS depth INT NOT NULL DEFAULT 0 CHECK (depth BETWEEN 0 AND 5),
  ADD COLUMN IF NOT EXISTS assignee_id UUID REFERENCES profiles(id),
  ADD COLUMN IF NOT EXISTS due_at TIMESTAMPTZ,
  ADD COLUMN IF NOT EXISTS reaction_counts JSONB NOT NULL DEFAULT '{}'::jsonb;

CREATE INDEX IF NOT EXISTS idx_comments_thread ON comments(thread_root_id, created_at)
  WHERE NOT is_deleted;
CREATE INDEX IF NOT EXISTS idx_comments_assignee_open ON comments(assignee_id)
  WHERE NOT is_resolved AND NOT is_deleted;

-- auto-set thread_root_id + depth
CREATE OR REPLACE FUNCTION set_comment_thread() RETURNS TRIGGER LANGUAGE plpgsql AS $$
DECLARE v_parent comments%ROWTYPE;
BEGIN
  IF NEW.parent_id IS NULL THEN
    NEW.thread_root_id := NEW.id;
    NEW.depth := 0;
  ELSE
    SELECT * INTO v_parent FROM comments WHERE id = NEW.parent_id;
    IF v_parent.depth >= 5 THEN RAISE EXCEPTION 'COMMENT_THREAD_TOO_DEEP'; END IF;
    NEW.thread_root_id := COALESCE(v_parent.thread_root_id, v_parent.id);
    NEW.depth := v_parent.depth + 1;
  END IF;
  RETURN NEW;
END $$;
CREATE TRIGGER trg_comment_thread BEFORE INSERT ON comments
  FOR EACH ROW EXECUTE FUNCTION set_comment_thread();

-- resolve cascade لكل الـ thread
CREATE OR REPLACE FUNCTION resolve_thread(p_root_id UUID, p_user_id UUID) RETURNS INT
LANGUAGE plpgsql SECURITY DEFINER AS $$
DECLARE v_count INT;
BEGIN
  UPDATE comments SET is_resolved = TRUE, resolved_by = p_user_id, resolved_at = now()
  WHERE thread_root_id = p_root_id AND NOT is_resolved;
  GET DIAGNOSTICS v_count = ROW_COUNT;
  RETURN v_count;
END $$;
```

### CRDT Update Guard

```sql
CREATE OR REPLACE FUNCTION assert_crdt_update_allowed()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
DECLARE page_ws TEXT;
BEGIN
  SELECT workspace_id INTO page_ws FROM pages WHERE id = NEW.page_id;
  IF page_ws IS DISTINCT FROM NEW.workspace_id THEN
    RAISE EXCEPTION 'CRDT_WORKSPACE_MISMATCH';
  END IF;
  IF octet_length(NEW.update_data) > 262144 THEN
    RAISE EXCEPTION 'CRDT_UPDATE_TOO_LARGE';
  END IF;
  RETURN NEW;
END $$;
```

## 🧠 Master Architect Upgrade — Collaboration Enterprise Boundary

- **CRDT tenant contract:** كل Yjs update/snapshot يحمل `workspace_id`, `page_id`, `actor_user_id`, client_id، وDB trigger يرفض أي page/workspace mismatch.
- **Vault exclusion:** صفحات Vault/E2EE لا تدخل CRDT، presence، comments، mentions، snapshots، search، أو AI collaboration context.
- **Permission resolver:** كل read/write/comment/share uses `get_page_role` أو resolver واحد؛ RLS لا يعتمد على UI ولا workspace_members فقط.
- **Operational controls:** compaction worker idempotent، snapshot checksums، update size limits، rate limits per page/client، وrecovery playbook من snapshot+updates.
- **Admin-grade sharing:** ownership transfer requires MFA، invites email-bound + one-time + expiry، password shares bcrypt/argon2id، وaudit لكل permission change.

## 🎯 نظرة عامة

المرحلة دي بتنقل Zenith من single-user لـ multi-user collaboration platform بمستوى Notion/Google Docs. بنبني CRDT-based real-time editing بـ Yjs، نظام تعليقات (inline + page-level) مع @mentions، ونظام workspaces بأدوار (owner/admin/editor/commenter/viewer) ودعوات بـ tokens محدودة المدة وصلاحيات per-page.

**المستخدم هيكسب إيه:**

- يدعو زملاء على نفس الـ workspace ويشتغلوا معاه على نفس الصفحة في نفس اللحظة بدون conflicts.
- يشوف cursors الزملاء وأسماءهم في الصفحة (presence) ومين بيكتب فين.
- يكتب offline والتعديلات تترتب أوتوماتيك لما الانترنت يرجع.
- يعلّق على block محدد (inline comment) أو على الصفحة كلها (page comment) ويعمل threads ردود.
- يـ @mention زميل ويترسله إشعار.
- يعمل resolve للتعليقات لما الموضوع يتقفل.
- ينشئ workspace جديد ويدعو أعضاء بأدوار مختلفة.
- يدّي صلاحيات per-page (صفحة معينة public أو لـ user محدد).
- يحمي شير بكلمة سر (bcrypt-stored).

**اللي هيتبني في المرحلة دي:**

- جدول `crdt_updates` (binary Yjs updates) + `crdt_snapshots` (compacted state) مع RLS بـ workspace membership.
- Supabase Realtime provider بيـ broadcast الـ Yjs updates بـ postgres_changes channel.
- BullMQ compaction job يدمج الـ updates في snapshot كل ساعة.
- Y-IndexedDB للـ offline persistence client-side.
- Awareness protocol للـ cursor positions + presence.
- جدول `comments` (inline + page-level + threaded replies) + `mentions` للـ notification tracking.
- جدول `workspaces` + `workspace_members` + `workspace_invitations` بـ enum للأدوار.
- جدول `page_permissions` (per-page overrides) + RPC `get_page_role(page_id, user_id)`.
- Password-protected shares (bcrypt) + token expiry 7 أيام للدعوات.
- 2FA confirmation لتحويل ownership.
- مفيش Vault content يدخل CRDT collab ولا يتعرض في تعليقات.

### 2) Supabase Realtime Yjs provider (client)

```tsx
// src/lib/crdt/supabase-provider.ts
'use client'
import * as Y from 'yjs'
import { Awareness } from 'y-protocols/awareness'
import { IndexeddbPersistence } from 'y-indexeddb'
import { createClient } from '@supabase/supabase-js'

export class SupabaseYjsProvider {
  doc: Y.Doc
  awareness: Awareness
  private supa = createClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
  )
  private persist: IndexeddbPersistence
  private channel: ReturnType<typeof this.supa.channel> | null = null
  private clientId = crypto.randomUUID()

  constructor(
    public pageId: string,
    public workspaceId: string,
    public userId: string,
    public displayName: string,
  ) {
    this.doc = new Y.Doc({ guid: pageId })
    this.awareness = new Awareness(this.doc)
    this.awareness.setLocalState({
      userId, name: displayName, color: pickColor(userId),
    })
    this.persist = new IndexeddbPersistence(`zenith:${pageId}`, this.doc)
    void this.bootstrap()
    this.subscribe()
    this.doc.on('update', (u, origin) => {
      if (origin !== this) void this.push(u)
    })
  }

  private async bootstrap() {
    const { data: snap } = await this.supa.from('crdt_snapshots')
      .select('snapshot').eq('page_id', this.pageId).maybeSingle()
    if (snap?.snapshot) Y.applyUpdate(this.doc, new Uint8Array(snap.snapshot), this)
    const { data: updates } = await this.supa.from('crdt_updates')
      .select('update_data').eq('page_id', this.pageId)
      .order('created_at', { ascending: true })
    for (const row of updates ?? []) {
      Y.applyUpdate(this.doc, new Uint8Array(row.update_data), this)
    }
  }

  private subscribe() {
    this.channel = this.supa.channel(`crdt:${this.pageId}`)
      .on('postgres_changes', {
        event: 'INSERT', schema: 'public', table: 'crdt_updates',
        filter: `page_id=eq.${this.pageId}`,
      }, (payload) => {
        if (payload.new.client_id === this.clientId) return
        Y.applyUpdate(this.doc, new Uint8Array(payload.new.update_data), this)
      })
      .subscribe()
  }

  private async push(update: Uint8Array) {
    await this.supa.from('crdt_updates').insert({
      page_id: this.pageId,
      workspace_id: this.workspaceId,
      update_data: Array.from(update),
      client_id: this.clientId,
    })
  }

  destroy() {
    this.channel?.unsubscribe()
    this.persist.destroy()
    this.doc.destroy()
  }
}

function pickColor(seed: string) {
  const c = ['#ef4444','#f97316','#22c55e','#0ea5e9','#a855f7','#ec4899']
  let h = 0; for (const ch of seed) h = (h * 31 + ch.charCodeAt(0)) >>> 0
  return c[h % c.length]
}
```

### 3) Compaction worker (server-side)

```tsx
// src/workers/crdt-compactor.ts
import * as Y from 'yjs'
import { serviceSupabase } from '@/lib/supabase/service'

export async function compactPage(pageId: string) {
  const { data: snap } = await serviceSupabase.from('crdt_snapshots')
    .select('snapshot, workspace_id').eq('page_id', pageId).maybeSingle()
  const { data: updates } = await serviceSupabase.from('crdt_updates')
    .select('id, update_data, created_at, workspace_id')
    .eq('page_id', pageId).order('created_at', { ascending: true })
  if (!updates?.length) return

  const doc = new Y.Doc()
  if (snap?.snapshot) Y.applyUpdate(doc, new Uint8Array(snap.snapshot))
  for (const u of updates) Y.applyUpdate(doc, new Uint8Array(u.update_data))

  const merged = Y.encodeStateAsUpdate(doc)
  const cutoff = updates.at(-1)!.created_at

  await serviceSupabase.from('crdt_snapshots').upsert({
    page_id: pageId,
    workspace_id: updates[0].workspace_id,
    snapshot: Array.from(merged),
    updated_at: new Date().toISOString(),
  }, { onConflict: 'page_id' })

  await serviceSupabase.from('crdt_updates')
    .delete().eq('page_id', pageId).lte('created_at', cutoff)
}
```

```tsx
// src/features/comments/actions/create-comment.ts
'use server'
import { z } from 'zod'
import { requireUser } from '@/lib/auth/require-user'
import { serviceSupabase } from '@/lib/supabase/service'
import { AppError } from '@/lib/result'
import { sendNotification } from '@/lib/notifications/send'

const schema = z.object({
  workspaceId: z.string().uuid(),
  pageId:      z.string().uuid(),
  blockId:     z.string().uuid().optional(),
  parentId:    z.string().uuid().optional(),
  content:     z.object({
    text:     z.string().min(1).max(10_000),
    mentions: z.array(z.string().uuid()).default([]),
  }),
})

export async function createCommentAction(input: unknown) {
  const parsed = schema.parse(input)
  const user   = await requireUser()

  const { data: page } = await serviceSupabase.from('pages')
    .select('is_vault, workspace_id').eq('id', parsed.pageId).single()
  if (!page) throw new AppError('PAGE_NOT_FOUND','not found', 404)
  if (page.is_vault) throw new AppError('VAULT_NO_COMMENTS','Cannot comment on vault', 403)

  const { data: comment, error } = await serviceSupabase.from('comments').insert({
    workspace_id: parsed.workspaceId,
    page_id:      parsed.pageId,
    block_id:     parsed.blockId ?? null,
    parent_id:    parsed.parentId ?? null,
    user_id:      user.id,
    content:      parsed.content,
  }).select('id').single()
  if (error) throw new AppError('COMMENT_FAILED', error.message, 500)

  for (const mUser of parsed.content.mentions) {
    await serviceSupabase.from('mentions').insert({
      comment_id: comment!.id, mentioned_user_id: mUser,
      workspace_id: parsed.workspaceId,
    })
    await sendNotification(mUser, {
      type: 'mention',
      title: `${user.name ?? 'حد'} ذكرك في تعليق`,
      link: `/p/${parsed.pageId}#c-${comment!.id}`,
    })
  }
  return { commentId: comment!.id }
}
```

### 7) Server action: قبول الدعوة

```tsx
// src/features/workspaces/actions/accept-invitation.ts
'use server'
import { z } from 'zod'
import { requireUser } from '@/lib/auth/require-user'
import { serviceSupabase } from '@/lib/supabase/service'
import { AppError } from '@/lib/result'

const schema = z.object({ token: z.string().regex(/^[a-f0-9]{64}$/) })

export async function acceptInvitationAction(input: unknown) {
  const { token } = schema.parse(input)
  const user = await requireUser()

  const { data: inv } = await serviceSupabase.from('workspace_invitations')
    .select('*').eq('token', token).maybeSingle()
  if (!inv) throw new AppError('INVITE_NOT_FOUND','not found', 404)
  if (inv.accepted_at) throw new AppError('INVITE_USED','already used', 409)
  if (new Date(inv.expires_at) < new Date())
    throw new AppError('INVITE_EXPIRED','expired', 410)

  await serviceSupabase.from('workspace_members').insert({
    workspace_id: inv.workspace_id, user_id: user.id,
    role: inv.role, invited_by: inv.invited_by,
    joined_at: new Date().toISOString(),
  })
  await serviceSupabase.from('workspace_invitations')
    .update({ accepted_at: new Date().toISOString() })
    .eq('id', inv.id)

  return { workspaceId: inv.workspace_id, role: inv.role }
}
```

### 8) Password-protected share verification

```tsx
// src/features/sharing/actions/verify-share.ts
'use server'
import bcrypt from 'bcryptjs'
import { serviceSupabase } from '@/lib/supabase/service'
import { AppError } from '@/lib/result'

export async function verifyShareAccess(token: string, password?: string) {
  const { data } = await serviceSupabase.from('page_permissions')
    .select('page_id, token_expires_at, token_password_hash, role')
    .eq('share_token', token).maybeSingle()
  if (!data) throw new AppError('SHARE_NOT_FOUND','not found', 404)
  if (data.token_expires_at && new Date(data.token_expires_at) < new Date())
    throw new AppError('SHARE_EXPIRED','expired', 410)
  if (data.token_password_hash) {
    if (!password) throw new AppError('PASSWORD_REQUIRED','password required', 401)
    const ok = await bcrypt.compare(password, data.token_password_hash)
    if (!ok) throw new AppError('WRONG_PASSWORD','wrong password', 401)
  }
  return { pageId: data.page_id, role: data.role }
}
```
