```tsx
// Master scheduler: runs every 15 min, dispatches per-user jobs at their local 00:00
```

### Rollover with SKIP LOCKED

```sql
-- /src/workers/rollover.sql
BEGIN;
SET TRANSACTION ISOLATION LEVEL SERIALIZABLE;

WITH locked AS (
  SELECT id FROM tasks_v2
  WHERE user_id = $1
    AND due_date < CURRENT_DATE
    AND status != 'done'
    AND archived_at IS NULL
  FOR UPDATE SKIP LOCKED
)
UPDATE tasks_v2
SET due_date = CURRENT_DATE,
    rolled_over_count = rolled_over_count + 1,
    last_rolled_at = now()
WHERE id IN (SELECT id FROM locked);

UPDATE user_settings SET last_rollover_at = now() WHERE user_id = $1;
COMMIT;
```

### 2) Auto-rollover cron (00:00 بتوقيت المستخدم)

```tsx
// src/workers/tasks-rollover.ts
import { db } from '@/lib/db'
import { sendPush } from '@/lib/push'

export async function rolloverForUser(userId: string, todayDate: string, yesterdayDate: string) {
  // 1) خلص اليوم الفايت: ادر incomplete = scheduled but not done
  const incomplete = await db.tasks_v2.select()
    .where({ user_id: userId, scheduled_for: yesterdayDate, status: 'todo' })
  if (incomplete.length === 0) return

  // 2) لو rollover_locked = true → archive
  // غير كده → ابعت notification وانتظر قرار. لو مفيش رد بحلول 6am → archive default.
  await db.task_history.insertMany(
    incomplete.map(t => ({ task_id: t.id, user_id: userId, event: 'rollover_pending' })))
  await sendPush(userId, {
    title: 'مهام امبارح',
    body:  `عندك ${incomplete.length} مهام مكملتش. تنقلها لليوم؟`,
    action: 'review_rollover',
  })
}

export async function defaultArchiveAt6am(userId: string, yesterdayDate: string) {
  await db.tasks_v2.update({ status: 'archived', archived_at: 'now()' })
    .where({ user_id: userId, scheduled_for: yesterdayDate, status: 'todo' })
}

export async function rolloverToToday(userId: string, taskIds: string[], today: string) {
  await db.tasks_v2.update(t => ({
    scheduled_for: today,
    rollover_count: t.rollover_count + 1,
    updated_at: 'now()',
  })).where('id', 'in', taskIds).where({ user_id: userId })
  await db.task_history.insertMany(taskIds.map(id => ({
    task_id: id, user_id: userId, event: 'rolled_over', payload: { to: today },
  })))
}
```

### 7) Energy-aware scheduler

```tsx
export function computeEnergySlots(peakHours: number[], tasks: TaskRow[]) {
  const sorted = tasks.sort((a,b) => (b.priority - a.priority) || (b.importance - a.importance))
  const slots: Array<{ task_id:string; when:{ date:string; time:string } }> = []
  let cursor = new Date()
  for (const t of sorted) {
    const hour = pickHour(t.energy_required, peakHours, cursor)
    cursor = setHour(cursor, hour)
    slots.push({ task_id: t.id, when: { date: cursor.toISOString().slice(0,10), time: `${String(hour).padStart(2,'0')}:00` }})
    cursor = addMinutes(cursor, t.estimated_min ?? 30)
  }
  return slots
}
function pickHour(energy: string, peaks: number[], from: Date) {
  if (energy === 'high') return peaks[0] ?? 10
  if (energy === 'low')  return 14
  return peaks[1] ?? 11
}
```

```tsx
  { rx: /(\d{1,2})(:(\d{2}))?\s*(صباحًا|ص|مساءً|م)/g, set: 'time' },
]
const PRIORITIES = { '!!!':5, '!!':4, '!':3 } as const
const TAG_RX = /#(\S+)/g, CTX_RX = /@(\S+)/g

export function parseQuickAdd(input: string) {
  let title = input
  const tags  = Array.from(input.matchAll(TAG_RX), m => m[1]); title = title.replace(TAG_RX, '')
  const ctxs  = Array.from(input.matchAll(CTX_RX), m => m[1]); title = title.replace(CTX_RX, '')
  let priority = 3
  for (const [k,v] of Object.entries(PRIORITIES)) if (input.includes(k)) { priority = v; title = title.replace(k, '') }
  const parsed = chrono.parse(input, new Date(), { forwardDate: true })[0]
  return {
    title: title.trim(),
    scheduled_for: parsed?.start.date().toISOString().slice(0,10) ?? null,
    due_time: parsed?.start.isCertain('hour') ? `${parsed.start.get('hour')}:${String(parsed.start.get('minute')||0).padStart(2,'0')}` : null,
    tags, context_tags: ctxs, priority,
  }
}
```

### 9) Recurring engine

```tsx
import { RRule } from 'rrule'
export async function expandRecurring(userId: string, range: { from: Date; to: Date }) {
  const templates = await db.tasks_v2.select().where({ user_id: userId }).whereNotNull('recurrence_rule')
  for (const t of templates) {
    const rule = RRule.fromString(t.recurrence_rule!)
    for (const occ of rule.between(range.from, range.to)) {
      const exists = await db.tasks_v2.where({ user_id: userId, parent_task_id: t.id, scheduled_for: occ.toISOString().slice(0,10) }).one()
      if (!exists) await db.tasks_v2.insert({ ...t, id: undefined, parent_task_id: t.id, scheduled_for: occ.toISOString().slice(0,10), recurrence_rule: null, status: 'todo' })
    }
  }
}
```
