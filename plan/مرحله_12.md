export async function recomputeGoalProgress(goalId: string, ctx: RequestContext) {
  const [milestones, keyResults, linkedTasks] = await Promise.all([
    goalRepo.listMilestones(goalId, ctx),
    goalRepo.listKeyResults(goalId, ctx),
    taskRepo.listByGoal(goalId, ctx), // db_rows من Tasks system DB
  ])
  const milestonePct = ratio(milestones.filter(m => m.isDone).length, milestones.length)
  const krPct = average(keyResults.map(kr => Math.min(100, (kr.current / kr.target) * 100)))
  const taskPct = ratio(linkedTasks.filter(t => t.status === 'done').length, linkedTasks.length)
  return Math.round(milestonePct * 0.4 + krPct * 0.35 + taskPct * 0.25)
}

### Migration 1201 — compute_goal_progress RPC

```sql
-- supabase/migrations/1201_goal_progress_rpc.sql
BEGIN;
CREATE OR REPLACE FUNCTION public.compute_goal_progress(p_goal UUID)
RETURNS INT LANGUAGE plpgsql SECURITY DEFINER SET search_path = public, pg_temp AS $$
DECLARE
  v_mil_total INT; v_mil_done INT;
  v_task_total INT; v_task_done INT;
  v_kr_progress NUMERIC;
  v_combined INT;
BEGIN
  SELECT COUNT(*), COUNT(*) FILTER (WHERE is_done)
  INTO v_mil_total, v_mil_done
  FROM public.goal_milestones WHERE goal_id = p_goal;

  SELECT COUNT(*), COUNT(*) FILTER (WHERE status = 'done')
  INTO v_task_total, v_task_done
  FROM public.tasks WHERE goal_id = p_goal AND NOT is_deleted;

  SELECT COALESCE(
    AVG(CASE WHEN target > 0 THEN LEAST(current * 100.0 / target, 100) ELSE 0 END),
    0
  ) INTO v_kr_progress
  FROM public.goal_key_results WHERE goal_id = p_goal;

  -- Weighted blend: 40% milestones, 30% tasks, 30% key results
  v_combined := ROUND(
    (CASE WHEN v_mil_total > 0 THEN v_mil_done * 100.0 / v_mil_total ELSE 0 END) * 0.4 +
    (CASE WHEN v_task_total > 0 THEN v_task_done * 100.0 / v_task_total ELSE 0 END) * 0.3 +
    (v_kr_progress) * 0.3
  )::INT;

  RETURN GREATEST(0, LEAST(100, v_combined));
END $$;
COMMIT;
```

import { z } from 'zod'

  keyResults:    z.array(z.object({
    title: z.string().min(1).max(200),
    target: z.number().int().positive(),

})

    title: data.title,
    description: data.description ?? null,
    category: data.category ?? null,
    target_date: data.targetDate ?? null,
    why_it_matters: data.why_it_matters ?? null,
    cover_color: data.coverColor,
    priority: data.priority,
  }).select().single()

  if (error) throw new AppError('GOAL_CREATE_FAILED', error.message, 500)

  if (data.milestones?.length) {

      data.milestones.map((m, i) => ({
        goal_id: goal.id, user_id: user.id,
        title: m.title, position: i,

    )

        goal_id: goal.id, user_id: user.id,
        title: kr.title, target: kr.target, unit: kr.unit,
      }))
    )
  }

  await audit({ actorId: user.id, action: 'goal.created',
    targetId: goal.id, targetType: 'goal' })
  revalidateTag('goals')
  revalidateTag('life-score')  // W16 integration
  return goal
}

CREATE OR REPLACE FUNCTION public.has_dependency_cycle(
  p_task UUID, p_depends_on UUID
) RETURNS BOOLEAN LANGUAGE plpgsql SECURITY DEFINER SET search_path = public, pg_temp AS $$
DECLARE
  v_visited UUID[];
  v_queue   UUID[];
  v_current UUID;
BEGIN
  v_visited := ARRAY[p_task];
  v_queue   := ARRAY[p_depends_on];

  WHILE array_length(v_queue, 1) > 0 LOOP
    v_current := v_queue[1];
    v_queue   := v_queue[2:];
    IF v_current = p_task THEN RETURN TRUE; END IF;
    IF v_current = ANY(v_visited) THEN CONTINUE; END IF;
    v_visited := v_visited || v_current;
    SELECT array_agg(depends_on_id) INTO v_queue
    FROM public.task_dependencies WHERE task_id = v_current;
  END LOOP;
  RETURN FALSE;
END $$;

-- Enforce no cycles at insert
CREATE OR REPLACE FUNCTION public.enforce_no_cycle()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
BEGIN
  IF public.has_dependency_cycle(NEW.task_id, NEW.depends_on_id) THEN
    RAISE EXCEPTION 'cycle_detected: dependency would create a cycle';
  END IF;
  RETURN NEW;
END $$;

// src/features/projects/lib/priority-matrix.ts
import 'server-only'
import { unstable_cache } from 'next/cache'
import { serviceSupabase } from '@/lib/supabase/service'

interface MatrixTask {
  id: string; title: string; quadrant: 'Q1'|'Q2'|'Q3'|'Q4'
  dueDate?: string; priority: 'urgent'|'high'|'medium'|'low'; goalId?: string
}

export const getPriorityMatrix = unstable_cache(
  async (userId: string): Promise<Record<'Q1'|'Q2'|'Q3'|'Q4', MatrixTask[]>> => {
    const { data: tasks } = await serviceSupabase
      .from('tasks')
      .select('id, title, due_date, priority, goal_id, status')
      .eq('user_id', userId)
      .neq('status', 'done')
      .eq('is_deleted', false)
      .limit(200)

    const today = new Date()
    const m = { Q1: [], Q2: [], Q3: [], Q4: [] } as Record<'Q1'|'Q2'|'Q3'|'Q4', MatrixTask[]>
    for (const t of tasks ?? []) {
      const urgent = isUrgent(t.due_date, today)
      const important = isImportant(t.priority, t.goal_id)
      const q = urgent && important ? 'Q1'
              : !urgent && important ? 'Q2'
              : urgent && !important ? 'Q3' : 'Q4'
      m[q].push({ id: t.id, title: t.title, dueDate: t.due_date, priority: t.priority, goalId: t.goal_id, quadrant: q })
    }
    return m
  },
  ['priority-matrix'],
  { revalidate: 120, tags: ['tasks', 'priority-matrix'] }
)

function isUrgent(dueDate: string | null, today: Date): boolean {
  if (!dueDate) return false
  const days = Math.ceil((new Date(dueDate).getTime() - today.getTime()) / 86_400_000)
  return days <= 3
}
function isImportant(priority: string, goalId: string | null): boolean {
  return priority === 'urgent' || priority === 'high' || !!goalId
}

### Life Score Integration (W16)

The `compute_life_score` function in W16 reads from goals via:

```sql
-- Inside W16 compute_life_score
SELECT AVG(progress_pct) FILTER (WHERE status = 'active')
FROM public.goals
WHERE user_id = p_user AND NOT is_deleted;
```

Goal progress contributes **20%** to the Life Score (alongside habits 30%, tasks 25%, focus 15%, mood 10%).
