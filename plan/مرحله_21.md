## 🧠 Master Architect Upgrade — Vault Cryptography Gate

- **Terminology:** هذه Zero-Knowledge Encryption/E2EE vault، وليست ZKP protocol. أي استخدام لكلمة ZKP يحتاج proof design منفصل.
- **Client-only plaintext:** server لا يستقبل password plaintext أو decrypted item. clipboard/autofill/preview كلها client-side ومؤقتة مع auto-clear.
- **Key material policy:** vault key memory-only، no localStorage/IndexedDB، wipe best-effort، lock on logout/session expiry، وno persistence في service worker.
- **Crypto review:** Argon2id/AES-GCM/envelope sharing/recovery key flow يحتاج external crypto review قبل launch.
- **AI/search/export wall:** AI, analytics, search, Sentry, public share، وgeneric export لا يروا vault plaintext. encrypted export فقط عبر user-controlled key.

### 2) Key derivation (client-side)

```tsx
// src/lib/vault/keys.ts — runs in browser only
import { argon2id } from '@noble/hashes/argon2'
import { hmac }     from '@noble/hashes/hmac'
import { sha256 }   from '@noble/hashes/sha256'
import { gcm }      from '@noble/ciphers/aes'
import { randomBytes } from '@noble/ciphers/webcrypto'

export type KdfParams = { algo:'argon2id'; m:number; t:number; p:number }

export const DEFAULT_KDF: KdfParams = { algo:'argon2id', m:131072, t:3, p:4 }

export async function deriveVaultKey(
  masterPassword: string, salt: Uint8Array, params = DEFAULT_KDF,
): Promise<Uint8Array> {
  return argon2id(masterPassword, salt, {
    m: params.m, t: params.t, p: params.p, dkLen: 32,
  })
}

export function makeVerifier(vaultKey: Uint8Array): Uint8Array {
  return hmac(sha256, vaultKey, new TextEncoder().encode('zenith-vault-verifier-v1'))
}

export function encryptItem(plaintext: Uint8Array, vaultKey: Uint8Array) {
  const iv = randomBytes(12)
  const c  = gcm(vaultKey, iv)
  const out = c.encrypt(plaintext)
  return {
    iv,
    ciphertext: out.slice(0, out.length - 16),
    tag:        out.slice(out.length - 16),
  }
}

export function decryptItem(ct: Uint8Array, iv: Uint8Array, tag: Uint8Array, vaultKey: Uint8Array) {
  const c = gcm(vaultKey, iv)
  const concat = new Uint8Array(ct.length + tag.length)
  concat.set(ct); concat.set(tag, ct.length)
  return c.decrypt(concat)
}
```

### 3) Vault session store (in-memory only)

```tsx
// src/lib/vault/session.ts
let _vaultKey: Uint8Array | null = null
let _lockTimer: number | null = null
const LOCK_AFTER_MS = 10 * 60_000   // 10 دقايق default

export function unlockVault(key: Uint8Array) {
  _vaultKey = key
  resetAutoLock()
  window.addEventListener('blur', onTabBlur)
  document.addEventListener('visibilitychange', onVisibility)
}

export function lockVault() {
  if (_vaultKey) crypto.subtle.importKey('raw', new Uint8Array(32), 'AES-GCM', false, []) // best-effort wipe
  _vaultKey = null
  if (_lockTimer) clearTimeout(_lockTimer)
  document.removeEventListener('visibilitychange', onVisibility)
  window.removeEventListener('blur', onTabBlur)
  window.dispatchEvent(new CustomEvent('vault:locked'))
}

export function getVaultKey(): Uint8Array {
  if (!_vaultKey) throw new Error('VAULT_LOCKED')
  resetAutoLock()
  return _vaultKey
}

function resetAutoLock() {
  if (_lockTimer) clearTimeout(_lockTimer)
  _lockTimer = window.setTimeout(lockVault, LOCK_AFTER_MS)
}
function onTabBlur() { /* optional: lock on blur if setting enabled */ }
function onVisibility() { if (document.hidden) { /* same */ } }
```

### 4) Password generator + strength

```tsx
// src/lib/vault/generator.ts
import { zxcvbn } from '@zxcvbn-ts/core'

const LOWER = 'abcdefghijkmnpqrstuvwxyz'
const UPPER = 'ABCDEFGHJKLMNPQRSTUVWXYZ'
const DIGIT = '23456789'
const SYM   = '!@#$%^&*-_=+?'

export function generate({ length=20, lower=true, upper=true, digit=true, sym=true, readable=false }) {
  const pool = (lower?LOWER:'')+(upper?UPPER:'')+(digit?DIGIT:'')+(sym&&!readable?SYM:'')
  if (!pool) throw new Error('NEED_CHARSET')
  const arr = new Uint32Array(length)
  crypto.getRandomValues(arr)
  return Array.from(arr, x => pool[x % pool.length]).join('')
}

export function strength(pw: string) {
  const r = zxcvbn(pw)
  return { score: r.score, crackTime: r.crackTimesDisplay.offlineSlowHashing1e4PerSecond }
}
```

### 5) TOTP generator

```tsx
// src/lib/vault/totp.ts (RFC 6238)
import { hmac } from '@noble/hashes/hmac'
import { sha1 } from '@noble/hashes/sha1'
import { base32 } from '@scure/base'

export function totp(secretBase32: string, t = Date.now(), digits = 6, period = 30) {
  const secret = base32.decode(secretBase32.toUpperCase().replace(/\s+/g,''))
  const counter = Math.floor(t / 1000 / period)
  const buf = new ArrayBuffer(8); new DataView(buf).setBigUint64(0, BigInt(counter))
  const mac = hmac(sha1, secret, new Uint8Array(buf))
  const off = mac[mac.length - 1] & 0xf
  const code = ((mac[off]&0x7f)<<24 | mac[off+1]<<16 | mac[off+2]<<8 | mac[off+3]) % 10**digits
  return code.toString().padStart(digits,'0')
}
export function secondsLeft(period=30) { return period - (Math.floor(Date.now()/1000) % period) }
```

### 6) Breach monitoring (k-anonymity)

```tsx
// src/lib/vault/breach.ts
import { sha1 } from '@noble/hashes/sha1'

export async function checkBreach(password: string): Promise<number> {
  const hash = Array.from(sha1(new TextEncoder().encode(password)))
    .map(b=>b.toString(16).padStart(2,'0')).join('').toUpperCase()
  const prefix = hash.slice(0,5), suffix = hash.slice(5)
  // أول 5 بس بنبعتها — k-anonymity
  const res = await fetch(`https://api.pwnedpasswords.com/range/${prefix}`)
  const txt = await res.text()
  for (const line of txt.split('\n')) {
    const [s,c] = line.trim().split(':')
    if (s === suffix) return parseInt(c,10)
  }
  return 0
}
```

### 7) Server actions (DB only — never sees plaintext)

```tsx
// src/app/(app)/vault/actions.ts
'use server'
import { db } from '@/lib/db'
import { z } from 'zod'
import { currentUser } from '@/lib/auth'
import { audit } from '@/lib/vault/audit'

const CreateItem = z.object({
  type: z.enum(['login','card','identity','note','crypto','ssh','license']),
  folder_id: z.string().uuid().nullable(),
  data_enc: z.instanceof(Uint8Array),
  data_iv:  z.instanceof(Uint8Array),
  data_tag: z.instanceof(Uint8Array),
  url_domain: z.string().nullable(),
  has_totp: z.boolean(),
})

export async function createVaultItem(input: z.infer<typeof CreateItem>) {
  const u = await currentUser()
  const v = CreateItem.parse(input)
  const row = await db.vault_items.insert({ ...v, user_id: u.id }).returning('id').one()
  await audit({ user_id: u.id, item_id: row.id, event: 'create' })
  return row.id
}

export async function listVaultItems({ folderId, query, type, favorite }: { folderId?:string; query?:string; type?:string; favorite?:boolean }) {
  const u = await currentUser()
  let q = db.vault_items.select().where({ user_id: u.id, archived: false })
  if (folderId) q = q.where({ folder_id: folderId })
  if (type)     q = q.where({ type })
  if (favorite) q = q.where({ favorite: true })
  if (query)    q = q.where('url_domain', 'ilike', `%${query}%`)
  return q.orderBy('last_used_at desc nulls last')
}
```
