### Sensitive Action Wrapper

```tsx
export async function runAdminAction<T>(args: {
  admin: AdminContext
  action: string
  targetType: string
  targetId: string
  reason: string
  stepUpToken?: string
  run: () => Promise<T>
}) {
  if (args.reason.trim().length < 20) throw new AppError('ADMIN_REASON_REQUIRED', 'Reason too short', 422)
  if (SENSITIVE_ADMIN_ACTIONS.has(args.action)) await verifyStepUp(args.admin.userId, args.stepUpToken)
  const before = await adminSnapshot(args.targetType, args.targetId)
  const result = await args.run()
  const after = await adminSnapshot(args.targetType, args.targetId)
  await adminAudit(args.admin, args.action, args.targetType, args.targetId, before, after, args.reason)
  return result
}
```

## 🧠 Master Architect Upgrade — Admin Blast-Radius Gate

- **Admin zero-trust:** كل admin route يحتاج MFA حديث، IP/device policy، reason، rate limit، step-up confirmation للعمليات الحساسة، وimmutable audit.
- **No vault bypass:** support/admin لا يقرأ Vault/E2EE plaintext. أي investigation قانوني يرى metadata فقط إلا export encrypted بآلية user-controlled/legal-hold موثقة.
- **Impersonation transparency:** max 1h، explicit reason، visible banner، user notification، session recording/audit، وممنوع أثناء vault/secret screens.
- **Governance readiness:** W27.5 capabilities (SAML/SCIM/domain claim/retention/legal hold/DLP/eDiscovery/data residency) تدخل كـ launch blockers لو target enterprise.
- **SRE linkage:** health dashboard يربط SLO/error budgets/on-call/runbooks، وليس charts فقط. كل alert له owner + playbook + severity.

### 2) Admin middleware

```tsx
// src/lib/admin/middleware.ts
import 'server-only'
import { headers } from 'next/headers'
import { requireUser } from '@/lib/auth/require-user'
import { serviceSupabase } from '@/lib/supabase/service'
import { AppError } from '@/lib/result'
import { audit } from '@/lib/audit'

export async function requireAdmin(role?: 'super_admin' | 'support' | 'moderator' | 'analyst') {
  const user = await requireUser()
  const h    = await headers()
  const ip   = h.get('x-forwarded-for')?.split(',')[0]?.trim() ?? ''
  const ua   = h.get('user-agent') ?? ''

  const { data: admin } = await serviceSupabase.from('admin_users')
    .select('*').eq('user_id', user.id).single()

  if (!admin || !admin.is_active)
    throw new AppError('NOT_ADMIN','admin required', 403)
  if (admin.expires_at && new Date(admin.expires_at) < new Date())
    throw new AppError('ADMIN_EXPIRED','admin expired', 403)
  if (admin.ip_allowlist?.length && !admin.ip_allowlist.includes(ip))
    throw new AppError('IP_NOT_ALLOWED','IP not in allowlist', 403)
  if (role && admin.role !== role && admin.role !== 'super_admin')
    throw new AppError('INSUFFICIENT_ROLE', `requires ${role}`, 403)
  if (!admin.totp_enabled)
    throw new AppError('TOTP_REQUIRED','enable TOTP first', 403)

  return { user, admin, ip, ua }
}

export async function adminAudit(
  adminId: string,
  action: string,
  targetType: string | null,
  targetId: string | null,
  before: unknown, after: unknown,
  ip: string, ua: string, reason: string,
) {
  await serviceSupabase.from('admin_audit_logs').insert({
    admin_id: adminId, action, target_type: targetType, target_id: targetId,
    before_data: before, after_data: after,
    ip_address: ip, user_agent: ua, reason,
  })
}
```

```tsx
import { requireAdmin, adminAudit } from '@/lib/admin/middleware'
import { serviceSupabase } from '@/lib/supabase/service'

export async function banUserAction(input: { userId: string; reason: string }) {
  const { admin, ip, ua } = await requireAdmin('support')
  if (input.reason.length < 10)
    throw new AppError('REASON_TOO_SHORT','reason must be > 10 chars', 422)

  const { data: before } = await serviceSupabase.from('profiles')
    .select('id, is_banned').eq('id', input.userId).single()

    before, { is_banned: true }, ip, ua, input.reason)
  return { ok: true }
}

export async function deleteUserAction(input: {
  userId: string; reason: string; totpCode: string;
}) {
  const { admin, ip, ua } = await requireAdmin('super_admin')
  // TOTP verify
  await verifyAdminTotp(admin.user_id, input.totpCode)

    .select('*').eq('id', input.userId).single()
```

### 5) SLA tracker

```tsx
// src/lib/support/sla.ts
import 'server-only'
import { serviceSupabase } from '@/lib/supabase/service'

const SLA = {
  urgent: { firstResponseMinutes: 60,      resolveHours: 4 },
  high:   { firstResponseMinutes: 4 * 60,  resolveHours: 24 },
  medium: { firstResponseMinutes: 24 * 60, resolveHours: 72 },
  low:    { firstResponseMinutes: 48 * 60, resolveHours: 168 },
} as const

export async function computeSlaStatus(ticketId: string) {
  const { data: t } = await serviceSupabase.from('support_tickets')
    .select('priority, created_at, first_response_at, resolved_at, status')
    .eq('id', ticketId).single()
  if (!t) return null
  const sla = SLA[t.priority]
  const createdMs = +new Date(t.created_at)
  const now = Date.now()

  const firstResponseDue = createdMs + sla.firstResponseMinutes * 60_000
  const resolveDue       = createdMs + sla.resolveHours * 3_600_000

  return {
    firstResponseBreached:
      !t.first_response_at && now > firstResponseDue,
    resolveBreached:
      !t.resolved_at && now > resolveDue,
    firstResponseRemainingMs: t.first_response_at
      ? 0 : Math.max(0, firstResponseDue - now),
    resolveRemainingMs: t.resolved_at
      ? 0 : Math.max(0, resolveDue - now),
  }
}
```

```sql
CREATE TABLE public.feature_flags (
  key                TEXT PRIMARY KEY,
  description        TEXT,
  is_enabled         BOOLEAN NOT NULL DEFAULT FALSE,
  rollout_percentage INT NOT NULL DEFAULT 0 CHECK (rollout_percentage BETWEEN 0 AND 100),
  target_users       UUID[],

ALTER TABLE public.feature_flags FORCE ROW LEVEL SECURITY;

CREATE OR REPLACE FUNCTION public.is_feature_enabled(p_key TEXT, p_user UUID)
RETURNS BOOLEAN LANGUAGE plpgsql STABLE SECURITY DEFINER
SET search_path = public, pg_temp AS $$
DECLARE v_flag public.feature_flags%ROWTYPE;
BEGIN
  SELECT * INTO v_flag FROM feature_flags WHERE key = p_key;
  IF NOT FOUND OR NOT v_flag.is_enabled THEN RETURN FALSE; END IF;
  IF p_user = ANY(v_flag.target_users) THEN RETURN TRUE; END IF;
  -- deterministic rollout بـ hash(user_id + key) % 100
  RETURN (abs(hashtextextended(p_user::text || p_key, 0)) % 100) < v_flag.rollout_percentage;
END $$;
COMMIT;
```

### 8) Impersonation system

```tsx
// src/app/admin/impersonate/route.ts
import 'server-only'
import { z } from 'zod'
import { requireAdmin, adminAudit } from '@/lib/admin/middleware'
import { serviceSupabase } from '@/lib/supabase/service'
import { randomBytes } from 'crypto'
import { cookies } from 'next/headers'

const schema = z.object({
  userId:  z.string().uuid(),
  reason:  z.string().min(20),
  totpCode: z.string().regex(/^\d{6}$/),
})

export async function POST(req: Request) {
  const { admin, ip, ua } = await requireAdmin('support')
  const input = schema.parse(await req.json())

  await verifyAdminTotp(admin.user_id, input.totpCode)

  const token = randomBytes(48).toString('base64url')
  await serviceSupabase.from('impersonation_sessions').insert({
    admin_id:    admin.user_id,
    target_user_id: input.userId,
    token_hash:  hash(token),
    expires_at:  new Date(Date.now() + 3600_000).toISOString(),  // 1 hour max
    reason:      input.reason,
  })

  await adminAudit(admin.user_id, 'user.impersonation_started',
    'user', input.userId, null, { expiresIn: '1h' }, ip, ua, input.reason)

  // notify the target user via email (transparency)
  await notifyUserOfImpersonation(input.userId, admin.user_id, input.reason)

  ;(await cookies()).set('imp_token', token, {
    httpOnly: true, secure: true, sameSite: 'strict', maxAge: 3600,
  })
  return Response.json({ ok: true, expiresIn: 3600 })
}

function hash(s: string) {
  return require('crypto').createHash('sha256').update(s).digest('hex')
}
```
