## Phase 53 — Template Engine

### الفلسفة

أي حاجة في Zenith ممكن تتحول لـ template — Page, Database Row, Task, Habit, Goal, Project, Event, Note. القالب بيخزّن: (1) الـ schema/structure، (2) blocks/content مع `variables`، (3) sub-entities (sub-tasks, milestones). عند الإستخدام: variables بتتعبّى من سياق المستخدم (today, user_name, current_goal...).

### Architecture Rules

- **Soft delete** على templates (لو حد بيستخدمها).
- **Workspace-scoped** + **owner-private** بـ `visibility` enum.
- **Variables بصيغة `var_name`** فقط — لا JS evaluation.
- **Max 200 templates per user**.
- **Vault content ممنوع** يدخل templates.
- **Built-in templates** (system-owned) لا تتعدل ولا تتحذف.

```sql
CREATE TABLE public.templates (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id       UUID REFERENCES public.profiles(id) ON DELETE CASCADE,  -- NULL = system

                CHECK (visibility IN ('private','workspace','public_marketplace')),

  variables     JSONB NOT NULL DEFAULT '[]'::jsonb,

  is_deleted    BOOLEAN NOT NULL DEFAULT FALSE,
  created_at    TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at    TIMESTAMPTZ NOT NULL DEFAULT now(),
  CONSTRAINT system_or_user CHECK (

    (is_system = FALSE AND user_id IS NOT NULL)
  )

ALTER TABLE public.templates ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.templates FORCE ROW LEVEL SECURITY;

CREATE POLICY templates_read ON public.templates FOR SELECT
  USING (
    is_system = TRUE OR
    user_id = (SELECT auth.uid()) OR
    (visibility = 'workspace' AND workspace_id IN (
      SELECT workspace_id FROM public.workspace_members WHERE user_id = (SELECT auth.uid())
    )) OR
    visibility = 'public_marketplace'
  );

CREATE POLICY templates_write ON public.templates FOR ALL
  USING (user_id = (SELECT auth.uid()) AND is_system = FALSE)
  WITH CHECK (user_id = (SELECT auth.uid()) AND is_system = FALSE);

CREATE INDEX templates_user_kind ON public.templates(user_id, kind) WHERE NOT is_deleted;
CREATE INDEX templates_marketplace ON public.templates(visibility, use_count DESC)
  WHERE visibility = 'public_marketplace' AND NOT is_deleted;

CREATE TRIGGER trg_templates_updated BEFORE UPDATE ON public.templates
  FOR EACH ROW EXECUTE FUNCTION public.set_updated_at();

-- Per-user template count guard
CREATE OR REPLACE FUNCTION public.check_template_limit()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
DECLARE v_count INT;
BEGIN
  IF NEW.is_system THEN RETURN NEW; END IF;
  SELECT COUNT(*) INTO v_count FROM public.templates
  WHERE user_id = NEW.user_id AND NOT is_deleted;
  IF v_count >= 200 THEN
    RAISE EXCEPTION 'template_limit_reached: max 200 per user';
  END IF;
  RETURN NEW;
END $$;
CREATE TRIGGER trg_template_limit BEFORE INSERT ON public.templates
  FOR EACH ROW EXECUTE FUNCTION public.check_template_limit();

COMMIT;
```

### TypeScript: applyTemplate Action

```tsx
// src/features/templates/actions/apply-template.ts
'use server'
import { z } from 'zod'
import { requireUser } from '@/lib/auth/require-user'
import { serviceSupabase } from '@/lib/supabase/service'
import { AppError } from '@/lib/result'
import { renderTemplateVariables } from '../lib/render-variables'
import { audit } from '@/lib/audit'

const schema = z.object({
  templateId: z.string().uuid(),
  variables:  z.record(z.string(), z.string().max(500)).default({}),
  targetParentId: z.string().uuid().optional(),  // for page templates
})

export async function applyTemplateAction(input: unknown) {
  const { templateId, variables, targetParentId } = schema.parse(input)
  const user = await requireUser()

  const { data: template } = await serviceSupabase
    .from('templates')
    .select('*, template_blocks(*)')
    .eq('id', templateId)
    .single()
  if (!template) throw new AppError('TEMPLATE_NOT_FOUND', 'Template not found', 404)

  // Render variables in payload + blocks
  const renderedPayload = renderTemplateVariables(template.payload, variables, {
    today:     new Date().toISOString().slice(0, 10),
    user_name: user.name ?? user.email,
  })

  let resultId: string
  let resultType: string

  switch (template.kind) {
    case 'task':
      resultType = 'task'
      resultId = await createTaskFromTemplate(user.id, renderedPayload, template.template_blocks)
      break
    case 'habit':
      resultType = 'habit'
      resultId = await createHabitFromTemplate(user.id, renderedPayload)
      break
    case 'goal':
      resultType = 'goal'
      resultId = await createGoalFromTemplate(user.id, renderedPayload)
      break
    case 'project':
      resultType = 'project'
      resultId = await createProjectFromTemplate(user.id, renderedPayload)
      break
    case 'page':
      resultType = 'page'
      resultId = await createPageFromTemplate(user.id, renderedPayload, template.template_blocks, targetParentId)
      break
    case 'event':
      resultType = 'event'
      resultId = await createEventFromTemplate(user.id, renderedPayload)
      break
    case 'expense':
      resultType = 'expense'
      resultId = await createExpenseFromTemplate(user.id, renderedPayload)
      break
    case 'review':
      resultType = 'review'
      resultId = await createReviewFromTemplate(user.id, renderedPayload)
      break
    default:
      throw new AppError('UNSUPPORTED_KIND', `kind ${template.kind} not implemented`, 400)
  }

  // Log use
  await serviceSupabase.from('template_uses').insert({
    template_id: templateId,
    user_id: user.id,
    result_entity_type: resultType,
    result_entity_id: resultId,
    variables_used: variables,
  })

  await audit({ actorId: user.id, action: 'template.applied',
    targetId: templateId, targetType: 'template',
    metadata: { result: { type: resultType, id: resultId } } })

  return { resultType, resultId }
}
```

### TypeScript: Variable Renderer (Safe)

```tsx
// src/features/templates/lib/render-variables.ts
import 'server-only'

const VAR_RE = /\{\{\s*([a-zA-Z_][a-zA-Z0-9_]*)\s*\}\}/g

export function renderTemplateVariables(
  payload: unknown,
  userVars: Record<string, string>,
  systemVars: Record<string, string>
): unknown {
  if (typeof payload === 'string') {
    return payload.replace(VAR_RE, (_, name) => {
      if (userVars[name] !== undefined) return userVars[name]
      if (systemVars[name] !== undefined) return systemVars[name]
      return `{{${name}}}`  // leave unresolved as-is
    })
  }
  if (Array.isArray(payload)) {
    return payload.map(item => renderTemplateVariables(item, userVars, systemVars))
  }
  if (payload && typeof payload === 'object') {
    const out: Record<string, unknown> = {}
    for (const [k, v] of Object.entries(payload)) {
      out[k] = renderTemplateVariables(v, userVars, systemVars)
    }
    return out
  }
  return payload
}
```

```tsx
export function TemplatePicker({ kind, onApply }: {
```

### Migration 1506 — Trigger Dispatcher

```sql
-- supabase/migrations/1506_automation_dispatcher.sql
BEGIN;

CREATE OR REPLACE FUNCTION public.dispatch_automation_trigger(
  p_user UUID, p_trigger_kind TEXT, p_payload JSONB
) RETURNS INT
LANGUAGE plpgsql SECURITY DEFINER SET search_path = public, pg_temp AS $$
DECLARE
  v_count INT := 0;
  v_auto RECORD;
BEGIN
  FOR v_auto IN
    SELECT id FROM public.automations
    WHERE user_id = p_user
      AND trigger_kind = p_trigger_kind
      AND is_active = TRUE
  LOOP
    INSERT INTO public.automation_runs(automation_id, user_id, trigger_payload, status)
    VALUES (v_auto.id, p_user, p_payload, 'queued');
    v_count := v_count + 1;
  END LOOP;
  -- notify worker (LISTEN/NOTIFY)
  PERFORM pg_notify('automation_queue', p_user::text);
  RETURN v_count;
END $$;

-- Example: hook task.completed
CREATE OR REPLACE FUNCTION public.hook_task_completed()
RETURNS TRIGGER LANGUAGE plpgsql SECURITY DEFINER SET search_path = public, pg_temp AS $$
BEGIN
  IF NEW.status = 'done' AND (OLD.status IS NULL OR OLD.status <> 'done') THEN
    PERFORM public.dispatch_automation_trigger(
      NEW.user_id,
      'task.completed',
      jsonb_build_object(
        'task_id', NEW.id,
        'title', NEW.title,
        'priority', NEW.priority,
        'completed_at', NEW.updated_at,
        'goal_id', NEW.goal_id,
        'project_id', NEW.project_id
      )
    );
  END IF;
  RETURN NEW;
END $$;

CREATE TRIGGER trg_task_completed_hook
  AFTER UPDATE ON public.tasks
  FOR EACH ROW EXECUTE FUNCTION public.hook_task_completed();

COMMIT;
```

### TypeScript: Automation Worker

```tsx
// src/features/automations/worker.ts
import 'server-only'
import { serviceSupabase } from '@/lib/supabase/service'
import { runAction } from './actions/run-action'
import { evaluateFilter } from './lib/filter'

export async function processAutomationQueue(): Promise<{ processed: number }> {
  const { data: queue } = await serviceSupabase
    .from('automation_runs')
    .select('*, automations(*)')
    .eq('status', 'queued')
    .order('created_at', { ascending: true })
    .limit(20)

  let processed = 0
  for (const run of queue ?? []) {
    const auto = (run as any).automations
    if (!auto?.is_active) {
      await markRun(run.id, 'skipped', 'automation inactive')
      continue
    }
    // Apply filter
    if (Object.keys(auto.filter_jsonb).length > 0 &&
        !evaluateFilter(auto.filter_jsonb, run.trigger_payload)) {
      await markRun(run.id, 'skipped', 'filter did not match')
      continue
    }

    const started = Date.now()
    await markRun(run.id, 'running')
    try {
      const results: unknown[] = []
      for (const action of auto.actions) {
        const result = await runAction(run.user_id, action, run.trigger_payload)
        results.push(result)
      }
      await markRun(run.id, 'completed', null, results, Date.now() - started)
      await serviceSupabase.from('automations')
        .update({ run_count: auto.run_count + 1, last_run_at: new Date().toISOString() })
        .eq('id', auto.id)
      processed++
    } catch (err: any) {
      await markRun(run.id, 'failed', err.message, null, Date.now() - started)
      await serviceSupabase.from('automations')
        .update({ failure_count: auto.failure_count + 1, last_error: err.message })
        .eq('id', auto.id)
      // Auto-disable after 10 consecutive failures
      if (auto.failure_count + 1 >= 10) {
        await serviceSupabase.from('automations')
          .update({ is_active: false })
          .eq('id', auto.id)
      }
    }
  }
  return { processed }
}
```

### TypeScript: Action Runner (Per-Action Type)

```tsx
// src/features/automations/actions/run-action.ts
import 'server-only'
import { serviceSupabase } from '@/lib/supabase/service'
import { renderTemplateVariables } from '@/features/templates/lib/render-variables'

export async function runAction(
  userId: string,
  action: AutomationAction,
  triggerPayload: Record<string, unknown>
): Promise<unknown> {
  const params = renderTemplateVariables(action.params, {}, triggerPayload as any)

  switch (action.kind) {
    case 'create_task': {
      const { data } = await serviceSupabase.from('tasks').insert({
        user_id: userId,
        title: (params as any).title,
        priority: (params as any).priority ?? 'medium',
        due_date: (params as any).due_date,
        project_id: (params as any).project_id,
      }).select('id').single()
      return { task_id: data?.id }
    }
    case 'send_notification': {
      return await sendNotification(userId, params as any)
    }
    case 'apply_template': {
      // delegates to applyTemplateAction with elevated context
      return await applyTemplateInternal(userId, (params as any).template_id, (params as any).variables)
    }
    case 'award_xp': {
      const { data } = await serviceSupabase.rpc('award_xp', {
        p_user: userId,
        p_source_type: 'automation',
        p_source_id: null,
        p_xp: (params as any).amount ?? 10,
        p_idempotency: `auto:${(params as any).idempotency ?? crypto.randomUUID()}`,
      })
      return data
    }
    case 'webhook': {
      const url = (params as any).url
      if (!url?.startsWith('https://')) throw new Error('webhook url must be HTTPS')
      const res = await fetch(url, {
        method: 'POST',
        headers: { 'content-type': 'application/json' },
        body: JSON.stringify(triggerPayload),
        signal: AbortSignal.timeout(10_000),
      })
      return { status: res.status }
    }
    case 'ai_summarize':
      return await runAISummary(userId, params as any)
    case 'tag_entity':
      return await tagEntity(userId, params as any)
    default:
      throw new Error(`unknown action: ${action.kind}`)
  }
}
```

```sql
-- supabase/migrations/1508_reminder_dispatch.sql
BEGIN;

CREATE OR REPLACE FUNCTION public.dispatch_due_reminders()
RETURNS INT LANGUAGE plpgsql SECURITY DEFINER SET search_path = public, pg_temp AS $$
DECLARE
  v_count INT := 0;
  v_rem RECORD;
BEGIN
  FOR v_rem IN
    SELECT * FROM public.reminders
    WHERE (
      (status = 'pending' AND remind_at <= now()) OR
      (status = 'snoozed' AND snoozed_to <= now())
    )
    ORDER BY remind_at
    LIMIT 100
  LOOP
    -- Mark as sent (will be picked up by notification worker)
    UPDATE public.reminders
    SET status = 'sent', sent_at = now()
    WHERE id = v_rem.id;
    -- Trigger automation
    PERFORM public.dispatch_automation_trigger(
      v_rem.user_id,
      'reminder.fired',
      jsonb_build_object(
        'reminder_id', v_rem.id,
        'title', v_rem.title,
        'entity_type', v_rem.entity_type,
        'entity_id', v_rem.entity_id,
        'channels', v_rem.channels
      )
    );
    -- Re-schedule if recurring
    IF v_rem.recurrence IS NOT NULL THEN
      INSERT INTO public.reminders(
        user_id, entity_type, entity_id, title, message,
        remind_at, timezone, recurrence, channels
      ) VALUES (
        v_rem.user_id, v_rem.entity_type, v_rem.entity_id, v_rem.title, v_rem.message,
        public.next_rrule_occurrence(v_rem.recurrence, v_rem.remind_at, v_rem.timezone),
        v_rem.timezone, v_rem.recurrence, v_rem.channels
      );
    END IF;
    v_count := v_count + 1;
  END LOOP;
  RETURN v_count;
END $$;

-- Hook cron every minute

COMMIT;
```

```tsx
// src/features/reminders/actions/snooze.ts
'use server'
import { z } from 'zod'
import { requireUser } from '@/lib/auth/require-user'
import { serviceSupabase } from '@/lib/supabase/service'

const schema = z.object({
  reminderId: z.string().uuid(),
  minutes:    z.number().int().min(1).max(10080),  // max 1 week
})

export async function snoozeReminderAction(input: unknown) {
  const { reminderId, minutes } = schema.parse(input)
  const user = await requireUser()
  const snoozedTo = new Date(Date.now() + minutes * 60_000).toISOString()

  const { data: r } = await serviceSupabase
    .from('reminders').select('snooze_count').eq('id', reminderId).single()
  if ((r?.snooze_count ?? 0) >= 5) throw new Error('max_snoozes_reached')

  await serviceSupabase
    .from('reminders')
    .update({
      status: 'snoozed',
      snoozed_to: snoozedTo,
      snooze_count: (r?.snooze_count ?? 0) + 1,
    })
    .eq('id', reminderId)
    .eq('user_id', user.id)

  return { snoozedTo }
}
```

### TypeScript: AI Review Summary (W20 Integration)

```tsx
// src/features/reviews/actions/generate-ai-summary.ts
'use server'
import { z } from 'zod'
import { requireUser } from '@/lib/auth/require-user'
import { runAIWithQuota } from '@/features/ai/run-with-quota'
import { serviceSupabase } from '@/lib/supabase/service'
import { buildPeriodContext } from '../lib/build-period-context'

const schema = z.object({ reviewId: z.string().uuid() })

export async function generateReviewAISummaryAction(input: unknown) {
  const { reviewId } = schema.parse(input)
  const user = await requireUser()

  const { data: review } = await serviceSupabase
    .from('reviews').select('*').eq('id', reviewId).eq('user_id', user.id).single()
  if (!review) throw new Error('review_not_found')

  // Pull period data: tasks, habits, expenses, mood logs, focus sessions
  const ctx = await buildPeriodContext(user.id, review.period_start, review.period_end)

  const prompt = `
أنت مستشار life-OS. حلل هذه الفترة وقدم:
1. ملخص (3 جمل)
2. أكبر 3 إنجازات
3. أكبر 2 تحديات
4. 3 نصائح للفترة القادمة
البيانات:
- Tasks: ${ctx.tasksDone}/${ctx.tasksTotal} (${ctx.tasksDonePct}%)
- Habits streak: ${ctx.habitsStreakAvg} days avg
- Mood avg: ${ctx.moodAvg}/10
- Energy avg: ${ctx.energyAvg}/10
- Focus minutes: ${ctx.focusMinutes}
- Top categories: ${ctx.topCategories.join(', ')}
الإجابات المباشرة من المستخدم:
${JSON.stringify(review.answers, null, 2)}
`

  const { text } = await runAIWithQuota(user.id, async (client) =>
    client.chat({ system: prompt, user: '', maxTokens: 700 })
  , { kind: 'review_summary', mode: 'pro' })

  await serviceSupabase
    .from('reviews')
    .update({ ai_summary: text, updated_at: new Date().toISOString() })
    .eq('id', reviewId)

  return { summary: text }
}
```
