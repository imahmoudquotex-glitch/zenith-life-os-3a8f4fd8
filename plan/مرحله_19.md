## 🧠 توسعة تنفيذية كاملة — Reviews & Life Score Operating System

مرحلة 19 لازم تبقى **نظام مراجعة وتحليل حياة كامل** مش مجرد جدول Reviews ورقم Life Score. الهدف إن كل يوم/أسبوع/شهر يتحول إلى snapshot قابل للتفسير وإعادة الحساب، مع أسئلة تأمل، توصيات هادئة، وقرارات تنفيذية مبنية على بيانات المستخدم الفعلية.

<aside>
⚠️

Life Score ليس أداة تقييم إنسان ولا ضغط نفسي. هو مؤشر خاص بالمستخدم فقط، قابل للإخفاء، قابل للحذف، وقابل للتفسير. أي مقارنة اجتماعية أو notification مُشعِرة بالذنب تعتبر blocker.

</aside>

### أهداف المرحلة الموسعة

- بناء Review Engine بأربع cadences: يومي، أسبوعي، شهري، ربع سنوي.
- بناء Life Score deterministic من 0 إلى 100 عبر 6 محاور.
- حفظ snapshot immutable عند إغلاق أي review.
- تفسير كل تغيير في النتيجة: "ليه زادت؟ ليه قلت؟".
- اكتشاف Energy Curve وFocus Windows بشكل تدريجي.
- ربط المراجعات بالمهام والعادات والأهداف والمال والتركيز بدون قراءة Vault.
- إنتاج read models سريعة للـ dashboard والـ AI لاحقاً.
- توفير export/delete كامل لتاريخ المراجعات.

CREATE OR REPLACE FUNCTION guard_closed_review_immutable()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
BEGIN
  IF OLD.closed_at IS NOT NULL THEN
    IF OLD.life_score IS DISTINCT FROM NEW.life_score
      OR OLD.axis_scores IS DISTINCT FROM NEW.axis_scores
      OR OLD.snapshot_json IS DISTINCT FROM NEW.snapshot_json
      OR OLD.snapshot_hash IS DISTINCT FROM NEW.snapshot_hash THEN
      RAISE EXCEPTION 'REVIEW_CLOSED_IMMUTABLE';
    END IF;
  END IF;
  RETURN NEW;
END $$;

CREATE TRIGGER trg_reviews_closed_immutable
BEFORE UPDATE ON reviews
FOR EACH ROW EXECUTE FUNCTION guard_closed_review_immutable();

## ⚙️ Review Service

```tsx
export class ReviewService {
  constructor(
    private reviews: ReviewRepository,
    private snapshots: ReviewSnapshotBuilder,
    private scores: LifeScoreService,
    private audit: AuditWriter,
  ) {}

  async open(ctx: RequestContext, cadence: ReviewCadence, period: ReviewPeriod) {
    return this.reviews.upsertOpen({
      id: ulid(),
      workspaceId: ctx.workspaceId,
      userId: ctx.userId,
      cadence,
      periodStart: period.start,
      periodEnd: period.end,
      timezone: ctx.timezone,
    })
  }

  async close(ctx: RequestContext, reviewId: string) {
    return db.tx(async (tx) => {
      const review = await this.reviews.getForUpdate(tx, reviewId, ctx)
      if (!review) throw new AppError('REVIEW_NOT_FOUND', 'Review not found', 404)
      if (review.closedAt) throw new AppError('REVIEW_ALREADY_CLOSED', 'Review already closed', 409)

      const snapshot = await this.snapshots.build(tx, ctx, review)
      const score = await this.scores.compute(ctx, snapshot)
      assertExplainableScore(score)

      const snapshotHash = sha256(JSON.stringify({ snapshot, score }))
      await this.reviews.close(tx, reviewId, {
        lifeScore: score.score,
        axisScores: score.axes,
        snapshotJson: snapshot,
        snapshotHash,
      })

      await this.audit.write(tx, {
        workspaceId: ctx.workspaceId,
        actorId: ctx.userId,
        action: 'review.closed',
        resourceId: reviewId,
        metadata: { cadence: review.cadence, score: score.score },
      })

      return { reviewId, score }
    })
  }
}
```

## 🧮 Life Score Engine مفصل

```tsx
export class LifeScoreService {
  async compute(ctx: RequestContext, snapshot: ReviewSnapshotInput): Promise<LifeScoreBreakdown> {
    const weights = await loadLifeScoreWeights(ctx)
    const axes = {
      health: await scoreHealth(snapshot),
      mind: await scoreMind(snapshot),
      wealth: await scoreWealth(snapshot),
      work: await scoreWork(snapshot),
      relations: await scoreRelations(snapshot),
      joy: await scoreJoy(snapshot),
    } satisfies Record<LifeAxis, AxisScore>

    const score = Math.round(
      LIFE_AXES.reduce((sum, axis) => sum + axes[axis].value * weights[axis], 0)
    )

    return {
      score: clamp(score, 0, 100),
      axes: Object.fromEntries(
        LIFE_AXES.map((axis) => [axis, { ...axes[axis], weight: weights[axis] }])
      ) as LifeScoreBreakdown['axes'],
      computedAt: new Date().toISOString(),
      timezone: ctx.timezone,
    }
  }
}
```

export async function scoreHealth(s: ReviewSnapshotInput): Promise<AxisScore> {
  const habitRatio = ratio(s.habits.completed, s.habits.expected)
  const energy = normalizeScale(s.energy.avg, 1, 10)
  const focus = normalizeMinutes(s.focus.minutes, 120)
  const value = Math.round((habitRatio * 0.5 + energy * 0.3 + focus * 0.2) * 100)
  return {
    value,
    weight: 0,
    reasons: [
      `${s.habits.completed}/${s.habits.expected} عادات مكتملة`,
      `متوسط الطاقة ${s.energy.avg ?? 'غير متاح'}`,
      `${s.focus.minutes} دقيقة تركيز`,
    ],
    dataFreshness: s.energy.samples > 0 ? 'fresh' : 'missing',
  }
}

export async function scoreWealth(s: ReviewSnapshotInput): Promise<AxisScore> {
  const income = s.money.incomeCents ?? 0
  const expense = s.money.expenseCents ?? 0
  const savings = income > 0 ? clamp((income - expense) / income, -1, 1) : 0
  const value = Math.round(clamp((savings + 1) / 2, 0, 1) * 100)
  return {
    value,
    weight: 0,
    reasons: [
      `الدخل: ${income} cents`,
      `المصروف: ${expense} cents`,
      `نسبة الادخار محسوبة بدون float للأموال`,
    ],
    dataFreshness: income || expense ? 'fresh' : 'missing',
  }
}

## 🧠 تطوير تنفيذي إضافي — Ethical Scoring Execution Pack

Life Score لازم يكون أداة وضوح، مش أداة ضغط نفسي. التنفيذ لازم يطلع score قابل للتفسير وإعادة الحساب بنفس النتائج.

### Score Breakdown Contract

```tsx
export type LifeScoreBreakdown = {
  score: number
  axes: Record<'health'|'mind'|'wealth'|'work'|'relations'|'joy', {
    value: number
    weight: number
    reasons: string[]
    dataFreshness: 'fresh' | 'stale' | 'missing'
  }>
  computedAt: string
  timezone: string
}

export function assertExplainableScore(b: LifeScoreBreakdown) {
  for (const axis of Object.values(b.axes)) {
    if (axis.reasons.length === 0) throw new Error('LIFE_SCORE_REASON_REQUIRED')
  }
}
```

export async function recomputeDailyLifeScore(ctx: ScoreContext) {

### UX Guardrails

- اعرض "ليه الرقم اتغير؟" بجانب كل score.
- زر إخفاء Life Score لا يمسح البيانات، فقط يخفي الـ widget.
- لا تعرض مقارنات مع مستخدمين آخرين افتراضيًا.
- أي notification عن انخفاض score تكون مساعدة وهادئة، وليس blame.
- **Explainable score:** كل Life Score لازم يعرض breakdown وسبب التغيير، بدون black-box shame أو addictive dark patterns.
- **Data sources:** score يقرأ من canonical Habits/Tasks/Goals/Focus/Mood فقط، مع tenant filters وno vault access.
- **Numeric rules:** energy/mood averages ممكن `NUMERIC` لأنها قياسات وليست money؛ أي money داخل reviews يظل `*_cents BIGINT`.
- **Recompute model:** recompute async/idempotent مع snapshot date/timezone. UI لا يكتب score النهائي مباشرة.
- **User control:** allow pause/hide score، export history، delete personal analytics، وavoid comparisons unless user explicitly enables.

-- 1711__reviews_immutability.sql
BEGIN;
CREATE OR REPLACE FUNCTION public.guard_review_closed() RETURNS TRIGGER LANGUAGE plpgsql AS $$
BEGIN
  IF OLD.closed_at IS NOT NULL AND (OLD.life_score IS DISTINCT FROM NEW.life_score
     OR OLD.axis_scores IS DISTINCT FROM NEW.axis_scores
     OR OLD.snapshot_json IS DISTINCT FROM NEW.snapshot_json) THEN
    RAISE EXCEPTION 'review_closed_immutable';
  END IF;
  RETURN NEW;
END $$;
CREATE TRIGGER trg_reviews_daily_immut BEFORE UPDATE ON reviews_daily FOR EACH ROW EXECUTE FUNCTION guard_review_closed();
COMMIT;

### Life Score Engine

```tsx
export const LIFE_SCORE_AXES = ['health','mind','wealth','work','relations','joy'] as const;
export async function recomputeLifeScore(db, ctx) {
  const [health, mind, wealth, work, relations, joy] = await Promise.all([
    healthScorer(db, ctx), mindScorer(db, ctx), wealthScorer(db, ctx),
    workScorer(db, ctx), relationsScorer(db, ctx), joyScorer(db, ctx),
  ]);
  const axisScores = { health, mind, wealth, work, relations, joy };
  const weights = await getUserWeights(db, ctx.userId);
  const score = Math.round(LIFE_SCORE_AXES.reduce((s, a) => s + axisScores[a] * weights[a], 0));
  await db.query(`INSERT INTO life_score_history (id, workspace_id, user_id, day_local, life_score, axis_scores, weights, source)
     VALUES ($1,$2,$3,$4,$5,$6,$7,'auto')`,
    [createUlid(), ctx.workspaceId, ctx.userId, ctx.day, score, axisScores, weights]);
  return { score, axisScores };
}
export async function healthScorer(db, ctx) {
  const h = (await db.query(`SELECT count(*) FILTER (WHERE c.id IS NOT NULL)::int AS done, count(h.id)::int AS total
     FROM habits h LEFT JOIN habit_checkins c ON c.habit_id = h.id AND c.day_local = $2
     WHERE h.user_id = $1 AND h.is_deleted = false`, [ctx.userId, ctx.day])).rows[0] ?? { done: 0, total: 0 };
  const e = (await db.query(`SELECT COALESCE(avg(level),5)::float AS avg FROM energy_samples
     WHERE user_id = $1 AND (sampled_at AT TIME ZONE 'UTC')::date = $2`, [ctx.userId, ctx.day])).rows[0]?.avg ?? 5;
  const habitPct = h.total === 0 ? 0.5 : h.done / h.total;
  return Math.round(((habitPct * 0.6) + ((e / 10) * 0.4)) * 100);
}
```

### Review Service

```tsx
export const reviewService = {
  async openDaily(db, ctx) {
    const id = createUlid();
    const { rows } = await db.query(
      `INSERT INTO reviews_daily (id, workspace_id, user_id, day_local) VALUES ($1,$2,$3,$4)
       ON CONFLICT (user_id, day_local) DO UPDATE SET updated_at = now() RETURNING *`,
      [id, ctx.workspaceId, ctx.userId, ctx.day]);
    await auditWriter.write({ workspaceId: ctx.workspaceId, actor:{type:'user',id:ctx.userId}, action:'review.opened', resourceType:'review_daily', resourceId: rows[0].id });
    return rows[0];
  },
  async closeDaily(db, ctx) {
    const r = (await db.query(`SELECT * FROM reviews_daily WHERE id=$1 AND user_id=$2`, [ctx.reviewId, ctx.userId])).rows[0];
    if (!r) throw new AppError('REVIEW_NOT_FOUND');
    if (r.closed_at) throw new AppError('REVIEW_ALREADY_CLOSED');
    const snapshot = await buildDailySnapshot(db, { userId: ctx.userId, day: r.day_local });
    const { score, axisScores } = await recomputeLifeScore(db, { userId: ctx.userId, workspaceId: ctx.workspaceId, day: r.day_local });
    await db.query(`UPDATE reviews_daily SET closed_at=now(), life_score=$2, axis_scores=$3, snapshot_json=$4 WHERE id=$1`,
      [ctx.reviewId, score, axisScores, snapshot]);
    return { reviewId: ctx.reviewId, score, axisScores };
  },
};
```

### Energy + Focus + Stickiness

```tsx
export function bucketOfHour(h) { return h < 11 ? 'morning' : h < 14 ? 'midday' : h < 18 ? 'afternoon' : 'evening'; }
export async function recordEnergy(db, ctx) {
  if (ctx.level < 1 || ctx.level > 10) throw new AppError('ENERGY_LEVEL_OUT_OF_RANGE');
  const bucket = bucketOfHour(new Date().getUTCHours());
  await db.query(`INSERT INTO energy_samples (id, workspace_id, user_id, sampled_at, bucket, level, source, note)
     VALUES ($1,$2,$3, now(), $4, $5, 'manual', $6)
     ON CONFLICT (user_id, ((sampled_at AT TIME ZONE 'UTC')::date), bucket) DO UPDATE SET level=EXCLUDED.level`,
    [createUlid(), ctx.workspaceId, ctx.userId, bucket, ctx.level, ctx.note ?? null]);
}
export async function snapshotStickiness(db, workspaceId, day) {
  const r = (await db.query(`WITH dau AS (SELECT count(DISTINCT actor_user_id)::int AS c FROM audit_events WHERE workspace_id=$1 AND created_at::date=$2 AND actor_type='user'),
          mau AS (SELECT count(DISTINCT actor_user_id)::int AS c FROM audit_events WHERE workspace_id=$1 AND created_at >= ($2::date - INTERVAL '27 days') AND actor_type='user')
     SELECT dau.c AS dau, mau.c AS mau FROM dau, mau`, [workspaceId, day])).rows[0];
  const dau = r?.dau ?? 0; const mau = r?.mau ?? 0;
  const ratio = mau === 0 ? 0 : (dau / mau) * 100;
  await db.query(`INSERT INTO stickiness_daily (id, workspace_id, day_local, dau, mau_28d, ratio_pct) VALUES ($1,$2,$3,$4,$5,$6)
     ON CONFLICT (workspace_id, day_local) DO UPDATE SET dau=EXCLUDED.dau, mau_28d=EXCLUDED.mau_28d, ratio_pct=EXCLUDED.ratio_pct`,
    [createUlid(), workspaceId, day, dau, mau, ratio.toFixed(2)]);
}
```
