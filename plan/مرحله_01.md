```tsx
// scripts/check-encryption-primitives.ts
import { readFileSync } from 'node:fs'
import { globSync } from 'glob'

const FORBIDDEN = [
  /\bAES-ECB\b/i, /\bMD5\b/i, /\bSHA-?1\b/i,
  /\b3DES\b/i, /\bRC4\b/i, /createHash\(['"]md5['"]\)/, /createHash\(['"]sha1['"]\)/,
  /createCipheriv\(['"]aes-\d+-ecb['"]/i,
]
const files = globSync('{apps,packages}/**/*.{ts,tsx,sql}', { ignore: ['**/node_modules/**'] })
for (const f of files) {
  const text = readFileSync(f, 'utf8')
  for (const re of FORBIDDEN) {
    if (re.test(text)) {
      console.error(`❌ Forbidden crypto primitive in ${f}: ${re.source}`)
      process.exit(1)
    }
  }
}
console.log('✅ crypto primitives clean')
```

### Required Repo Guards

```tsx
// scripts/check-tenant-boundaries.ts
import { readFileSync, readdirSync } from 'node:fs'
import { join } from 'node:path'

const MIGRATIONS_DIR = 'supabase/migrations'
const tenantTableRe = /CREATE TABLE(?: IF NOT EXISTS)? public\.([a-z_]+)/gi

for (const file of readdirSync(MIGRATIONS_DIR).filter(f => f.endsWith('.sql'))) {
  const sql = readFileSync(join(MIGRATIONS_DIR, file), 'utf8')
  for (const [, table] of sql.matchAll(tenantTableRe)) {
    if (['audit_events','feature_flags','webhook_nonces'].includes(table)) continue
    if (!new RegExp(`CREATE TABLE[^;]+${table}[^;]+workspace_id\\s+TEXT`, 'is').test(sql)) {
      throw new Error(`${file}: ${table} missing workspace_id TEXT`)
    }
    if (!new RegExp(`ALTER TABLE public\\.${table} ENABLE ROW LEVEL SECURITY`, 'i').test(sql)) {
      throw new Error(`${file}: ${table} missing RLS`)
    }
    if (!new RegExp(`ALTER TABLE public\\.${table} FORCE ROW LEVEL SECURITY`, 'i').test(sql)) {
      throw new Error(`${file}: ${table} missing FORCE RLS`)
    }
  }
}
```
